34/// @brief Scene entity transform.
35class Transform final {
36private:
37
38 glm::vec3 position_{};
39 glm::quat rotation_{glm::vec3{}}; // initializes zero rotation.
40 glm::vec3 scale_{1.0, 1.0, 1.0};
41
42public:
43
44 /// @brief Construct the transform.
45 constexpr Transform() = default;
46
47 /// @brief Transform position.
48 constexpr const glm::vec3& position() const noexcept {
49 return position_;
50 }
51
52 /// @brief Transform rotation.
53 /// @{
54 constexpr const glm::quat& rotation() const noexcept {
55 return rotation_;
56 }
57 glm::vec3 rotation_degrees() const noexcept {
58 return glm::degrees(glm::eulerAngles(rotation_));
59 }
60 /// @}
61
62 /// @brief Transform scale.
63 constexpr const glm::vec3& scale() const noexcept {
64 return scale_;
65 }
66
67 /// @brief Translate the transform.
68 constexpr void translate(const glm::vec3& delta) noexcept {
69 position_ += delta;
70 }
71
72 /// @brief Rotate the transform.
73 /// @{
74 constexpr void rotate(const glm::quat& delta) noexcept {
75 rotation_ *= delta;
76 }
77 constexpr void rotate_degrees(const glm::vec3& delta_degrees) noexcept {
78 rotation_ *= glm::quat{glm::radians(delta_degrees)};
79 }
80 /// @}
81
82 /// @brief Rescale the transform.
83 /// @{
84 constexpr void rescale(float factor) noexcept {
85 scale_ *= factor;
86 }
87 constexpr void rescale(const glm::vec3& factor) noexcept {
88 scale_ *= factor;
89 }
90 /// @}
91
92 /// @brief Transform model matrix.
93 /// @todo GLM's matmul operator is not constexpr?
94 glm::mat4 model_matrix() const noexcept {
95 auto model_matrix = glm::translate(glm::mat4(1.0f), position_) *
96 glm::toMat4(rotation_) *
97 glm::scale(glm::mat4(1.0f), scale_);
98 return model_matrix;
99 }

101}; // class Transform

105/// @brief Scene camera.
106class Camera final {
107private:
108
109 float near_ = 0.01f, far_ = 0.99f;
110 float orbit_ = 1.0f;
111 Transform transform_{};
112 glm::mat4 projection_matrix_{};
113
114public:
115
116 /// @brief Construct the camera.
117 constexpr Camera() = default;
118
119 /// @brief Construct the camera with @p parent.
120 constexpr explicit Camera(const Transform& parent) : transform_{parent} {}
121
122 /// @brief Camera transform.
123 /// @{
124 constexpr Transform& transform() noexcept {
125 return transform_;
126 }
127 constexpr const Transform& transform() const noexcept {
128 return transform_;
129 }
130 /// @}
131
132 /// @brief Camera orbit.
133 /// @{
134 constexpr float orbit() const noexcept {
135 return orbit_;
136 }
137 constexpr void set_orbit(float orbit) noexcept {
138 orbit_ = std::clamp(orbit, near_, far_);
139 }
140 /// @}
141
142 /// @brief Set perspective projection matrix.
143 void set_perspective(float aspect_ratio, float fov_degrees = 60.0f,
144 float the_near = 0.001f,
145 float the_far = 1000.0f) noexcept {
146 // Beware: GLM's documentations says that FOV is in degress,
147 // but actually it is in radians!
148 projection_matrix_ = glm::perspective(glm::radians(fov_degrees),
149 aspect_ratio, the_near, the_far);
150 near_ = the_near, far_ = the_far;
151 }
152
153 /// @brief Set orthographic projection matrix.
154 void set_ortographic(float aspect_ratio, float height = 1.0f,
155 float the_near = 0.001f,
156 float the_far = 1000.0f) noexcept {
157 const float width = aspect_ratio * height;
158 projection_matrix_ =
159 glm::ortho(-0.5f * width, +0.5f * width, //
160 -0.5f * height, +0.5f * height, the_near, the_far);
161 near_ = the_near, far_ = the_far;
162 }
163
164 /// @brief Camera view-projection matrix.
165 /// @todo GLM's matmul operator is not constexpr?
166 glm::mat4 view_projection_matrix() const noexcept {
167 const glm::mat4 view_matrix = glm::inverse(
168 transform_.model_matrix() *
169 glm::translate(glm::mat4(1.0f), glm::vec3(0.0f, 0.0f, orbit_)));
170 return projection_matrix_ * view_matrix;
171 }
172
173}; // class Camera
