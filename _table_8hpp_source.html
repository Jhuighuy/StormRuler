55/// @brief Copy the @p in_table to @p out_table.
56template<table InTable, table OutTable>
57void copy_table(const InTable& in_table, OutTable& out_table)
58{
59 if constexpr (std::assignable_from<OutTable, InTable>) {
60 out_table = in_table;
61 } else {
62 out_table.clear();
63 out_table.reserve(in_table.size());
64 for (auto row_index : in_table.rows()) {
65 out_table.push_back(in_table[row_index]);
66 }
67 }
68}
69/// @brief Move the @p in_table to @p out_table.
70template<table InTable, table OutTable>
71void move_table(InTable&& in_table, OutTable& out_table)
72{
73 if constexpr (std::assignable_from<OutTable, InTable>) {
74 out_table = std::move(in_table);
75 } else {
76 out_table.clear();
77 out_table.reserve(in_table.size());
78 for (auto row_index : in_table.rows()) {
79 out_table.push_back(std::move(in_table[row_index]));
80 }
81 in_table = {};
82 }
83}

85/// @brief CRTP interface to a table.
86template<crtp_derived Derived>
87class TableInterface
88{
89private:
90
91 constexpr Derived& self_() noexcept
92 {
93 static_assert(std::derived_from<Derived, TableInterface>);
94 return static_cast<Derived&>(*this);
95 }
96 constexpr const Derived& self_() const noexcept
97 {
98 return const_cast<TableInterface&>(*this).self_();
99 }
100
101public:
102
103 /// @brief Number of the table rows.
104 constexpr size_t size() const noexcept
105 {
106 return std::ranges::size(self_().rows());
107 }
108
109 /// @brief Assign the @p table.
110 /// @todo Refactor this in some point of time.
111 /// @{
112 void assign(Derived&& table)
113 {
114 self_() = std::move(table);
115 }
116 void assign(const Derived& table)
117 {
118 self_() = table;
119 }
120 template<table Table>
121 void assign(const Table& table)
122 {
123 self_().clear();
124 self_().reserve(table.size());
125 for (auto row_index : table.rows()) {
126 self_().push_back(table[row_index]);
127 }
128 }
129 /// @}
130
131}; // TableInterface

135/// @brief Compressed sparse row table.
136template<index RowIndex, class ColValue>
137 requires std::is_object_v<ColValue>
138class CsrTable final : public TableInterface<CsrTable<RowIndex, ColValue>>
139{
140private:
141
142 IndexedVector<RowIndex, OffsetIndex> row_offsets_{OffsetIndex{0}};
143 IndexedVector<OffsetIndex, ColValue> col_values_;
144
145public:
146
147 /// @brief Index range of the rows.
148 constexpr auto rows() const noexcept
149 {
150 return std::views::iota(RowIndex{0}, RowIndex{row_offsets_.size() - 1});
151 }
152
153 /// @brief Get the column indices range of a row @p row_index.
154 /// @{
155 constexpr auto operator[](RowIndex row_index) noexcept
156 {
157 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
158 // Here we do `{col_values_[begin - 1], &col_values_[end - 1] + 1}`,
159 // since `col_values_[end - 1]` may be outsize of the vector.
160 return std::span{&col_values_[row_offsets_[row_index]],
161 &col_values_[row_offsets_[row_index + 1] - 1] + 1};
162 }
163 constexpr auto operator[](RowIndex row_index) const noexcept
164 {
165 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
166 return std::span{&col_values_[row_offsets_[row_index]],
167 &col_values_[row_offsets_[row_index + 1] - 1] + 1};
168 }
169 /// @}
170
171 /// @brief Clear the table.
172 constexpr void clear()
173 {
174 row_offsets_.clear(), row_offsets_.emplace_back(0);
175 col_values_.clear();
176 }
177
178 /// @brief Reserve the row storage.
179 constexpr void reserve(size_t capacity)
180 {
181 row_offsets_.reserve(capacity + 1);
182 }
183
184 /// @brief Push back an empty row.
185 constexpr void push_back()
186 {
187 row_offsets_.emplace_back(col_values_.size());
188 }
189 /// @brief Push back a row with @p row_values.
190 template<std::ranges::input_range Range>
191 requires std::same_as<std::ranges::range_value_t<Range>, ColValue>
192 constexpr void push_back(Range&& row_values)
193 {
194 col_values_.insert(col_values_.end(), row_values.begin(), row_values.end());
195 row_offsets_.emplace_back(col_values_.size());
196 }
197
198 /// @brief Insert a connection at @p row_index, @p col_value.
199 /// @warning Complexity is linear!
200 void insert(RowIndex row_index, ColValue col_value)
201 {
202 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
203 const auto offset = static_cast<size_t>(row_offsets_[row_index + 1]);
204 col_values_.insert(col_values_.begin() + offset, col_value);
205 std::for_each(row_offsets_.begin() + static_cast<size_t>(row_index) + 1,
206 row_offsets_.end(), [](auto& offset) { offset += 1; });
207 }
208
209}; // class CsrTable

213/// @brief Vector-of-vectors sparse row table.
214template<index RowIndex, class ColValue = void>
215 requires std::is_object_v<ColValue>
216class VovTable final : public TableInterface<VovTable<RowIndex, ColValue>>
217{
218private:
219
220 IndexedVector<RowIndex, IndexedVector<OffsetIndex, ColValue>> rows_;
221
222public:
223
224 /// @brief Index range of the rows.
225 constexpr auto rows() const noexcept
226 {
227 return std::views::iota(RowIndex{0}, RowIndex{rows_.size()});
228 }
229
230 /// @brief Get the column indices range of a row @p row_index.
231 /// @{
232 constexpr auto& operator[](RowIndex row_index) noexcept
233 {
234 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
235 return rows_[row_index];
236 }
237 constexpr const auto& operator[](RowIndex row_index) const noexcept
238 {
239 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
240 return rows_[row_index];
241 }
242 /// @}
243
244 /// @brief Clear the table.
245 constexpr void clear()
246 {
247 rows_.clear();
248 }
249
250 /// @brief Reserve the row storage.
251 constexpr void reserve(size_t capacity)
252 {
253 rows_.reserve(capacity);
254 }
255
256 /// @brief Push back an empty row.
257 constexpr void push_back()
258 {
259 rows_.emplace_back();
260 }
261 /// @brief Push back a row with @p row_col_indices.
262 /// @{
263 template<std::ranges::input_range Range>
264 requires std::same_as<std::ranges::range_value_t<Range>, ColValue>
265 constexpr void push_back(Range&& row_values)
266 {
267 rows_.emplace_back();
268 std::ranges::copy(row_values, std::back_inserter(rows_.back()));
269 }
270 constexpr void push_back(IndexedVector<OffsetIndex, ColValue>&& row_values)
271 {
272 rows_.emplace_back(std::move(row_values));
273 }
274 /// @}
275
276 /// @brief Insert a connection at @p row_index, @p col_value.
277 void insert(RowIndex row_index, ColValue col_value)
278 {
279 STORM_ASSERT_(row_index < this->size(), "Row index is out of range!");
280 rows_[row_index].push_back(col_value);
281 }
282
283}; // class VovTable
