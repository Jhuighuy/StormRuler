58template<class Func, matrix Matrix, matrix... RestMatrices>
59 requires compatible_matrices_v<Matrix, RestMatrices...> &&
60 std::invocable<Func, //
61 matrix_element_ref_t<Matrix>,
62 matrix_element_ref_t<RestMatrices>...>
63constexpr void matrix_for_each(Func func, //
64 Matrix&& mat,
65 RestMatrices&&... rest_mats) noexcept {
66 STORM_ASSERT((mat.shape() == rest_mats.shape()) && ...,
67 "Matrix shapes do not match!");
68 if constexpr (matrix_r<Matrix, 1>) {
69 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
70 func(mat(row_index), rest_mats(row_index)...);
71 }
72 } else {
73 /// @todo Below:
74 static_assert(matrix_r<Matrix, 2>, "General rank is not implemented yet!");
75 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
76 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
77 func(mat(row_index, col_index), rest_mats(row_index, col_index)...);
78 }
79 }
80 }
81}

85/// @brief Move the matrix @p mat elements into the matrix @p out_mat.
86template<output_matrix OutMatrix, matrix Matrix>
87 requires assignable_matrix<OutMatrix, Matrix>
88constexpr OutMatrix&& move_elements(OutMatrix&& out_mat,
89 Matrix&& mat) noexcept {
90 matrix_for_each(MoveAssign{}, out_mat, std::forward<Matrix>(mat));
91 return std::forward<OutMatrix>(out_mat);
92}

94/// @brief Copy the matrix @p mat elements to the matrix @p out_mat.
95template<output_matrix OutMatrix, matrix Matrix>

103/// @brief Evaluate the matrix @p out_mat elements with
104/// action function @p action_func and arguments @p mats.
105template<class ActionFunc, output_matrix OutMatrix, matrix... Matrices>
106 requires compatible_matrices_v<OutMatrix, Matrices...> &&
107 std::regular_invocable<ActionFunc, //
108 matrix_element_ref_t<OutMatrix>,
109 matrix_element_ref_t<Matrices>...>
110constexpr OutMatrix&& eval_elements(ActionFunc action_func, //
111 OutMatrix&& out_mat,
112 Matrices&&... mats) noexcept {
113 matrix_for_each(std::move(action_func), //
114 out_mat, std::forward<Matrices>(mats)...);
115 return std::forward<OutMatrix>(out_mat);
116}

137/// @brief Fill the matrix @p out_mat elements with
138/// the uniformly-distributed random numbers.
139/// @note This is a sequential operation!
140template<output_matrix Matrix>
141STORM_CPP23_CONSTEXPR Matrix&
142fill_randomly(Matrix&& out_mat, //
143 matrix_element_t<Matrix> min = 0,
144 matrix_element_t<Matrix> max = 1) noexcept {
145 static std::mt19937_64 random_engine{}; // NOSONAR
146 std::uniform_real_distribution distribution{min, max};
147 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
148 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
149 out_mat(row_index, col_index) = distribution(random_engine);
150 }
151 }
152 return out_mat;
153}

157/// @brief Reduce the matrix @p mat coefficients to a single value.
158/// @param init Initial reduction value.
159/// @param reduce_func Reduction function.
160/// @todo Restrictions!
161/// @{
162template<class Value, class ReduceFunc, matrix_r<1> Matrix>
163constexpr auto reduce(Value init, ReduceFunc reduce_func, Matrix&& mat) {
164 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
165 init = reduce_func(init, mat(row_index));
166 }
167 return init;
168}
169template<class Value, class ReduceFunc, matrix_r<2> Matrix>
170constexpr auto reduce(Value init, ReduceFunc reduce_func, Matrix&& mat) {
171 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
172 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
173 init = reduce_func(init, mat(row_index, col_index));
174 }
175 }
176 return init;
177}

273/// @brief Element-wise matrix @p mat @f$ L_{p} @f$-norm.
274/// @{
275template<matrix Matrix, scalar Number>
276constexpr auto norm_p_p(Matrix&& mat, Number p) {
277 STORM_ASSERT(p > 0, "Invalid p-norm parameter!");
278 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
279 return reduce(Result{}, Add{}, Compose{Abs{}, BindLast{Pow{}, std::move(p)}},
280 std::forward<Matrix>(mat));
281}
282template<matrix Matrix, scalar Number>
283constexpr auto norm_p(Matrix&& mat, Number p) {
284 /// @todo We should have something better than `1.0 / p`.
285 return pow(norm_p_p(std::forward<Matrix>(mat), p), 1.0 / p);
286}
287/// @}
288
289/// @brief Element-wise matrix @p mat @f$ L_{\infty} @f$-norm.
290template<matrix Matrix>
291constexpr auto norm_inf(Matrix&& mat) {
292 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
293 return reduce(Result{}, Max{}, Abs{}, std::forward<Matrix>(mat));
294}
