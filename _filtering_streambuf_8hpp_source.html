34/// @brief Filtering stream buffer.
35/// @tparam BeginFilter First filtering character, typically '#'.
36/// @tparam EndFilter Last filtering character, typically '\n'.
37template<class Char, Char BeginFilter, Char EndFilter>
38class FilteringStreambuf final :
39 public std::basic_streambuf<Char>,
40 public NonCopyable {
41private:
42
43 std::istream* p_stream_;
44 std::streambuf* p_streambuf_;
45 Char buffer_;
46
47 static constexpr auto begin_filter_ =
48 std::basic_streambuf<Char>::traits_type::to_int_type(BeginFilter);
49 static constexpr auto end_filter_ =
50 std::basic_streambuf<Char>::traits_type::to_int_type(EndFilter);
51 static constexpr auto eof_ = std::basic_streambuf<Char>::traits_type::eof();
52
53public:
54
55 /// @brief Construct a filtering stream buffer.
56 explicit FilteringStreambuf(std::basic_istream<Char>& stream)
57 : p_stream_{&stream}, p_streambuf_{p_stream_->rdbuf()} {
58 p_stream_->rdbuf(this);
59 }
60
61 /// @brief Destroy a filtering stream buffer.
62 ~FilteringStreambuf() override {
63 p_stream_->rdbuf(p_streambuf_);
64 }
65
66protected:
67
68 typename std::basic_streambuf<Char>::int_type underflow() override {
69 auto bumped = p_streambuf_->sbumpc();
70 if (bumped == begin_filter_) {
71 while (bumped != eof_ && bumped != end_filter_) {
72 bumped = p_streambuf_->sbumpc();
73 }
74 }
75 if (bumped != eof_) {
76 buffer_ = static_cast<Char>(bumped);
77 this->setg(&buffer_, &buffer_, &buffer_ + 1);
78 }
79 return bumped;
80 }
81
82}; // class FilteringStreambuf
