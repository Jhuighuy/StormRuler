42/// @brief Bind the first arguments of the function to values.
43template<class Func, class... FirstArgs>
44 requires std::is_object_v<Func>
45class BindFirst final
46{
47private:
48
49 STORM_NO_UNIQUE_ADDRESS_ Func func_;
50 STORM_NO_UNIQUE_ADDRESS_ std::tuple<FirstArgs...> first_args_;
51
52public:
53
54 constexpr explicit BindFirst(Func func, FirstArgs... first_args)
55 : func_{std::move(func)}, first_args_{std::move(first_args)...}
56 {
57 }
58
59 template<class... RestArgs>
60 requires std::regular_invocable<Func, FirstArgs..., RestArgs...>
61 constexpr decltype(auto) operator()(RestArgs&&... rest_args) const noexcept
62 {
63 auto call = [this, &rest_args...](const FirstArgs&... first_args) {
64 return func_(first_args..., std::forward<RestArgs>(rest_args)...);
65 };
66 return std::apply(call, first_args_);
67 }
68
69}; // class BindFirst

100/// @brief Compose multiple functions: @f$ f_{1} \circ f_{2} \circ \ldots @f$.
101template<class Func, class... RestFuncs>
102 requires std::is_object_v<Func> && (... && std::is_object_v<RestFuncs>)
103class Compose final
104{
105private:
106
107 STORM_NO_UNIQUE_ADDRESS_ Func func_;
108 STORM_NO_UNIQUE_ADDRESS_ Compose<RestFuncs...> rest_funcs_;
109
110public:
111
112 constexpr Compose(Func func, RestFuncs... rest_funcs)
113 : func_{std::move(func)}, rest_funcs_{std::move(rest_funcs)...}
114 {
115 }
116
117 template<class... Args>
118 requires std::regular_invocable<Compose<RestFuncs...>, Args...> &&
119 std::regular_invocable<
120 Func, std::invoke_result_t<Compose<RestFuncs...>, Args...>>
121 constexpr decltype(auto) operator()(Args&&... args) const noexcept
122 {
123 return func_(rest_funcs_(std::forward<Args>(args)...));
124 }
125
126}; // class Compose

129template<class Func1, class Func2>

132{

134

137
138public:
139
140 constexpr Compose(Func1 func1, Func2 func2)
141 : func1_{std::move(func1)}, func2_{std::move(func2)}
142 {
143 }
144
145 template<class... Args>
146 requires std::regular_invocable<Func2, Args...> && //
147 std::regular_invocable<Func1, std::invoke_result_t<Func2, Args...>>
148 constexpr decltype(auto) operator()(Args&&... args) const noexcept
149 {
150 return func1_(func2_(std::forward<Args>(args)...));
151 }

229/// @brief Approximately equals function.

231{
232private:

234 long double tolerance_;
235
236public:
237
238 constexpr explicit ApproxEqual(
239 long double tolerance = sqrt(std::numeric_limits<long double>::epsilon()))
240 : tolerance_{tolerance}
241 {
242 STORM_ASSERT_(tolerance > 0.0l, "Negative tolerance!");
243 }

245 template<class Arg1, class Arg2>

247 {
248 decltype(auto) delta =
249 abs(std::forward<Arg1>(arg1) - std::forward<Arg2>(arg2));
250 return static_cast<long double>(delta) <= tolerance_;
251 }
