36/// @brief Least Squares gradient estimation scheme, cell-based:
37/// computes cell-centered gradients based on the cell-centered values.
38template<mesh Mesh>
39class LeastSquaresGradientScheme final
40{
41private:
42
43 const Mesh* p_mesh_;
44 CellMatrixField<Mesh, real_t> g_mats_;
45
46public:
47
48 /// @brief Construct the gradient scheme.
49 constexpr explicit LeastSquaresGradientScheme(const Mesh& mesh)
50 : p_mesh_{&mesh}, g_mats_(p_mesh_->num_cells())
51 {
52 // Compute the inverse least squares matrices.
53 std::ranges::for_each(p_mesh_->interior_cells(), [&](CellView<Mesh> cell) {
54 // Compute the direct least squares matrices.
55 g_mats_[cell][0] = {};
56 cell.for_each_cell([&](CellView<Mesh> adj_cell) {
57 const auto dr = adj_cell.center() - cell.center();
58 g_mats_[cell][0] += glm::outerProduct(dr, dr);
59 });
60
61 // Compute the inverse.
62 static const mat2_t eps(1e-14);
63 g_mats_[cell][0] = glm::inverse(eps + g_mats_[cell][0]);
64 });
65 }
66
67 /// @brief Compute cell-centered gradients.
68 template<class Real, size_t NumVars>
69 void operator()(CellVectorField<Mesh, Real, NumVars>& grad_u,
70 const CellField<Mesh, Real, NumVars>& u) const noexcept
71 {
72 std::ranges::for_each(p_mesh_->interior_cells(), [&](CellView<Mesh> cell) {
73 // Compute the least squares RHS.
74 grad_u[cell].fill({});
75 cell.for_each_cell([&](CellView<Mesh> adj_cell) {
76 const auto dr = adj_cell.center() - cell.center();
77 for (size_t i = 0; i < NumVars; ++i) {
78 grad_u[cell][i] += (u[adj_cell][i] - u[cell][i]) * dr;
79 }
80 });
81
82 // Compute the least squares gradients.
83 for (size_t i = 0; i < NumVars; ++i) {
84 grad_u[cell][i] = g_mats_[cell][0] * grad_u[cell][i];
85 }
86 });
87 }
88
89}; // class LeastSquaresGradientScheme
