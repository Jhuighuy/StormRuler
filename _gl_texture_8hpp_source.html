51/// @brief OpenGL default pixel description.
52/// Copied straightly from the OpenGL ES 3.0's `glTexImage2D`
53/// sized internal format table.
54template<class Pixel>
55inline constexpr auto pixel_desc_v = []() -> pixel_desc_t {
56 if constexpr (std::same_as<Pixel, GLbyte>)
57 return {GL_R8I, GL_RED_INTEGER, GL_BYTE};
58 if constexpr (std::same_as<Pixel, GLubyte>)
59 return {GL_R8UI, GL_RED_INTEGER, GL_UNSIGNED_BYTE};
60 if constexpr (std::same_as<Pixel, GLshort>)
61 return {GL_R16I, GL_RED_INTEGER, GL_SHORT};
62 if constexpr (std::same_as<Pixel, GLushort>)
63 return {GL_R16UI, GL_RED_INTEGER, GL_UNSIGNED_SHORT};
64 if constexpr (std::same_as<Pixel, GLint>)
65 return {GL_R32I, GL_RED_INTEGER, GL_INT};
66 if constexpr (std::same_as<Pixel, GLuint>)
67 return {GL_R32UI, GL_RED_INTEGER, GL_UNSIGNED_INT};
68 if constexpr (std::same_as<Pixel, GLfloat>)
69 return {GL_R32F, GL_RED, GL_FLOAT};
70
71 if constexpr (std::same_as<Pixel, glm::tvec2<GLbyte, glm::packed>>)
72 return {GL_RG8I, GL_RG_INTEGER, GL_BYTE};
73 if constexpr (std::same_as<Pixel, glm::tvec2<GLubyte, glm::packed>>)
74 return {GL_RG8UI, GL_RG_INTEGER, GL_UNSIGNED_BYTE};
75 if constexpr (std::same_as<Pixel, glm::tvec2<GLshort, glm::packed>>)
76 return {GL_RG16I, GL_RG_INTEGER, GL_SHORT};
77 if constexpr (std::same_as<Pixel, glm::tvec2<GLushort, glm::packed>>)
78 return {GL_RG16UI, GL_RG_INTEGER, GL_UNSIGNED_SHORT};
79 if constexpr (std::same_as<Pixel, glm::tvec2<GLint, glm::packed>>)
80 return {GL_RG32I, GL_RG_INTEGER, GL_INT};
81 if constexpr (std::same_as<Pixel, glm::tvec2<GLuint, glm::packed>>)
82 return {GL_RG32UI, GL_RG_INTEGER, GL_UNSIGNED_INT};
83 if constexpr (std::same_as<Pixel, glm::tvec2<GLfloat, glm::packed>>)
84 return {GL_RG32F, GL_RG, GL_FLOAT};
85
86 if constexpr (std::same_as<Pixel, glm::tvec3<GLbyte, glm::packed>>)
87 return {GL_RGB8I, GL_RGB_INTEGER, GL_BYTE};
88 if constexpr (std::same_as<Pixel, glm::tvec3<GLubyte, glm::packed>>)
89 return {GL_RGB8UI, GL_RGB_INTEGER, GL_UNSIGNED_BYTE};
90 if constexpr (std::same_as<Pixel, glm::tvec3<GLshort, glm::packed>>)
91 return {GL_RGB16I, GL_RGB_INTEGER, GL_SHORT};
92 if constexpr (std::same_as<Pixel, glm::tvec3<GLushort, glm::packed>>)
93 return {GL_RGB16UI, GL_RGB_INTEGER, GL_UNSIGNED_SHORT};
94 if constexpr (std::same_as<Pixel, glm::tvec3<GLint, glm::packed>>)
95 return {GL_RGB32I, GL_RGB_INTEGER, GL_INT};
96 if constexpr (std::same_as<Pixel, glm::tvec3<GLuint, glm::packed>>)
97 return {GL_RGB32UI, GL_RGB_INTEGER, GL_UNSIGNED_INT};
98 if constexpr (std::same_as<Pixel, glm::tvec3<GLfloat, glm::packed>>)
99 return {GL_RGB32F, GL_RGB, GL_FLOAT};
100
101 if constexpr (std::same_as<Pixel, glm::tvec4<GLbyte, glm::packed>>)
102 return {GL_RGBA8I, GL_RGBA_INTEGER, GL_BYTE};
103 if constexpr (std::same_as<Pixel, glm::tvec4<GLubyte, glm::packed>>)
104 return {GL_RGBA8UI, GL_RGBA_INTEGER, GL_UNSIGNED_BYTE};
105 if constexpr (std::same_as<Pixel, glm::tvec4<GLshort, glm::packed>>)
106 return {GL_RGBA16I, GL_RGBA_INTEGER, GL_SHORT};
107 if constexpr (std::same_as<Pixel, glm::tvec4<GLushort, glm::packed>>)
108 return {GL_RGBA16UI, GL_RGBA_INTEGER, GL_UNSIGNED_SHORT};
109 if constexpr (std::same_as<Pixel, glm::tvec4<GLint, glm::packed>>)
110 return {GL_RGBA32I, GL_RGBA_INTEGER, GL_INT};
111 if constexpr (std::same_as<Pixel, glm::tvec4<GLuint, glm::packed>>)
112 return {GL_RGBA32UI, GL_RGBA_INTEGER, GL_UNSIGNED_INT};
113 if constexpr (std::same_as<Pixel, glm::tvec4<GLfloat, glm::packed>>)
114 return {GL_RGBA32F, GL_RGBA, GL_FLOAT};
115
116 return {};
117}();

125/// @brief OpenGL texture.
126class Texture : detail_::noncopyable_ {
127private:
128
129 GLuint texture_id_;
130
131public:
132
133 /// @brief Construct a texture.
134 Texture() {
135 glGenTextures(1, &texture_id_);
136 }
137
138 /// @brief Move-construct a texture.
139 Texture(Texture&&) = default;
140 /// @brief Move-assign the texture.
141 Texture& operator=(Texture&&) = default;
142
143 /// @brief Destruct the texture.
144 ~Texture() {
145 glDeleteTextures(1, &texture_id_);
146 }
147
148 /// @brief Cast to texture ID.
149 constexpr operator GLuint() const noexcept {
150 return texture_id_;
151 }
152
153protected:
154
155 void bind_(TextureTarget target) const {
156 glBindTexture(static_cast<GLenum>(target), texture_id_);
157 }
158 void bind_(TextureTarget target, size_t slot) const {
159 glActiveTexture(static_cast<GLenum>(GL_TEXTURE0 + slot));
160 bind_(target);
161 }
162
163}; // class Texture

167/// @brief OpenGL 2D texture.
168template<pixel Pixel>
169class Texture2D final : public Texture {
170public:
171
172 /// @brief Construct a 2D texture.
173 Texture2D() = default;
174
175 /// @brief Consruct a 2D texture with @p width, @p height and @p data.
176 Texture2D(GLsizei width, GLsizei height, const Pixel* data = nullptr) {
177 assign(width, height);
178 }
179
180 /// @brief Bind the 2D texture buffer to @p slot.
181 void bind(size_t slot) const {
182 bind_(TextureTarget::texture2D, slot);
183 }
184
185 /// @brief Assign the 2D texture @p width, @p height and @p data.
186 void assign(GLsizei width, GLsizei height, const Pixel* data = nullptr) {
187 bind_(TextureTarget::texture2D);
188 glTexImage2D(GL_TEXTURE_2D,
189 /*level*/ 0, pixel_desc_v<Pixel>.internal_format, //
190 width, height,
191 /*border*/ 0, pixel_desc_v<Pixel>.format,
192 pixel_desc_v<Pixel>.type, data);
193 /// @todo Sampling properties!
194 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
195 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
196 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP);
197 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP);
198 }
199
200 /// @brief Read the 2D texture into the pixel buffer object.
201 /// @see https://riptutorial.com/opengl/example/28872/using-pbos
202 void read_pixels(Buffer<Pixel>& buffer) {
203 buffer.bind(BufferTarget::pixel_pack_buffer);
204 bind_(TextureTarget::texture2D);
205 glGetTexImage(GL_TEXTURE_2D, /*level*/ 0, pixel_desc_v<Pixel>.format,
206 pixel_desc_v<Pixel>.type, /*data*/ nullptr);
207 }
208
209}; // class Texture2D
