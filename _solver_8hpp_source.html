59/// -=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=- ///
60/// @brief Abstract operator equation iterative solver.
61/// -=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=- ///
62template<legacy_vector_like InVector, legacy_vector_like OutVector = InVector>
63class IterativeSolver : public Solver<InVector, OutVector> {
64public:
65
66 size_t iteration{0};
67 size_t num_iterations{2000};
68 real_t absolute_error{0.0};
69 real_t relative_error{0.0};
70
71 real_t absolute_error_tolerance{1.0e-6};
72 real_t relative_error_tolerance{1.0e-6};
73
74 PreconditionerSide pre_side{PreconditionerSide::Right};
75 std::unique_ptr<Preconditioner<InVector>> pre_op{nullptr};
76 std::string name;
77
78protected:
79
80 /// @brief Initialize the iterative solver.
81 ///
82 /// @param x_vec Initial guess for the solution vector, 𝒙.
83 /// @param b_vec Right-hand-side vector, 𝒃.
84 /// @param any_op Equation operator, 𝓐(𝒙).
85 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
86 ///
87 /// @returns Residual norm of the initial guess, ‖𝒃 - 𝓐(𝒙)‖.
88 virtual real_t init(const InVector& x_vec, const OutVector& b_vec,
89 const Operator<InVector, OutVector>& any_op,
90 const Preconditioner<InVector>* pre_op) = 0;
91
92 /// @brief Iterate the solver.
93 ///
94 /// @param x_vec Solution vector, 𝒙.
95 /// @param b_vec Right-hand-side vector, 𝒃.
96 /// @param any_op Equation operator, 𝓐(𝒙).
97 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
98 ///
99 /// @returns Residual norm, ‖𝒃 - 𝓐(𝒙)‖.
100 virtual real_t iterate(InVector& x_vec, const OutVector& b_vec,
101 const Operator<InVector, OutVector>& any_op,
102 const Preconditioner<InVector>* pre_op) = 0;
103
104 /// @brief Finalize the iterations.
105 ///
106 /// @param x_vec Solution vector, 𝒙.
107 /// @param b_vec Right-hand-side vector, 𝒃.
108 /// @param any_op Equation operator, 𝓐(𝒙).
109 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
110 virtual void finalize(InVector& x_vec, const OutVector& b_vec,
111 const Operator<InVector, OutVector>& any_op,
112 const Preconditioner<InVector>* pre_op) {}
113
114public:
115
116 bool solve(InVector& x_vec, const OutVector& b_vec,
117 const Operator<InVector, OutVector>& any_op) final {
118 // Initialize the solver.
119 if (pre_op != nullptr) {
120 pre_op->build(x_vec, b_vec, any_op);
121 }
122 const real_t initial_error = init(x_vec, b_vec, any_op, pre_op.get());
123 absolute_error = initial_error;
124 if (absolute_error_tolerance > 0.0 &&
125 absolute_error < absolute_error_tolerance) {
126 finalize(x_vec, b_vec, any_op, pre_op.get());
127 return true;
128 }
129
130 // Iterate the solver:
131 bool converged = false;
132 for (iteration = 0; !converged && (iteration < num_iterations);
133 ++iteration) {
134 absolute_error = iterate(x_vec, b_vec, any_op, pre_op.get());
135 relative_error = absolute_error / initial_error;
136 converged |= (absolute_error_tolerance > 0.0) &&
137 (absolute_error < absolute_error_tolerance);
138 converged |= (relative_error_tolerance > 0.0) &&
139 (relative_error < relative_error_tolerance);
140 }
141
142 // Exit the solver.
143 finalize(x_vec, b_vec, any_op, pre_op.get());
144 STORM_INFO_("n_iter: {:>4d}, abs_err: {:>-12e}, rel_err: {:>-12e}", //
145 iteration, absolute_error, relative_error);
146 return converged;
147 }
148
149}; // class IterativeSolver

151/// -=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=- ///
152/// @brief Abstract inner-outer iterative solver.
153/// -=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=- ///
154template<legacy_vector_like InVector, legacy_vector_like OutVector = InVector>
155class InnerOuterIterativeSolver : public IterativeSolver<InVector, OutVector> {
156public:
157
158 size_t inner_iteration{0};
159 size_t num_inner_iterations{50};
160
161protected:
162
163 /// @brief Initialize the outer iterations.
164 ///
165 /// This function is used invoked only once,
166 /// in the initialization phase.
167 ///
168 /// @param x_vec Initial guess for the solution vector, 𝒙.
169 /// @param b_vec Right-hand-side vector, 𝒃.
170 /// @param any_op Equation operator, 𝓐(𝒙).
171 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
172 ///
173 /// @returns Residual norm of the initial guess, ‖𝒃 - 𝓐(𝒙)‖.
174 virtual real_t outer_init(const InVector& x_vec, const OutVector& b_vec,
175 const Operator<InVector, OutVector>& any_op,
176 const Preconditioner<InVector>* pre_op) = 0;
177
178 /// @brief Initialize the inner iterations.
179 ///
180 /// This function is invoked before the each inner iteration loop.
181 ///
182 /// @param x_vec Solution vector, 𝒙.
183 /// @param b_vec Right-hand-side vector, 𝒃.
184 /// @param any_op Equation operator, 𝓐(𝒙).
185 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
186 virtual void inner_init(const InVector& x_vec, const OutVector& b_vec,
187 const Operator<InVector, OutVector>& any_op,
188 const Preconditioner<InVector>* pre_op) {}
189
190 /// @brief Perform the inner iteration.
191 ///
192 /// @param x_vec Solution vector, 𝒙.
193 /// @param b_vec Right-hand-side vector, 𝒃.
194 /// @param any_op Equation operator, 𝓐(𝒙).
195 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
196 ///
197 /// @returns Residual norm, ‖𝒃 - 𝓐(𝒙)‖.
198 virtual real_t inner_iterate(InVector& x_vec, const OutVector& b_vec,
199 const Operator<InVector, OutVector>& any_op,
200 const Preconditioner<InVector>* pre_op) = 0;
201
202 /// @brief Finalize the inner iterations.
203 ///
204 /// This function is called in order to finalize
205 /// the inner iterations or if some stopping criterion is met.
206 ///
207 /// @param x_vec Solution vector, 𝒙.
208 /// @param b_vec Right-hand-side vector, 𝒃.
209 /// @param any_op Equation operator, 𝓐(𝒙).
210 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
211 virtual void inner_finalize(InVector& x_vec, const OutVector& b_vec,
212 const Operator<InVector, OutVector>& any_op,
213 const Preconditioner<InVector>* pre_op) {}
214
215 /// @brief Finalize the outer iterations.
216 ///
217 /// This function is used invoked only once,
218 /// when some stopping criterion is met.
219 ///
220 /// @param x_vec Solution vector, 𝒙.
221 /// @param b_vec Right-hand-side vector, 𝒃.
222 /// @param any_op Equation operator, 𝓐(𝒙).
223 /// @param pre_op Preconditioner operator, 𝓟(𝒙).
224 virtual void outer_finalize(InVector& x_vec, const OutVector& b_vec,
225 const Operator<InVector, OutVector>& any_op,
226 const Preconditioner<InVector>* pre_op) {}
227
228private:
229
230 real_t init(const InVector& x_vec, const OutVector& b_vec,
231 const Operator<InVector, OutVector>& any_op,
232 const Preconditioner<InVector>* pre_op) final {
233 return outer_init(x_vec, b_vec, any_op, pre_op);
234 }
235
236 real_t iterate(InVector& x_vec, const OutVector& b_vec,
237 const Operator<InVector, OutVector>& any_op,
238 const Preconditioner<InVector>* pre_op) final {
239 inner_iteration = this->iteration % num_inner_iterations;
240 if (inner_iteration == 0) {
241 inner_init(x_vec, b_vec, any_op, pre_op);
242 }
243 const real_t residual_norm{inner_iterate(x_vec, b_vec, any_op, pre_op)};
244 if (inner_iteration == num_inner_iterations - 1) {
245 inner_finalize(x_vec, b_vec, any_op, pre_op);
246 }
247 return residual_norm;
248 }
249
250 void finalize(InVector& x_vec, const OutVector& b_vec,
251 const Operator<InVector, OutVector>& any_op,
252 const Preconditioner<InVector>* pre_op) final {
253 if (inner_iteration != num_inner_iterations - 1) {
254 inner_finalize(x_vec, b_vec, any_op, pre_op);
255 }
256 outer_finalize(x_vec, b_vec, any_op, pre_op);
257 }
258
259}; // class InnerOuterIterativeSolver

267/// ----------------------------------------------------------------- ///
268/// @brief Solve an operator equation 𝓐(𝒙) = 𝒃,
269/// when 𝓐(𝒙) is a non-uniform operator (𝓐(𝟢) ≠ 𝟢).
270/// ----------------------------------------------------------------- ///
271template<legacy_vector_like Vector>
272bool solve_non_uniform(Solver<Vector>& solver, Vector& x_vec,
273 const Vector& b_vec, const Operator<Vector>& any_op) {
274 Vector z_vec, f_vec;
275
276 z_vec.assign(x_vec, false);
277 f_vec.assign(b_vec, false);
278
279 // Solve an equation with the "uniformed" operator:
280 // 𝓐(𝒙) - 𝓐(𝟢) = 𝒃 - 𝓐(𝟢).
281 fill_with(f_vec, 0.0);
282 any_op.mul(z_vec, f_vec);
283 f_vec <<= b_vec - z_vec;
284
285 const auto uni_op =
286 make_operator<Vector>([&](Vector& y_vec, const Vector& x_vec) {
287 any_op.mul(y_vec, x_vec);
288 y_vec -= z_vec;
289 });
290
291 return solver.solve(x_vec, f_vec, *uni_op);
292}
