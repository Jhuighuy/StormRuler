48/// @brief A semi-valid argument for the CRTP classes.
49template<class Derived>
50concept crtp_derived = std::is_class_v<Derived> &&
51 std::same_as<Derived, std::remove_cv_t<Derived>>;
52
53namespace detail {
54 // Based on /*is-derived-from-view-interface*/<T> implementation.
55 template<template<crtp_derived Derived> class CrtpInterface>
56 struct DerivedFromCrtpInterfaceImpl {
57 template<class T, class U>
58 DerivedFromCrtpInterfaceImpl(const T&,
59 const CrtpInterface<U>&); // not defined
60 };
61} // namespace detail

63/// @brief Checks if and only if @c T has exactly one public base class
64/// @c CrtpInterface&lt;U&gt; for some type @c U, and @c T has no base classes
65/// of type @c CrtpInterface&lt;V&gt; for any other type @c V.
66template<class T, template<class Derived> class CrtpInterface>
67concept derived_from_crtp_interface = //
68 requires(T x) {
69 detail::DerivedFromCrtpInterfaceImpl<CrtpInterface>{x, x};
70 };
