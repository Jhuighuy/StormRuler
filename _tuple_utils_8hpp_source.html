35// Apply `fixed_size_t<0>{}`, ..., `fixed_size_t<N-1>{}` to function.
36template<size_t N, class Func>
37constexpr decltype(auto) base_apply(Func&& func) {
38 return ([&]<size_t... Is>(std::index_sequence<Is...>) -> decltype(auto) {
39 return func(fixed_size_t<Is>{}...);
40 }(std::make_index_sequence<N>{}));
41}

43// Checks if `std::tuple_size` is applicable.
44template<class Tuple>
45concept has_tuple_size =
46 requires { typename std::tuple_size<Tuple>::type; } &&
47 std::derived_from<std::tuple_size<Tuple>,
48 std::integral_constant<size_t, std::tuple_size_v<Tuple>>>;
49
50// Checks if `std::tuple_element_t` is applicable.
51template<class Tuple, size_t Index>
52concept has_tuple_element = requires(Tuple tuple) {
53 typename std::tuple_element_t<Index, std::remove_const_t<Tuple>>;
54 // clang-format off
55 { std::get<Index>(tuple) } -> std::convertible_to<
56 const std::tuple_element_t<Index, std::remove_const_t<Tuple>>&>;
57 // clang-format on
58};
59
60// Checks if `std::tuple_element_t` is applicable.
61template<has_tuple_size Tuple>
62inline constexpr bool has_tuple_elements_v =
63 base_apply<std::tuple_size_v<Tuple>>([](auto... indices) {
64 return (has_tuple_element<Tuple, indices> && ...);
65 });
66
67// Check if type is tuple-like
68// (`std::array`, `std::pair`, `std::tuple`, ...).
69template<class Tuple>
70concept tuple_like = (!std::is_reference_v<Tuple>) && //
71 has_tuple_size<Tuple> &&
72 has_tuple_elements_v<Tuple>;

74// Wrap an argument into a tuple, if it is not a tuple already.
75class ToTuple final {
76public:
77
78 template<class Tuple>
79 requires tuple_like<std::remove_cvref_t<Tuple>>
80 constexpr decltype(auto) operator()(Tuple&& tuple) const {
81 return std::forward<Tuple>(tuple);
82 }
83 template<class... Args>
84 constexpr auto operator()(Args&&... args) const {
85 return std::tuple{std::forward<Args>(args)...};

95// Same as `std::invocable`, but for `std::apply` instead of `std::invoke`.
96template<class Func, class Tuple>
97concept applicable =
98 tuple_like<Tuple> && //
99 requires(Func& func, Tuple& tuple) { std::apply(func, tuple); };
100
101// Same as `std::regular_invocable`, but for `std::apply`.
102template<class Func, class Tuple>
103concept regular_applicable = applicable<Func, Tuple>;
104
105// Same as `std::invoke_result_t`, but for `std::apply`.
106template<class Func, tuple_like Tuple>
107 requires applicable<Func, Tuple>
108using apply_result_t =
109 decltype(std::apply(std::declval<Func>(), std::declval<Tuple>()));

111// Drop the first N tuple elements and pass the rest to the callback.
112template<size_t N, class Func, class Tuple>
113 requires tuple_like<std::remove_cvref_t<Tuple>>
114constexpr decltype(auto) drop_n_apply(Func&& func, Tuple&& tuple) {
115 constexpr size_t Size = std::tuple_size_v<std::remove_cvref_t<Tuple>>;
116 static_assert(N <= Size);
117 return base_apply<Size - N>([&](auto... indices) -> decltype(auto) {
118 return func(std::get<indices + N>(tuple)...);
119 });
120}
121template<size_t N, class Func, class... Args>
122constexpr decltype(auto) drop_n(Func&& func, Args&&... args) {
123 auto args_tuple = std::forward_as_tuple(std::forward<Args>(args)...);
124 return drop_n_apply<N>(func, args_tuple);
125}
126
127// Pack the first N tuple elements and pass them and the rest to the callback.
128template<size_t N, class Func, class Tuple>
129 requires tuple_like<std::remove_cvref_t<Tuple>>
130constexpr decltype(auto) pack_n_apply(Func&& func, Tuple&& tuple) {
131 constexpr size_t Size = std::tuple_size_v<std::remove_cvref_t<Tuple>>;
132 static_assert(N <= Size);
133 auto packed = base_apply<N>(
134 [&](auto... indices) { return std::tuple{std::get<indices>(tuple)...}; });
135 return base_apply<Size - N>([&](auto... indices) -> decltype(auto) {
136 return func(std::move(packed), std::get<indices + N>(tuple)...);
137 });
138}
139template<size_t N, class Func, class... Args>
140constexpr decltype(auto) pack_n(Func&& func, Args&&... args) {
141 auto args_tuple = std::forward_as_tuple(std::forward<Args>(args)...);
142 return pack_n_apply<N>(func, args_tuple);
143}

145// Apply chunks of size N of tuple elements to a function.
146template<size_t N, class Func = ToVoid, class Chunk = ToTuple, class Tuple>
147 requires tuple_like<std::remove_cvref_t<Tuple>>
148constexpr decltype(auto) chunk_n_apply(Func&& func, Chunk&& chunk,
149 Tuple&& tuple) {
150 constexpr size_t Size = std::tuple_size_v<std::remove_cvref_t<Tuple>>;
151 constexpr size_t NumChunks = Size / N;
152 return base_apply<NumChunks>([&](auto... indices) -> decltype(auto) {
153 const auto chunk_at = [&](auto index) -> decltype(auto) {
154 return base_apply<N>([&](auto... subindices) -> decltype(auto) {
155 return chunk(std::get<index * N + subindices>(tuple)...);
156 });
157 };
158 return func(chunk_at(indices)...);
159 });
160}
161template<size_t N, class Func = ToVoid, class Chunk = ToTuple, class... Args>
162constexpr decltype(auto) chunk_n(Func&& func, Chunk&& chunk, Args&&... args) {
163 auto args_tuple = std::forward_as_tuple(std::forward<Args>(args)...);
164 return chunk_n_apply<N>(func, chunk, args_tuple);
165}

167// Apply zips of tuple elements to a function.
168template<class Func = ToVoid, class Zip = ToTuple, class... Tuples>
169 requires (... && tuple_like<std::remove_cvref_t<Tuples>>)
170constexpr decltype(auto) zip_apply(Func&& func, Zip&& zip, Tuples&&... tuples) {
171 constexpr size_t MinSize =
172 std::min({std::tuple_size_v<std::remove_cvref_t<Tuples>>...});
173 return base_apply<MinSize>([&](auto... indices) -> decltype(auto) {
174 const auto zip_at = [&](auto index) -> decltype(auto) {
175 return zip(std::get<index>(tuples)...);
176 };
177 return func(zip_at(indices)...);
178 });
179}

181// Apply N copies of the argument to a function.
182template<size_t N, class Func, class Arg>
183constexpr decltype(auto) n_times(Func&& func, Arg&& arg) {
184 return base_apply<N>([&](auto... indices) -> decltype(auto) {
185 const auto get_at = [&](auto /*index*/) -> decltype(auto) { return arg; };
186 return func(get_at(indices)...);
187 });
