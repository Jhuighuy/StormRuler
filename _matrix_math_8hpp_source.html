41/// @brief Element-wise apply function to the matrices.
42template<std::copy_constructible Func, matrix_view... Matrices>
43 requires std::is_object_v<Func> && (sizeof...(Matrices) >= 1) &&
44 std::regular_invocable<Func, matrix_element_t<Matrices>...>
45class MapMatrixView final :
46 public MatrixViewInterface<MapMatrixView<Func, Matrices...>>
47{
48private:
49
50 static_assert(std::copyable<Func>, "Boxing is not implemented yet!");
51
52 STORM_NO_UNIQUE_ADDRESS_ Func func_;
53 STORM_NO_UNIQUE_ADDRESS_ std::tuple<Matrices...> mats_;
54
55public:
56
57 /// @brief Construct a map view.
58 constexpr explicit MapMatrixView(Func func, Matrices... mats)
59 : func_{std::move(func)}, mats_{std::move(mats)...}
60 {
61 std::apply(
62 [](const auto& first_mat, const auto&... rest_mats) {
63 STORM_ASSERT_((first_mat.shape() == rest_mats.shape()) && ...,
64 "Shapes of the matrix arguments are mismatched.");
65 },
66 mats_);
67 }
68
69 /// @brief Get the matrix shape.
70 constexpr auto shape() const noexcept
71 {
72 return std::get<0>(mats_).shape();
73 }
74
75 /// @brief Get the matrix element at @p indices.
76 template<class... Indices>
77 requires compatible_matrix_indices_v<MapMatrixView, Indices...>
78 constexpr auto operator()(Indices... indices) const noexcept
79 {
80 STORM_ASSERT_(in_range(shape(), indices...), "Indices are out of range!");
81 auto compute_element = [this, &indices...](const Matrices&... mats) {
82 return func_(mats(indices...)...);
83 };
84 return std::apply(compute_element, mats_);
85 }
86
87}; // class MapMatrixView
88
89template<class Func, class... Matrices>
90MapMatrixView(Func, Matrices&&...)
91 -> MapMatrixView<Func, matrix_view_t<Matrices>...>;

93/// @brief Element-wise apply function @p func to the matrices @p mats.
94template<class Func, viewable_matrix... Matrices>
95 requires compatible_matrices_v<Matrices...> &&
96 std::regular_invocable<Func, matrix_element_t<Matrices>...>
97constexpr auto map(Func&& func, Matrices&&... mats)
98{
99 return MapMatrixView{std::forward<Func>(func),
100 std::forward<Matrices>(mats)...};
101}

336/// @brief Normalize the matrix @p mat (divide by it's norm).
337template<viewable_matrix Matrix>
338 requires numeric_matrix<Matrix>
339constexpr auto normalize(Matrix&& mat)
340{
341 auto mat_view = make_matrix_view(std::forward<Matrix>(mat));
342 const auto mat_norm = norm_2(mat_view);
343 return safe_divide(1.0_dp, mat_norm) * std::move(mat_view);
344}
