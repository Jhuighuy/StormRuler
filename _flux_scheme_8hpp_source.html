39/// @brief Local Lax-Friedrichs (Rusanov) numerical flux.
40/// Use this numerical flux if all other fails. It should always work.
41template<class Physics>
42class LaxFriedrichsFluxScheme;
43
44template<>
45class LaxFriedrichsFluxScheme<tGasPhysics> final
46{
47public:
48
49 template<class Real, size_t NumVars>
50 Subfield<Real, NumVars>
51 operator()(const vec2_t& n, //
52 const Subfield<Real, NumVars>& cons_r,
53 const Subfield<Real, NumVars>& cons_l) const noexcept
54 {
55 return (*this)(vec3_t(n, 0.0), cons_r, cons_l);
56 }
57
58 /// @brief Compute the numerical flux.
59 template<class Real, size_t NumVars>
60 Subfield<Real, NumVars>
61 operator()(const vec3_t& n, //
62 const Subfield<Real, NumVars>& cons_r,
63 const Subfield<Real, NumVars>& cons_l) const noexcept
64 {
65 const tGasPhysics::tFluidState ur(n, cons_r.data());
66 const tGasPhysics::tFluidState ul(n, cons_l.data());
67
68 /* Approximate |J| with its maximum eigenvalue.
69 * [1] Eq. (10.55-10.56). */
70 const real_t ss =
71 std::max(std::abs(ur.vel_n) + ur.c_snd, std::abs(ul.vel_n) + ul.c_snd);
72 Subfield<Real, NumVars> flux_r{}, flux_l{};
73 ur.make_flux(NumVars, n, flux_r.data());
74 ul.make_flux(NumVars, n, flux_l.data());
75 Subfield<Real, NumVars> flux{};
76 for (size_t i = 0; i < NumVars; ++i) {
77 flux[i] = 0.5 * ((flux_r[i] + flux_l[i]) - ss * (cons_r[i] - cons_l[i]));
78 }
79
80 return flux;
81 }
82
83}; // class LaxFriedrichsFluxScheme<tGasPhysics>

86/**
87 * Harten-Lax-van Leer-Einfeldt numerical flux.
88 *
89 * Use this numerical flux if HLLC fails.
90 * It should (almost) always work.
91 */
92/** @{ */
93template<typename tPhysics>
94class tHllFluxScheme;
95template<>
96class tHllFluxScheme<tGasPhysics> : public iFluxScheme {
97public:
98
99 /** Compute the numerical flux. */
100 void get_numerical_flux(size_t num_vars, const vec3_t& n,
101 tScalarConstSubField cons_r,
102 tScalarConstSubField cons_l,
103 tScalarSubField flux) const final {
104 const tGasPhysics::tFluidState ur(n, cons_r.data());
105 const tGasPhysics::tFluidState ul(n, cons_l.data());
106
107 /* Calculate Roe average sound speed.
108 * [1] Eq. (10.53-10.54). */
109 const real_t rr = std::sqrt(ur.rho);
110 const real_t rl = std::sqrt(ul.rho);
111 const real_t rs = 0.5 * rr * rl / std::pow(rr + rl, 2);
112 const real_t cs = std::sqrt(
113 (rr * std::pow(ur.c_snd, 2) + rl * std::pow(ul.c_snd, 2)) / (rr + rl) +
114 rs * std::pow(ur.vel_n - ul.vel_n, 2));
115
116 /* Calculate signal speeds.
117 * [1], Eq. (10.52). */
118 const real_t sr = 0.5 * (ur.vel_n + ul.vel_n) + cs;
119 const real_t sl = 0.5 * (ur.vel_n + ul.vel_n) - cs;
120
121 /* Supersonic cases.
122 * [1], Eq. (10.20-10.21). */
123 if (sr <= 0.0) {
124 ur.make_flux(num_vars, n, flux.data());
125 return;
126 }
127 if (sl >= 0.0) {
128 ul.make_flux(num_vars, n, flux.data());
129 return;
130 }
131
132 /* Subsonic case.
133 * [1], Eq. (10.20-10.21). */
134 FEATHERS_TMP_SCALAR_FIELD(flux_r, num_vars);
135 FEATHERS_TMP_SCALAR_FIELD(flux_l, num_vars);
136 ur.make_flux(num_vars, n, flux_r.data());
137 ul.make_flux(num_vars, n, flux_l.data());
138 if (sl <= 0.0 && 0.0 <= sr) {
139 const real_t is = 1.0 / (sr - sl);
140 for (size_t i = 0; i < num_vars; ++i) {
141 flux[i] = is * ((sr * flux_l[i] - sl * flux_r[i]) +
142 sr * sl * (cons_r[i] - cons_l[i]));
143 }
144 return;
145 }
146
147 STORM_THROW_("Broken signal velocities, sl = {}, sr = {}!", sl, sr);
148 }

152/**
153 * Harten-Lax-van Leer-Contact numerical flux.
154 *
155 * Optimal choice for both gas and plasma physics.
156 * In plasma physics case may be a bit more dissipative, but more consistent
157 * than HLLD/Roe.
158 */
159/** @{ */
160template<typename tPhysics>
161class tHllcFluxScheme;
162template<>
163class tHllcFluxScheme<tGasPhysics> : public iFluxScheme {
164public:
165
166 /** Compute the numerical flux. */
167 void get_numerical_flux(size_t num_vars, const vec3_t& n,
168 tScalarConstSubField cons_r,
169 tScalarConstSubField cons_l,
170 tScalarSubField flux) const override {
171 const tGasPhysics::tFluidState ur(n, cons_r.data());
172 const tGasPhysics::tFluidState ul(n, cons_l.data());
173
174 /* Calculate average variables.
175 * [1], Eq. (10.61-10.62). */
176 const real_t rho = 0.5 * (ur.rho + ul.rho);
177 const real_t c_snd = 0.5 * (ur.c_snd + ul.c_snd);
178 const real_t p = std::max(
179 0.0, 0.5 * (ur.p + ul.p - rho * c_snd * (ur.vel_n - ul.vel_n)));
180
181 /* Calculate sound speed coefficients.
182 * [1], Eq. (10.60). */
183 real_t gp;
184 if (p > ur.p) {
185 gp = std::sqrt(1.0 + /*m_phys->*/ gamma_2 * (p / ur.p - 1.0));
186 } else {
187 gp = 1.0;
188 }
189 real_t gm;
190 if (p > ul.p) {
191 gm = std::sqrt(1.0 + /*m_phys->*/ gamma_2 * (p / ul.p - 1.0));
192 } else {
193 gm = 1.0;
194 }
195
196 /* Calculate signal speeds.
197 * [1], Eq. (10.59). */
198 const real_t sr = ur.vel_n + ur.c_snd * gp;
199 const real_t sl = ul.vel_n - ul.c_snd * gm;
200
201 /* Supersonic cases.
202 * [1], Eq. (10.20-10.21). */
203 if (sr <= 0.0) {
204 ur.make_flux(num_vars, n, flux.data());
205 return;
206 }
207 if (sl >= 0.0) {
208 ul.make_flux(num_vars, n, flux.data());
209 return;
210 }
211
212 /* Subsonic cases.
213 * [1], Eq. (10.37-10.39). */
214 const real_t ss = ((ur.rho * ur.vel_n * (sr - ur.vel_n) - ur.p) -
215 (ul.rho * ul.vel_n * (sl - ul.vel_n) - ul.p)) /
216 (ur.rho * (sr - ur.vel_n) - ul.rho * (sl - ul.vel_n));
217 if (ss <= 0.0 && 0.0 <= sr) {
218 tGasPhysics::tFluidState us;
219 const real_t is = 1.0 / (sr - ss);
220 us.rho = ur.rho * (sr - ur.vel_n) * is;
221 us.nrg = ur.nrg + (ss - ur.vel_n) * (ss + ur.p / ur.rho * is);
222 us.vel = ur.vel + (ss - ur.vel_n) * n;
223 FEATHERS_TMP_SCALAR_FIELD(cons_s, num_vars);
224 us.make_cons(num_vars, cons_s.data());
225 ur.make_flux(num_vars, n, flux.data());
226 for (size_t i = 0; i < num_vars; ++i) {
227 flux[i] += sr * (cons_s[i] - cons_r[i]);
228 }
229 return;
230 }
231 if (sl <= 0.0 && 0.0 <= ss) {
232 tGasPhysics::tFluidState us;
233 const real_t is = 1.0 / (sl - ss);
234 us.rho = ul.rho * (sl - ul.vel_n) * is;
235 us.nrg = ul.nrg + (ss - ul.vel_n) * (ss + ul.p / ul.rho * is);
236 us.vel = ul.vel + (ss - ul.vel_n) * n;
237 FEATHERS_TMP_SCALAR_FIELD(cons_s, num_vars);
238 us.make_cons(num_vars, cons_s.data());
239 ul.make_flux(num_vars, n, flux.data());
240 for (size_t i = 0; i < num_vars; ++i) {
241 flux[i] += sl * (cons_s[i] - cons_l[i]);
242 }
243 return;
244 }
245
246 STORM_THROW_( //
247 "Broken signal velocities, sl = {}, ss = {}, sr = {}!", sl, ss, sr);
248 }
249}; // class tHllcFluxScheme<tGasPhysics>
