35/// @brief Statically-sized matrix.
36template<class Elem, size_t NumRows, size_t NumCols>
37class StaticMatrix final :
38 public TargetMatrixInterface<StaticMatrix<Elem, NumRows, NumCols>>
39{
40private:
41
42 std::array<Elem, NumRows * NumCols> elems_{};
43
44public:
45
46 /// @brief Construct a matrix.
47 constexpr StaticMatrix(const Elem& init = {}) noexcept
48 {
49 fill(init);
50 }
51
52 /// @brief Construct a matrix with the elements.
53 template<class... RestElems>
54 requires (std::convertible_to<RestElems, Elem> && ...) &&
55 (sizeof...(RestElems) + 1 == NumRows * NumCols)
56 constexpr explicit StaticMatrix(const Elem& first_elem,
57 const RestElems&... rest_elems) noexcept
58 : elems_{first_elem, static_cast<Elem>(rest_elems)...}
59 {
60 }
61
62 /// @brief Construct a matrix with another matrix.
63 template<matrix Matrix>
64 requires (!std::same_as<StaticMatrix, std::remove_cvref_t<Matrix>>)
65 constexpr StaticMatrix(Matrix&& other) noexcept
66 {
67 assign(*this, std::forward<Matrix>(other));
68 }
69
70 using TargetMatrixInterface<StaticMatrix<Elem, NumRows, NumCols>>::operator=;
71
72 /// @brief Fill the matrix with @p value.
73 constexpr void fill(const Elem& value) noexcept
74 {
75 for (size_t row_index = 0; row_index < NumRows; ++row_index) {
76 for (size_t col_index = 0; col_index < NumCols; ++col_index) {
77 (*this)(row_index, col_index) = value;
78 }
79 }
80 }
81
82 /// @brief Matrix shape.
83 static constexpr auto shape() noexcept
84 {
85 return std::array{NumRows, NumCols};
86 }
87
88 /// @brief Get the matrix coefficient at @p row_index and @p col_index.
89 /// @{
90 constexpr Elem& operator()(size_t row_index, //
91 size_t col_index = 0) noexcept
92 {
93 STORM_ASSERT_(in_range(shape(), row_index, col_index),
94 "Indices are out of range!");
95 return elems_[row_index * NumCols + col_index];
96 }
97 constexpr const Elem& operator()(size_t row_index,
98 size_t col_index = 0) const noexcept
99 {
100 STORM_ASSERT_(in_range(shape(), row_index, col_index),
101 "Indices are out of range!");
102 return elems_[row_index * NumCols + col_index];
103 }
104 /// @}
105
106 /// @todo Transition code! Remove me!
107 /// @{
108 constexpr Elem& operator[](size_t row_index) noexcept
109 {
110 return (*this)(row_index);
111 }
112 constexpr const Elem& operator[](size_t row_index) const noexcept
113 {
114 return (*this)(row_index);
115 }
116 /// @}
117
118}; // class StaticMatrix
