137/// @brief Dense matrix (fixed shape specialization).
138template<scalar Elem, fixed_shape Shape>
139class DenseMatrix<Elem, Shape> final :
140 public TargetMatrixInterface<DenseMatrix<Elem, Shape>> {
141private:
142
143 static constexpr bool _Multirank = shape_rank_v<Shape> > 1;
144
145 static constexpr auto _FirstExtent =
146 static_cast<size_t>(shape_extent_t<Shape, 0>{});
147
148 using _SliceShape = decltype( //
149 drop_n_apply<1>(to_tuple, std::declval<Shape>()));
150
151 using _Slice =
152 std::conditional_t<_Multirank, DenseMatrix<Elem, _SliceShape>, Elem>;
153
154 STORM_NO_UNIQUE_ADDRESS std::array<_Slice, _FirstExtent> _slices{};
155
156public:
157
158 // ---------------------------------------------------------------------------
159
160 /// @brief Construct a matrix.
161 DenseMatrix() = default;
162
163 // ---------------------------------------------------------------------------
164
165 /// @brief Construct a matrix with elements of the matrix @p mat.
166 template<matrix Matrix>
167 requires assignable_matrix<DenseMatrix, Matrix> &&
168 different_from<DenseMatrix, Matrix>
169 constexpr DenseMatrix(Matrix&& mat) { // NOSONAR
170 this->assign(std::forward<Matrix>(mat));
171 }
172
173 /// @brief Assign the current matrix elements from matrix @p mat.
174 template<matrix Matrix>
175 requires assignable_matrix<DenseMatrix, Matrix> &&
176 different_from<DenseMatrix, Matrix>
177 constexpr DenseMatrix& operator=(Matrix&& mat) {
178 return this->assign(std::forward<Matrix>(mat));
179 }
180
181 // ---------------------------------------------------------------------------
182
183 /// @brief Construct a matrix with the slice range @p slices.
184 /// @{
185 constexpr DenseMatrix(std::initializer_list<_Slice> slices) {
186 std::ranges::copy(slices, _slices.begin());
187 }
188 template<std::ranges::input_range SlicesRange>
189 constexpr explicit DenseMatrix(SlicesRange&& slices) {
190 std::ranges::copy(slices, _slices.begin());
191 }
192 /// @}
193
194 /// @brief Assign the current matrix to from slices range @p slices.
195 /// @{
196 constexpr DenseMatrix& operator=(std::initializer_list<_Slice> slices) {
197 std::ranges::copy(slices, _slices.begin());
198 return *this;
199 }
200 template<std::ranges::input_range SlicesRange>
201 constexpr DenseMatrix& operator=(SlicesRange&& slices) {
202 std::ranges::copy(slices, _slices.begin());
203 return *this;
204 }
205 /// @}
206
207 // ---------------------------------------------------------------------------
208
209 /// @brief Get the matrix shape.
210 constexpr auto shape() const noexcept {
211 return Shape{};
212 }
213
214 /// @brief Get the slice at @p index.
215 /// @{
216 constexpr _Slice& operator[](size_t slice_index) noexcept {
217 STORM_ASSERT(slice_index < _FirstExtent, "Slice index is out of range!");
218 return _slices[slice_index];
219 }
220 constexpr const _Slice& operator[](size_t slice_index) const noexcept {
221 STORM_ASSERT(slice_index < _FirstExtent, "Slice index is out of range!");
222 return _slices[slice_index];
223 }
224 /// @}
225
226 /// @brief Get the matrix element at @p indices.
227 /// @{
228 template<class... Indices>
229 requires compatible_matrix_indices_v<DenseMatrix, Indices...>
230 constexpr Elem& operator()(Indices... indices) noexcept {
231 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
232 return [&](auto first_index, auto... rest_indices) -> Elem& {
233 _Slice& slice = _slices[first_index];
234 if constexpr (_Multirank) return slice(rest_indices...);
235 else return slice;
236 }(indices...);
237 }
238 template<class... Indices>
239 requires compatible_matrix_indices_v<DenseMatrix, Indices...>
240 constexpr const Elem& operator()(Indices... indices) const noexcept {
241 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
242 return [&](auto first_index, auto... rest_indices) -> const Elem& {
243 const _Slice& slice = _slices[first_index];
244 if constexpr (_Multirank) return slice(rest_indices...);
245 else return slice;
246 }(indices...);
247 }
248 /// @}
249
250}; // class DenseMatrix
