40/// @brief Target matrix: an output matrix that is an assignment target.
41/// Additional requirement: target matrix overrides (or uses) the operator=
42/// from the TargetMatrixInterface.
43template<class TargetMatrix>
44concept target_matrix =
45 output_matrix<TargetMatrix> &&
46 derived_from_crtp_interface<TargetMatrix, TargetMatrixInterface>;
47
48/// @brief CRTP interface to a target matrix.
49template<crtp_derived TargetMatrix>
50class TargetMatrixInterface {
51private:
52
53 constexpr TargetMatrix& _self() noexcept {
54 static_assert(std::derived_from<TargetMatrix, TargetMatrixInterface>);
55 static_assert(target_matrix<TargetMatrix>);
56 return static_cast<TargetMatrix&>(*this);
57 }
58 constexpr const TargetMatrix& _self() const noexcept {
59 return const_cast<TargetMatrixInterface&>(*this)._self();
60 }
61
62public:
63
64 /// @brief Assign the current matrix coefficients from matrix @p mat.
65 template<matrix Matrix>
66 requires std::assignable_from< //
67 matrix_element_ref_t<TargetMatrix>, matrix_element_t<Matrix>>
68 constexpr decltype(auto) operator=(Matrix&& mat) noexcept {
69 return assign(_self(), std::forward<Matrix>(mat));
70 }
71
72 /// @brief Multiply-assign the current matrix by a scalar @p scal.
73 template<std::copyable Scalar>
74 constexpr decltype(auto) operator*=(Scalar scal) {
75 return assign(BindLast{MultiplyAssign{}, std::move(scal)}, *this);
76 }
77
78 /// @brief Divide-assign the current matrix by a scalar @p scal.
79 template<std::copyable Scalar>
80 constexpr decltype(auto) operator/=(Scalar scal) {
81 return assign(BindLast{DivideAssign{}, std::move(scal)}, *this);
82 }
83
84 /// @brief Add-assign the matrix @p mat to the current matrix.
85 template<matrix Matrix>
86 requires compatible_matrices_v<TargetMatrix, Matrix>
87 constexpr decltype(auto) operator+=(Matrix&& mat) {
88 return assign(AddAssign{}, _self(), std::forward<Matrix>(mat));
89 }
90
91 /// @brief Subtract-assign the matrix @p mat from the current matrix.
92 template<matrix Matrix>
93 requires compatible_matrices_v<TargetMatrix, Matrix>
94 constexpr decltype(auto) operator-=(Matrix&& mat) {
95 return assign(SubtractAssign{}, _self(), std::forward<Matrix>(mat));
96 }
97
98 /// @brief Element-wise multiply-assign the current matrix by
99 /// the matrix @p mat.
100 template<matrix Matrix>
101 requires compatible_matrices_v<TargetMatrix, Matrix>
102 constexpr decltype(auto) operator*=(Matrix&& mat) {
103 return assign(MultiplyAssign{}, _self(), std::forward<Matrix>(mat));
104 }
105
106 /// @brief Element-wise divide-assign the current matrix by the matrix @p mat.
107 template<matrix Matrix>
108 requires compatible_matrices_v<TargetMatrix, Matrix>
109 constexpr decltype(auto) operator/=(Matrix&& mat) {
110 return assign(DivideAssign{}, _self(), std::forward<Matrix>(mat));
111 }
112
113}; // class TargetMatrixInterface

117/// @brief Output matrix as target wrapper.
118template<matrix_view Matrix>
119 requires output_matrix<Matrix>
120class TargetMatrixView final :
121 public MatrixViewInterface<TargetMatrixView<Matrix>>,
122 public TargetMatrixInterface<TargetMatrixView<Matrix>>,
123 public NonMovableInterface {
124private:
125
126 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
127
128public:
129
130 /// @brief Construct a matrix target view.
131 constexpr explicit TargetMatrixView(Matrix mat) : _mat{std::move(mat)} {}
132
133 /// @brief Get the matrix shape.
134 constexpr auto shape() const noexcept {
135 return _mat.shape();
136 }
137
138 /// @brief Get the matrix element at @p indices.
139 template<class... Indices>
140 requires compatible_matrix_indices_v<TargetMatrixView, Indices...>
141 constexpr auto operator()(Indices... indices) const noexcept {
142 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
143 return _mat(indices...);
144 }
145
146 using TargetMatrixInterface<TargetMatrixView<Matrix>>::operator=;
147
148}; // class TargetMatrixView

153/// @brief Wrap the output matrix @p mat as a target.
154template<viewable_matrix Matrix>
155 requires output_matrix<Matrix>
156constexpr auto as_target(Matrix&& mat) {
157 return TargetMatrixView{std::forward<Matrix>(mat)};
158}
