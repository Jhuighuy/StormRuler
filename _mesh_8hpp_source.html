68/// @brief Mesh entity view.
69template<mesh Mesh, index Index>
70class EntityView {
71private:
72
73 template<mesh, index>
74 friend class EntityView;
75
76 const Mesh* p_mesh_;
77 Index index_;
78
79protected:
80
81 /// @brief Construct an entity view with a @p mesh and @p index.
82 constexpr EntityView(const Mesh& mesh, Index index) noexcept
83 : p_mesh_{&mesh}, index_{index} {}
84
85 /// @brief Destroy the entity view.
86 constexpr ~EntityView() = default;
87
88public:
89
90 /// @brief Entity mesh.
91 /// @{
92 constexpr Mesh& mesh() noexcept {
93 return *p_mesh_;
94 }
95 constexpr const Mesh& mesh() const noexcept {
96 return *p_mesh_;
97 }
98 /// @}
99
100 /// @brief Entity index.
101 /// @{
102 constexpr Index index() const noexcept {
103 return index_;
104 }
105 constexpr size_t index_sz() const noexcept {
106 return static_cast<size_t>(index_);
107 }
108 /// @}
109
110 /// @brief Cast to entity index operator.
111 constexpr operator Index() const noexcept {
112 return index_;
113 }
114
115 /// @brief Comparison operator.
116 constexpr auto operator<=>(const EntityView& other) const noexcept {
117 STORM_ASSERT_(p_mesh_ == other.p_mesh_,
118 "Entities correspond to the different meshes!");
119 return index_ <=> other.index_;
120 }
121
122 /// @brief Entity label.
123 constexpr auto label() const noexcept {
124 return mesh().label(index_);
125 }
126
127 /// @brief Range of the adjacent nodes.
128 constexpr auto nodes() const noexcept {
129 return mesh().adjacent(index_, meta::type_v<NodeIndex>) |
130 detail_::to_node_view_(*p_mesh_);
131 }
132
133 /// @brief Range of the adjacent edges.
134 constexpr auto edges() const noexcept {
135 return mesh().adjacent(index_, meta::type_v<EdgeIndex>) |
136 detail_::to_edge_view_(*p_mesh_);
137 }
138
139 /// @brief Range of the adjacent faces.
140 constexpr auto faces() const noexcept {
141 return mesh().adjacent(index_, meta::type_v<FaceIndex<Mesh>>) |
142 detail_::to_face_view_(*p_mesh_);
143 }
144
145 /// @brief Range of the adjacent cells.
146 constexpr auto cells() const noexcept {
147 return mesh().adjacent(index_, meta::type_v<CellIndex<Mesh>>) |
148 detail_::to_cell_view_(*p_mesh_);
149 }
150
151 /// @brief Iterate all the adjacent nodes.
152 template<std::invocable<NodeView<Mesh>> Func>
153 constexpr void for_each_node(Func func) const noexcept {
154 std::ranges::for_each(nodes(), func);
155 }
156
157 /// @brief Iterate all the adjacent edges.
158 template<std::invocable<EdgeView<Mesh>> Func>
159 constexpr void for_each_edge(Func func) const noexcept {
160 std::ranges::for_each(edges(), func);
161 }
162
163 /// @brief Iterate all the adjacent faces.
164 /// @{
165 template<std::invocable<FaceView<Mesh>> Func>
166 constexpr void for_each_face(Func func) const noexcept {
167 std::ranges::for_each(faces(), func);
168 }
169 template<std::invocable<CellView<Mesh>, CellView<Mesh>> Func>
170 constexpr void for_each_face_cells(Func func) const noexcept {
171 std::ranges::for_each(faces(), [&](FaceView<Mesh> face) {
172 func(face.inner_cell(), face.outer_cell());
173 });
174 }
175 /// @}
176
177 /// @brief Iterate all the adjacent cells.
178 template<std::invocable<CellView<Mesh>> Func>
179 constexpr void for_each_cell(Func func) const noexcept {
180 std::ranges::for_each(cells(), func);
181 }
182
183}; // class EntityView

239/// @brief Face view.
240template<mesh Mesh>
241class FaceView final : public EntityView<Mesh, FaceIndex<Mesh>> {
242public:
243
244 /// @brief Construct a face view.
245 constexpr FaceView(const Mesh& mesh, FaceIndex<Mesh> index) noexcept
246 : EntityView<Mesh, FaceIndex<Mesh>>(mesh, index) {}
247
248 /// @brief Face shape type.
249 constexpr shapes::Type shape_type() const noexcept {
250 return this->mesh().shape_type(this->index());
251 }
252
253 /// @brief Face area (or length in 2D).
254 constexpr real_t area() const noexcept {
255 return this->mesh().volume(this->index());
256 }
257
258 /// @brief Face barycenter position.
259 constexpr auto center() const noexcept {
260 return this->mesh().position(this->index());
261 }
262
263 /// @brief Face normal.
264 constexpr auto normal() const noexcept {
265 return this->mesh().normal(this->index());
266 }
267
268 /// @brief Get the adjacent inner cell.
269 constexpr CellView<Mesh> inner_cell() const noexcept {
270 STORM_ASSERT_(!this->cells().empty(),
271 "The face does not have an adjacent inner cell!");
272 return this->cells().front();
273 }
274
275 /// @brief Get the adjacent outer cell.
276 constexpr CellView<Mesh> outer_cell() const noexcept {
277 STORM_ASSERT_(this->cells().size() == 2,
278 "The face does not have an adjacent outer cell!");
279 return this->cells().back();
280 }
281
282}; // class FaceView

330/// @brief CRTP interface to a mesh.
331template<crtp_derived Derived>
332class MeshInterface {
333private:
334
335 constexpr Derived& self_() noexcept {
336 static_assert(std::derived_from<Derived, MeshInterface>);
337 return static_cast<Derived&>(*this);

340 return const_cast<MeshInterface&>(*this).self_();
341 }
342
343public:
344
345 /// @brief Number of the node labels.
346 constexpr size_t num_node_labels() const noexcept {
347 return self_().num_labels(meta::type_v<NodeIndex>);
348 }
349 /// @brief Number of the edge labels.
350 constexpr size_t num_edge_labels() const noexcept {
351 return self_().num_labels(meta::type_v<EdgeIndex>);
352 }
353 /// @brief Number of the face labels.
354 constexpr size_t num_face_labels() const noexcept {
355 return self_().num_labels(meta::type_v<FaceIndex<Derived>>);
356 }
357 /// @brief Number of the cell labels.
358 constexpr size_t num_cell_labels() const noexcept {
359 return self_().num_labels(meta::type_v<CellIndex<Derived>>);
360 }
361
362 /// @brief Number of the nodes.
363 constexpr size_t num_nodes() const noexcept {
364 return self_().num_entities(meta::type_v<NodeIndex>);
365 }
366 /// @brief Number of the nodes with a @p label.
367 constexpr size_t num_nodes(Label label) const noexcept {
368 return self_().num_entities(label, meta::type_v<NodeIndex>);
369 }
370
371 /// @brief Number of the edges.
372 constexpr size_t num_edges() const noexcept {
373 return self_().num_entities(meta::type_v<EdgeIndex>);
374 }
375 /// @brief Number of the edges with a @p label.
376 constexpr size_t num_edges(Label label) const noexcept {
377 return self_().num_entities(label, meta::type_v<EdgeIndex>);

381 constexpr size_t num_faces() const noexcept {
382 return self_().num_entities(meta::type_v<FaceIndex<Derived>>);
383 }
384 /// @brief Number of the faces with a @p label.
385 constexpr size_t num_faces(Label label) const noexcept {
386 return self_().num_entities(label, meta::type_v<FaceIndex<Derived>>);
387 }
388
389 /// @brief Number of the cells.
390 constexpr size_t num_cells() const noexcept {
391 return self_().num_entities(meta::type_v<CellIndex<Derived>>);
392 }
393 /// @brief Number of the cells with a @p label.
394 constexpr size_t num_cells(Label label) const noexcept {
395 return self_().num_entities(label, meta::type_v<CellIndex<Derived>>);
396 }
397
398 /// @brief Range of the nodes.
399 constexpr auto nodes() const noexcept {
400 return self_().entities(meta::type_v<NodeIndex>) |
401 detail_::to_node_view_(self_());
402 }
403 /// @brief Range of the nodes with a @p label.
404 constexpr auto nodes(Label label) const noexcept {
405 return self_().entities(label, meta::type_v<NodeIndex>) |
406 detail_::to_node_view_(self_());
407 }
408
409 /// @brief Range of the edges.
410 constexpr auto edges() const noexcept {
411 return self_().entities(meta::type_v<EdgeIndex>) |
412 detail_::to_edge_view_(self_());
413 }
414 /// @brief Range of the edges with @p label.
415 constexpr auto edges(Label label) const noexcept {
416 return self_().entities(label, meta::type_v<EdgeIndex>) |
417 detail_::to_edge_view_(self_());
418 }
419
420 /// @brief Range of the faces.
421 constexpr auto faces() const noexcept {
422 return self_().entities(meta::type_v<FaceIndex<Derived>>) |
423 detail_::to_face_view_(self_());
424 }
425 /// @brief Range of the faces with @p label.
426 constexpr auto faces(Label label) const noexcept {
427 return self_().entities(label, meta::type_v<FaceIndex<Derived>>) |
428 detail_::to_face_view_(self_());
429 }
430
431 /// @brief Range of the cells.
432 constexpr auto cells() const noexcept {
433 return self_().entities(meta::type_v<CellIndex<Derived>>) |
434 detail_::to_cell_view_(self_());
435 }
436 /// @brief Range of the cells with @p label.
437 constexpr auto cells(Label label) const noexcept {
438 return self_().entities(label, meta::type_v<CellIndex<Derived>>) |
439 detail_::to_cell_view_(self_());
440 }
441
442 /// @brief Range of the interior nodes.
443 constexpr auto interior_nodes() const noexcept {
444 return nodes(Label{0});
445 }
446
447 /// @brief Range of the interior edges.
448 constexpr auto interior_edges() const noexcept {

466
467 /// @brief Range of the boundary edges.

471
472 /// @brief Range of the boundary faces.
473 constexpr auto boundary_faces() const noexcept {
474 return faces() | std::views::drop(num_faces(Label{0}));
475 }
476
477 /// @brief Range of the boundary cells.
478 constexpr auto boundary_cells() const noexcept {
479 return cells() | std::views::drop(num_cells(Label{0}));
480 }

482}; // class MeshInterface
