55/// @brief Assign the matrices.
56/// @{
57template<output_matrix OutMatrix, matrix Matrix>
58constexpr OutMatrix& assign(OutMatrix&& out_mat, Matrix&& mat) noexcept {
59 STORM_ASSERT(out_mat.shape() == mat.shape(), "Matrix shapes do not match!");
60 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
61 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
62 out_mat(row_index, col_index) = mat(row_index, col_index);
63 }
64 }
65 return out_mat;
66}

68constexpr OutMatrix& assign(AssignFunc assign_func, //
69 OutMatrix&& out_mat, Matrices&&... mats) noexcept {
70 STORM_ASSERT((out_mat.shape() == mats.shape()) && ...,

72 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
73 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
74 assign_func(out_mat(row_index, col_index), mats(row_index, col_index)...);

90/// @brief Fill the matrix @p out_mat with a scalar @p scal.
91template<matrix OutMatrix, std::copyable Scalar>
92 requires (!matrix<Scalar>)
93constexpr OutMatrix& fill(OutMatrix& out_mat, Scalar scal) {
94 return assign(

99/// @brief Fill the matrix @p out_mat elements with
100/// the uniformly-distributed random numbers.
101/// @note This is a sequential operation!
102template<output_matrix Matrix>
103STORM_CPP23_CONSTEXPR Matrix&
104fill_randomly(Matrix&& out_mat, //
105 matrix_element_t<Matrix> min = 0,
106 matrix_element_t<Matrix> max = 1) noexcept {
107 static std::mt19937_64 random_engine{}; // NOSONAR
108 std::uniform_real_distribution distribution{min, max};
109 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
110 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
111 out_mat(row_index, col_index) = distribution(random_engine);
112 }
113 }
114 return out_mat;
115}

119/// @brief Reduce the matrix @p mat coefficients to a single value.
120/// @param init Initial reduction value.
121/// @param reduce_func Reduction function.
122/// @todo Restrictions!
123/// @{
124template<class Value, class ReduceFunc, matrix Matrix>
125constexpr auto reduce(Value init, ReduceFunc reduce_func, Matrix&& mat) {
126 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
127 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
128 init = reduce_func(init, mat(row_index, col_index));
129 }
130 }
131 return init;
132}
133template<class Value, class ReduceFunc, class Func, //
134 matrix Matrix, matrix... RestMatrices>
135 requires compatible_matrices_v<Matrix, RestMatrices...>
136constexpr auto reduce(Value init, ReduceFunc reduce_func, Func func,
137 Matrix&& mat, RestMatrices&&... mats) noexcept {
138 STORM_ASSERT((mat.shape() == mats.shape()) && ...,
139 "Matrix shapes doesn't match!");
140 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
141 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
142 init = reduce_func(
143 init, func(mat(row_index, col_index), mats(row_index, col_index)...));
144 }
145 }
146 return init;
147}

195/// @brief Element-wise matrix @p mat @f$ L_{1} @f$-norm.
196template<matrix Matrix>
197constexpr auto norm_1(Matrix&& mat) {
198 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
199 return reduce(Result{}, Add{}, Abs{}, std::forward<Matrix>(mat));
200}

202/// @brief Element-wise matrix @p mat @f$ L_{2} @f$-norm.
203/// @{
204template<matrix Matrix>
205constexpr auto norm_2_2(Matrix&& mat) {
206 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
207 return reduce(Result{}, Add{}, AbsSquared{}, std::forward<Matrix>(mat));
208}
209template<matrix Matrix>
210constexpr auto norm_2(Matrix&& mat) {
211 return sqrt(norm_2_2(std::forward<Matrix>(mat)));
212}
213/// @}

215/// @brief Element-wise matrix @p mat @f$ L_{p} @f$-norm.
216/// @{
217template<matrix Matrix, class Number>
218constexpr auto norm_p_p(Matrix&& mat, Number p) {
219 STORM_ASSERT(p > 0, "Invalid p-norm parameter!");
220 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
221 return reduce(Result{}, Add{}, Compose{Abs{}, BindLast{Pow{}, std::move(p)}},
222 std::forward<Matrix>(mat));
223}
224template<matrix Matrix, class Number>
225constexpr auto norm_p(Matrix&& mat, Number p) {
226 return pow(norm_p_p(std::forward<Matrix>(mat), p), 1.0 / p);
227}
228/// @}
229
230/// @brief Element-wise matrix @p mat @f$ L_{\infty} @f$-norm.
231template<matrix Matrix>
232constexpr auto norm_inf(Matrix&& mat) {
233 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
234 return reduce(Result{}, Max{}, Abs{}, std::forward<Matrix>(mat));
235}
