49public:

54private:
55
56 std::shared_ptr<Mesh> m_mesh;
57 std::shared_ptr<iConvectionScheme> m_conv;
58 std::map<Label, std::shared_ptr<MhdFvBcPT<MhdPhysicsT>>> m_bcs;

62 explicit MhdFvSolverT(std::shared_ptr<Mesh> mesh) : m_mesh(mesh),
63#if 0
64 m_conv(new UpwindConvectionScheme( //
65 *mesh, tHllcFluxScheme<MhdPhysicsT>{}))
66#else
67 m_conv(new LinearUpwindConvectionScheme(
68 *mesh, LaxFriedrichsFluxScheme<MhdPhysicsT>{},
69 LeastSquaresGradientScheme{*mesh},
70 GradientLimiterScheme{*mesh, //
71 CubicSlopeLimiter{}, CubicSecondLimiter{}}))
72#endif
73 {
74 m_bcs[Label{1}] = std::make_shared<MhdFvBcFarFieldT<MhdPhysicsT>>();
75 m_bcs[Label{2}] = std::make_shared<MhdFvBcSlipT<MhdPhysicsT>>();
76 m_conv->bcs_ = &m_bcs;
77 }

79 /**
80 * @brief Compute spacial discretization.
81 */
82 void calc_func(CellField<Mesh, real_t, 5>& u,
83 CellField<Mesh, real_t, 5>& f) const
84 {
85 std::ranges::for_each(m_mesh->cells(),
86 [&](CellView<Mesh> cell) { f[cell].fill(0.0); });
87 m_conv->get_cell_convection(f, u);
88 }
89
90 /*
91 * Compute time step.
92 */
93 void calc_step(real_t& dt, //
94 CellField<Mesh, real_t, 5>& u,
95 CellField<Mesh, real_t, 5>& u_hat) const
96 {
97 calc_func(u, u_hat);
98 std::ranges::for_each(m_mesh->interior_cells(), [&](CellView<Mesh> cell) {
99 for (size_t i = 0; i < 5; ++i) {
100 u_hat[cell][i] = u[cell][i] - dt * u_hat[cell][i];
101 }
102 });
103 }
