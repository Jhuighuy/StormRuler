35/// @brief Strictly-tagged index.
36/// @tparam Tag Any type to distinguish the different index types.
37template<class Tag>
38class Index final {
39private:
40
41 size_t value_;
42
43public:
44
45 /// @brief Construct an undefined index.
46 constexpr Index() = default;
47
48 /// @brief Construct an index with value.
49 template<std::integral Integral>
50 constexpr explicit Index(Integral value) noexcept
51 : value_{static_cast<size_t>(value)} {}
52
53 /// @brief Cast to the inegral value.
54 template<std::integral Integral>
55 constexpr explicit operator Integral() const noexcept {
56 return static_cast<Integral>(value_);
57 }
58
59 /// @brief Cast into another tagged index.
60 template<class OtherTag>
61 constexpr explicit operator Index<OtherTag>() const noexcept {
62 return Index<OtherTag>{value_};
63 }
64
65 /// @brief Comparison operators.
66 /// @{
67 constexpr auto operator<=>(const Index& other) const = default;
68 template<std::integral Integral>
69 constexpr auto operator<=>(Integral value) const noexcept {
70 return value_ <=> static_cast<size_t>(value);
71 }
72 /// @}
73
74 /// @brief Increment operators.
75 /// @{
76 constexpr Index& operator++() noexcept {
77 return ++value_, *this;
78 }
79 constexpr Index operator++(int) noexcept {
80 Index copy{*this};
81 return ++value_, copy;
82 }
83 /// @}
84
85 /// @brief Decrement operators.
86 /// @{
87 constexpr Index& operator--() noexcept {
88 return --value_, *this;
89 }
90 constexpr Index operator--(int) noexcept {
91 Index copy{*this};
92 return --value_, copy;
93 }
94 /// @}
95
96 /// @brief Addition operators.
97 /// @{
98 constexpr Index& operator+=(ptrdiff_t delta) noexcept {
99 return value_ += delta, *this;
100 }
101 friend constexpr Index operator+(Index i, ptrdiff_t d) noexcept {
102 return Index{i.value_ + d};
103 }
104 friend constexpr Index operator+(ptrdiff_t d, Index i) noexcept {
105 return Index{d + i.value_};
106 }
107 /// @}
108
109 /// @brief Subtraction operators.
110 /// @{
111 constexpr Index& operator-=(ptrdiff_t delta) noexcept {
112 return value_ -= delta, *this;
113 }
114 friend constexpr Index operator-(Index i, ptrdiff_t d) noexcept {
115 return Index{i.value_ - d};
116 }
117 /// @}
118
119 /// @brief Difference operator.
120 friend constexpr ptrdiff_t operator-(Index i1, Index i2) noexcept {
121 return static_cast<ptrdiff_t>(i1.value_ - i2.value_);
122 }
123
124 /// @brief Read @p index from the input @p stream.
125 friend std::istream& operator>>(std::istream& stream, Index& i) {
126 return stream >> i.value_;
127 }
128
129 /// @brief Write @p index to the output @p stream.
130 friend std::ostream& operator<<(std::ostream& stream, Index i) {
131 return stream << i.value_;
132 }
133
134}; // class Index
