92/// @brief If @p y is zero, return zero,
93/// else return value of @p x divided by @p y.

95constexpr auto safe_divide(Value x, Value y)
96{
97 constexpr Value zero{0.0};

221/// @brief Generate the Givens rotation.
222/// @todo Move me somewhere else!
223template<real_or_complex_type Value>
224auto sym_ortho(Value a, Value b)
225{
226 // Compute:
227 // ----------------------
228 // ùëüùëü ‚Üê (ùëé¬≤ + ùëè¬≤)¬π·êü¬≤,
229 // ùëêùë† ‚Üê ùëé/ùëüùëü, ùë†ùëõ ‚Üê ùëè/ùëüùëü.
230 // ----------------------
231 Value cs, sn, rr;
232 rr = std::hypot(a, b);
233 if (rr > Value{0.0}) {
234 cs = a / rr;
235 sn = b / rr;
236 } else {
237 cs = 1.0;
238 sn = 0.0;
239 }
240
241 return std::tuple(cs, sn, rr);
242
243} // sym_ortho
