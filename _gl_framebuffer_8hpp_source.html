34/// @brief OpenGL framebuffer.
35class Framebuffer : detail_::noncopyable_ {
36private:
37
38 GLuint framebuffer_id_;
39
40public:
41
42 /// @brief Construct a framebuffer.
43 Framebuffer() {
44 glGenFramebuffers(1, &framebuffer_id_);
45 }
46
47 /// @brief Construct a framebuffer with @p textures.
48 // template<pixel... Pixels>
49 explicit Framebuffer(const auto&... textures) : Framebuffer{} {
50 assign(textures...);
51 }
52
53 /// @brief Move-construct a framebuffer.
54 Framebuffer(Framebuffer&&) = default;
55 /// @brief Move-assign the framebuffer.
56 Framebuffer& operator=(Framebuffer&&) = default;
57
58 /// @brief Destruct the framebuffer.
59 ~Framebuffer() {
60 glDeleteFramebuffers(1, &framebuffer_id_);
61 }
62
63 /// @brief Cast to framebuffer ID.
64 constexpr operator GLuint() const noexcept {
65 return framebuffer_id_;
66 }
67
68 /// @brief Draw to framebuffer.
69 template<std::invocable DrawFunc>
70 void draw_into(DrawFunc draw_func) {
71 glBindFramebuffer(GL_FRAMEBUFFER, framebuffer_id_);
72 draw_func();
73 glBindFramebuffer(GL_FRAMEBUFFER, 0);
74 }
75
76 /// @brief Build the framebuffer with @p textures.
77 // template<pixel... Pixels>
78 void assign(const auto&... textures) {
79 glBindFramebuffer(GL_FRAMEBUFFER, framebuffer_id_);
80 const auto attachments = attach_textures_(textures...);
81 glDrawBuffers(static_cast<GLsizei>(attachments.size()), attachments.data());
82 auto status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
83 if (status != GL_FRAMEBUFFER_COMPLETE) {
84 STORM_THROW_GL_( //
85 "Failed to initialize framebuffer, status = {:#x}!", status);
86 }
87 glBindFramebuffer(GL_FRAMEBUFFER, 0);
88 }
89
90private:
91
92 static auto attach_textures_(const auto&... textures) {
93 GLenum index = 0;
94 std::array attachments{attach_single_texture_(index++, textures)...};
95 return attachments;
96 }
97
98 template<pixel Pixel>
99 static GLenum attach_single_texture_(GLenum index,
100 const Texture2D<Pixel>& texture) {
101 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + index,
102 GL_TEXTURE_2D, texture, /*level*/ 0);
103 return GL_COLOR_ATTACHMENT0 + index;
104 }
105 template<pixel Pixel>
106 static GLenum
107 attach_single_texture_(GLenum index,
108 const MultisampledTexture2D<Pixel>& texture) {
109 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + index,
110 GL_TEXTURE_2D_MULTISAMPLE, texture, /*level*/ 0);
111 return GL_COLOR_ATTACHMENT0 + index;
112 }
113
114}; // class Framebuffer
