40/// @brief Parmetrized GMRES solver.
41template<legacy_vector_like Vector, bool Flexible, bool Loose = false>
42class BaseGmresSolver : public InnerOuterIterativeSolver<Vector> {
43private:
44
45 DenseVector<real_t> beta_, cs_, sn_;
46 DenseMatrix<real_t> H_;
47 std::vector<Vector> q_vecs_;
48 std::conditional_t<Flexible, std::vector<Vector>, std::array<Vector, 1>>
49 z_vecs_;
50
51 real_t outer_init(const Vector& x_vec, const Vector& b_vec,
52 const Operator<Vector>& lin_op,
53 const Preconditioner<Vector>* pre_op) override {
54 const size_t m = this->num_inner_iterations;
55
56 beta_.assign(m + 1);
57 cs_.assign(m), sn_.assign(m);
58 H_.assign(m + 1, m);
59
60 q_vecs_.resize(m + 1);
61 for (Vector& q_vec : q_vecs_) q_vec.assign(x_vec, false);
62 if (pre_op != nullptr) {
63 if constexpr (Flexible) z_vecs_.resize(m);
64 for (Vector& z_vec : z_vecs_) z_vec.assign(x_vec, false);
65 }
66
67 /// @todo Refactor without duplication a code from
68 /// inner_init method.
69 const bool left_pre = (pre_op != nullptr) && (!Flexible) &&
70 (this->pre_side == PreconditionerSide::Left);
71
72 // Initialize:
73 // ----------------------
74 // 𝒒₀ ← 𝒃 - 𝓐𝒙,
75 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
76 // 𝒛₀ ← 𝒒₀,
77 // 𝒒₀ ← 𝓟𝒛₀,
78 // 𝗲𝗻𝗱 𝗶𝗳
79 // 𝛽₀ ← ‖𝒒₀‖,
80 // 𝒒₀ ← 𝒒₀/𝛽₀.
81 // ----------------------
82 lin_op.Residual(q_vecs_[0], b_vec, x_vec);
83 if (left_pre) {
84 std::swap(z_vecs_[0], q_vecs_[0]);
85 pre_op->mul(q_vecs_[0], z_vecs_[0]);
86 }
87 beta_(0) = norm_2(q_vecs_[0]);
88 q_vecs_[0] /= beta_(0);
89
90 return beta_(0);
91 }

93 void inner_init(const Vector& x_vec, const Vector& b_vec,
94 const Operator<Vector>& lin_op,
95 const Preconditioner<Vector>* pre_op) override {
96 // Force right preconditioning for the flexible GMRES.
97 const bool left_pre = (pre_op != nullptr) && (!Flexible) &&
98 (this->pre_side == PreconditionerSide::Left);
99
100 // Initialize:
101 // ----------------------
102 // 𝒒₀ ← 𝒃 - 𝓐𝒙,
103 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
104 // 𝒛₀ ← 𝒒₀,
105 // 𝒒₀ ← 𝓟𝒛₀,
106 // 𝗲𝗻𝗱 𝗶𝗳
107 // 𝛽₀ ← ‖𝒒₀‖,
108 // 𝒒₀ ← 𝒒₀/𝛽₀.
109 // ----------------------
110 lin_op.Residual(q_vecs_[0], b_vec, x_vec);
111 if (left_pre) {
112 std::swap(z_vecs_[0], q_vecs_[0]);
113 pre_op->mul(q_vecs_[0], z_vecs_[0]);
114 }
115 beta_(0) = norm_2(q_vecs_[0]);
116 q_vecs_[0] /= beta_(0);
117 }

119 real_t inner_iterate(Vector& x_vec, const Vector& b_vec,
120 const Operator<Vector>& lin_op,
121 const Preconditioner<Vector>* pre_op) override {
122 const size_t k = this->inner_iteration;
123
124 // Force right preconditioning for the flexible GMRES.
125 const bool left_pre =
126 (pre_op != nullptr) &&
127 (!Flexible && (this->pre_side == PreconditionerSide::Left));
128 const bool right_pre =
129 (pre_op != nullptr) &&
130 (Flexible || (this->pre_side == PreconditionerSide::Right));
131
132 // Compute the new 𝒒ₖ₊₁ vector:
133 // ----------------------
134 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
135 // 𝒒ₖ₊₁ ← 𝓟(𝒛₀ ← 𝓐𝒒ₖ),
136 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
137 // 𝑗 ← 𝘍𝘭𝘦𝘹𝘪𝘣𝘭𝘦 ? 𝑘 : 𝟢,
138 // 𝒒ₖ₊₁ ← 𝓐(𝒛ⱼ ← 𝓟𝒒ₖ),
139 // 𝗲𝗹𝘀𝗲:
140 // 𝒒ₖ₊₁ ← 𝓐𝒒ₖ,
141 // 𝗲𝗻𝗱 𝗶𝗳
142 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑘 𝗱𝗼:
143 // 𝐻ᵢₖ ← <𝒒ₖ₊₁⋅𝒒ᵢ>,
144 // 𝒒ₖ₊₁ ← 𝒒ₖ₊₁ - 𝐻ᵢₖ⋅𝒒ᵢ,
145 // 𝗲𝗻𝗱 𝗳𝗼𝗿
146 // 𝐻ₖ₊₁,ₖ ← ‖𝒒ₖ₊₁‖,
147 // 𝒒ₖ₊₁ ← 𝒒ₖ₊₁/𝐻ₖ₊₁,ₖ.
148 // ----------------------
149 if (left_pre) {
150 pre_op->mul(q_vecs_[k + 1], z_vecs_[0], lin_op, q_vecs_[k]);
151 } else if (right_pre) {
152 const size_t j{Flexible ? k : 0};
153 lin_op.mul(q_vecs_[k + 1], z_vecs_[j], *pre_op, q_vecs_[k]);
154 } else {
155 lin_op.mul(q_vecs_[k + 1], q_vecs_[k]);
156 }
157 for (size_t i = 0; i <= k; ++i) {
158 H_(i, k) = dot_product(q_vecs_[k + 1], q_vecs_[i]);
159 q_vecs_[k + 1] -= H_(i, k) * q_vecs_[i];
160 }
161 H_(k + 1, k) = norm_2(q_vecs_[k + 1]);
162 q_vecs_[k + 1] /= H_(k + 1, k);
163
164 // Eliminate the last element in 𝐻
165 // and and update the rotation matrix:
166 // ----------------------
167 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑘 - 𝟣 𝗱𝗼:
168 // 𝜒 ← 𝑐𝑠ᵢ⋅𝐻ᵢₖ + 𝑠𝑛ᵢ⋅𝐻ᵢ₊₁,ₖ,
169 // 𝐻ᵢ₊₁,ₖ ← -𝑠𝑛ᵢ⋅𝐻ᵢₖ + 𝑐𝑠ᵢ⋅𝐻ᵢ₊₁,ₖ,
170 // 𝐻ᵢₖ ← 𝜒,
171 // 𝗲𝗻𝗱 𝗳𝗼𝗿
172 // 𝑐𝑠ₖ, 𝑠𝑛ₖ ← 𝘚𝘺𝘮𝘖𝘳𝘵𝘩𝘰(𝐻ₖₖ, 𝐻ₖ₊₁,ₖ),
173 // 𝐻ₖₖ ← 𝑐𝑠ₖ⋅𝐻ₖₖ + 𝑠𝑛ₖ⋅𝐻ₖ₊₁,ₖ,
174 // 𝐻ₖ₊₁,ₖ ← 𝟢.
175 // ----------------------
176 for (size_t i = 0; i < k; ++i) {
177 const real_t chi = cs_(i) * H_(i, k) + sn_(i) * H_(i + 1, k);
178 H_(i + 1, k) = -sn_(i) * H_(i, k) + cs_(i) * H_(i + 1, k);
179 H_(i, k) = chi;
180 }
181 std::tie(cs_(k), sn_(k), std::ignore) = sym_ortho(H_(k, k), H_(k + 1, k));
182 H_(k, k) = cs_(k) * H_(k, k) + sn_(k) * H_(k + 1, k);
183 H_(k + 1, k) = 0.0;
184
185 // Update the 𝛽-solution and the residual norm:
186 // ----------------------
187 // 𝛽ₖ₊₁ ← -𝑠𝑛ₖ⋅𝛽ₖ, 𝛽ₖ ← 𝑐𝑠ₖ⋅𝛽ₖ.
188 // ----------------------
189 beta_(k + 1) = -sn_(k) * beta_(k), beta_(k) *= cs_(k);
190
191 return std::abs(beta_(k + 1));
192 }
193
194 void inner_finalize(Vector& x_vec, const Vector& b_vec,
195 const Operator<Vector>& lin_op,
196 const Preconditioner<Vector>* pre_op) override {
197 const size_t k = this->inner_iteration;
198
199 const bool right_pre =
200 (pre_op != nullptr) &&
201 (Flexible || (this->pre_side == PreconditionerSide::Right));
202
203 // Finalize the 𝛽-solution:
204 // ----------------------
205 // 𝛽₀:ₖ ← (𝐻₀:ₖ,₀:ₖ)⁻¹𝛽₀:ₖ.
206 // ----------------------
207 for (size_t i = k; i != SIZE_MAX; --i) {
208 for (size_t j{i + 1}; j <= k; ++j) {
209 beta_(i) -= H_(i, j) * beta_(j);
210 }
211 beta_(i) /= H_(i, i);
212 }
213
214 // Compute the 𝒙-solution:
215 // ----------------------
216 // 𝗶𝗳 𝗻𝗼𝘁 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
217 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑘 𝗱𝗼:
218 // 𝒙 ← 𝒙 + 𝛽ᵢ⋅𝒒ᵢ.
219 // 𝗲𝗻𝗱 𝗳𝗼𝗿
220 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘍𝘭𝘦𝘹𝘪𝘣𝘭𝘦:
221 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑘 𝗱𝗼:
222 // 𝒙 ← 𝒙 + 𝛽ᵢ⋅𝒛ᵢ.
223 // 𝗲𝗻𝗱 𝗳𝗼𝗿
224 // 𝗲𝗹𝘀𝗲:
225 // 𝒒₀ ← 𝛽₀⋅𝒒₀,
226 // 𝗳𝗼𝗿 𝑖 = 𝟣, 𝑘 𝗱𝗼:
227 // 𝒒₀ ← 𝒒₀ + 𝛽ᵢ⋅𝒒ᵢ,
228 // 𝗲𝗻𝗱 𝗳𝗼𝗿
229 // 𝒛₀ ← 𝓟𝒒₀,
230 // 𝒙 ← 𝒙 + 𝒛₀.
231 // 𝗲𝗻𝗱 𝗶𝗳
232 // ----------------------
233 if (!right_pre) {
234 for (size_t i = 0; i <= k; ++i) {
235 x_vec += beta_(i) * q_vecs_[i];
236 }
237 } else if constexpr (Flexible) {
238 for (size_t i = 0; i <= k; ++i) {
239 x_vec += beta_(i) * z_vecs_[i];
240 }
241 } else {
242 q_vecs_[0] *= beta_(0);
243 for (size_t i = 1; i <= k; ++i) {
244 q_vecs_[0] += beta_(i) * q_vecs_[i];
245 }
246 pre_op->mul(z_vecs_[0], q_vecs_[0]);
247 x_vec += z_vecs_[0];
248 }
249 }
250
251protected:
252
253 BaseGmresSolver() = default;
254
255}; // class BaseGmresSolver

259/// @brief The GMRES (Generalized Minimal Residual) linear operator equation
260/// solver.
261///
262/// GMRES is typically more robust than the BiCG type solvers, but it may be
263/// slower than the BiCG solvers for the well-conditioned moderate sized
264/// problems.
265///
266/// GMRES is algebraically equivalent to MINRES method in the self-adjoint
267/// operator unpreconditioned case, however, the need for restarts may lead to
268/// the much slower GMRES convergence rate.
269///
270/// GMRES may be applied to the singular problems, and the square least squares
271/// problems, although, similarly to MINRES, convergeance to minimum norm
272/// solution is not guaranteed.
273///
274/// References:
275/// @verbatim
276/// [1] Saad, Yousef and Martin H. Schultz.
277/// “GMRES: A generalized minimal residual algorithm for solving
278/// nonsymmetric linear systems.”
279/// SIAM J. Sci. Stat. Comput., 7:856–869, 1986.
280/// @endverbatim
281template<legacy_vector_like Vector>
282class GmresSolver final :
283 public BaseGmresSolver<Vector, false> {}; // class GmresSolver
284
285/// @brief The FGMRES (Flexible Generalized Minimal Residual) linear operator
286/// equation solver.
287///
288/// FGMRES is typically more robust than the BiCG type solvers, but it may be
289/// slower than the BiCG solvers for the well-conditioned moderate sized
290/// problems.
291///
292/// FGMRES does the same amount of operations per iteration as GMRES, but also
293/// allows usage of the variable (or flexible) preconditioners with the price of
294/// doubleing of the memory usage. For the static preconditioners, FGMRES
295/// requires one preconditioner-vector product less than GMRES.
296/// FGMRES supports only the right preconditioning.
297///
298/// FGMRES may be applied to the singular problems, and the square least squares
299/// problems, although, similarly to MINRES, convergeance to minimum norm
300/// solution is not guaranteed.
301///
302/// References:
303/// @verbatim
304/// [1] Saad, Yousef.
305/// “A Flexible Inner-Outer Preconditioned GMRES Algorithm.”
306/// SIAM J. Sci. Comput. 14 (1993): 461-469.
307/// @endverbatim
308template<legacy_vector_like Vector>
309class FgmresSolver final :
310 public BaseGmresSolver<Vector, true> {}; // class FgmresSolver
