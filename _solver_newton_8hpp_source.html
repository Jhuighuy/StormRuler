37/// @brief The Newton method nonlinear operator equation solver.
38///
39/// The classical Newton iterations are based on the linearization of 𝓐(𝒙)
40/// near 𝒙:
41///
42/// 𝓐(𝒙̂) ≈ 𝓐(𝒙) + [∂𝓐(𝒙)/∂𝒙](𝒙̂ - 𝒙) = 𝒃,
43///
44/// or, alternatively:
45///
46/// [∂𝓐(𝒙)/∂𝒙]𝒕 = 𝒓, 𝒕 = 𝒙̂ - 𝒙, 𝒓 = 𝒃 - 𝓐(𝒙)
47///
48/// where 𝒙 and 𝒙̂ are the current and updated solution vectors. Therefore, a
49/// linear equation has to be solved on each iteration, linear operator
50/// 𝓙(𝒙) ≈ ∂𝓐(𝒙)/∂𝒙 for computing Jacobian-vector products is required.
51///
52/// References:
53/// @verbatim
54/// [1] ???
55/// @endverbatim

79/// @brief The first-order JFNK (Jacobian free-Newton-Krylov) nonlinear operator
80/// equation solver.
81///
82/// For the @c Newton iterations, computing of the Jacobian-vector products
83/// 𝒛 = 𝓙(𝒙)𝒚, where 𝓙(𝒙) ≈ ∂𝓐(𝒙)/∂𝒙 is required. Consider the expansion:
84///
85/// 𝓐(𝒙 + 𝛿⋅𝒚) = 𝓐(𝒙) + 𝛿⋅[∂𝓐(𝒙)/∂𝒙]𝒚 + 𝓞(𝛿²),
86///
87/// where 𝛿 is some small number. Therefore,
88///
89/// 𝓙(𝒙)𝒚 = [𝓐(𝒙 + 𝛿⋅𝒚) - 𝓐(𝒙)]/𝛿 = [∂𝓐(𝒙)/∂𝒙]𝒚 + 𝓞(𝛿).
90///
91/// Expression above may be used as the formula for computing the (approximate)
92/// Jacobian-vector products. Parameter 𝛿 is commonly defined as [1]:
93///
94/// 𝛿 = 𝜇⋅‖𝒚‖⁺, 𝜇 = (𝜀ₘ)¹ᐟ²⋅(1 + ‖𝒙‖)¹ᐟ²,
95///
96/// where 𝜀ₘ is the machine roundoff, ‖𝒚‖⁺ is the pseudo-inverse to ‖𝒚‖.
97///
98/// References:
99/// @verbatim
100/// [1] Liu, Wei, Lilun Zhang, Ying Zhong, Yongxian Wang,
101/// Yonggang Che, Chuanfu Xu and Xinghua Cheng.
102/// “CFD High-order Accurate Scheme JFNK Solver.”
103/// Procedia Engineering 61 (2013): 9-15.
104/// @endverbatim
105template<legacy_vector_like Vector>
106class JfnkSolver final : public IterativeSolver<Vector>
107{
108private:
109
110 Vector s_vec_, t_vec_, r_vec_, w_vec_;
111
112 real_t init(const Vector& x_vec, const Vector& b_vec,
113 const Operator<Vector>& any_op,
114 const Preconditioner<Vector>* pre_op) override
115 {
116 s_vec_.assign(x_vec, false);
117 t_vec_.assign(x_vec, false);
118 r_vec_.assign(x_vec, false);
119 w_vec_.assign(x_vec, false);
120
121 // Initialize:
122 // ----------------------
123 // 𝒘 ← 𝓐(𝒙),
124 // 𝒓 ← 𝒃 - 𝒘.
125 // ----------------------
126 any_op.mul(w_vec_, x_vec);
127 r_vec_ <<= b_vec - w_vec_;
128
129 return norm_2(r_vec_);
130 }
131
132 real_t iterate(Vector& x_vec, const Vector& b_vec,
133 const Operator<Vector>& any_op,
134 const Preconditioner<Vector>* pre_op) override
135 {
136 // Solve the Jacobian equation:
137 // ----------------------
138 // 𝜇 ← (𝜀ₘ)¹ᐟ²⋅(1 + ‖𝒙‖)]¹ᐟ²,
139 // 𝒕 ← 𝒓,
140 // 𝒕 ← 𝓙(𝒙)⁻¹𝒓.
141 // ----------------------
142 static const real_t sqrt_of_epsilon =
143 std::sqrt(std::numeric_limits<real_t>::epsilon());
144 const real_t mu = sqrt_of_epsilon * sqrt(1.0 + norm_2(x_vec));
145 t_vec_ <<= r_vec_;
146 {
147 BiCgStabSolver<Vector> solver{};
148 solver.absolute_error_tolerance = 1.0e-8;
149 solver.relative_error_tolerance = 1.0e-8;
150 auto op = make_operator<Vector>([&](Vector& z_vec, const Vector& y_vec) {
151 // Compute the Jacobian-vector product:
152 // ----------------------
153 // 𝛿 ← 𝜇⋅‖𝒚‖⁺,
154 // 𝒔 ← 𝒙 + 𝛿⋅𝒚,
155 // 𝒛 ← 𝓐(𝒔),
156 // 𝒛 ← 𝛿⁺⋅𝒛 - 𝛿⁺⋅𝒘.
157 // ----------------------
158 const real_t delta = safe_divide(mu, norm_2(y_vec));
159 s_vec_ <<= x_vec + delta * y_vec;
160 any_op.mul(z_vec, s_vec_);
161 const real_t delta_inverse = safe_divide(1.0, delta);
162 z_vec <<= delta_inverse * (z_vec - w_vec_);
163 });
164 solver.solve(t_vec_, r_vec_, *op);
165 }
166
167 // Update the solution and the residual:
168 // ----------------------
169 // 𝒙 ← 𝒙 + 𝒕,
170 // 𝒘 ← 𝓐(𝒙),
171 // 𝒓 ← 𝒃 - 𝒘.
172 // ----------------------
173 x_vec += t_vec_;
174 any_op.mul(w_vec_, x_vec);
175 r_vec_ <<= b_vec - w_vec_;
176
177 return norm_2(r_vec_);
178 }
179
180}; // class JfnkSolver
