41/// @brief Bind the first arguments of the function to values.
42template<class Func, class... FirstArgs>
43 requires std::is_object_v<Func>
44class BindFirst final {
45private:
46
47 STORM_NO_UNIQUE_ADDRESS Func _func;
48 STORM_NO_UNIQUE_ADDRESS std::tuple<FirstArgs...> _first_args;
49
50public:
51
52 constexpr explicit BindFirst(Func func, FirstArgs... first_args)
53 : _func{std::move(func)}, _first_args{std::move(first_args)...} {}
54
55 template<class... RestArgs>
56 requires std::regular_invocable<Func, FirstArgs..., RestArgs...>
57 constexpr decltype(auto) operator()(RestArgs&&... rest_args) const {
58 const auto call = [&](const FirstArgs&... first_args) {
59 return _func(first_args..., std::forward<RestArgs>(rest_args)...);
60 };
61 return std::apply(call, _first_args);
62 }
63
64}; // class BindFirst

91/// @brief Compose multiple functions: @f$ f_{1} \circ f_{2} \circ \ldots @f$.
92template<class Func, class... RestFuncs>
93 requires std::is_object_v<Func> && (... && std::is_object_v<RestFuncs>)
94class Compose final {
95private:
96
97 STORM_NO_UNIQUE_ADDRESS Func _func;
98 STORM_NO_UNIQUE_ADDRESS Compose<RestFuncs...> _rest_funcs;
99
100public:
101
102 constexpr explicit Compose(Func func, RestFuncs... rest_funcs)
103 : _func{std::move(func)}, _rest_funcs{std::move(rest_funcs)...} {}
104
105 template<class... Args>
106 requires std::invocable<Compose<RestFuncs...>, Args...> &&
107 std::invocable<
108 Func, std::invoke_result_t<Compose<RestFuncs...>, Args...>>
109 constexpr decltype(auto) operator()(Args&&... args) const {
110 return _func(_rest_funcs(std::forward<Args>(args)...));
111 }
112
113}; // class Compose

115/// @brief Compose two functions: @f$ f_{1} \circ f_{2} @f$.
116template<class Func1, class Func2>
117 requires std::is_object_v<Func1> && std::is_object_v<Func2>
118class Compose<Func1, Func2> final {
119private:
120
121 STORM_NO_UNIQUE_ADDRESS Func1 func1_;
122 STORM_NO_UNIQUE_ADDRESS Func2 func2_;
123
124public:
125
126 constexpr Compose(Func1 func1, Func2 func2)
127 : func1_{std::move(func1)}, func2_{std::move(func2)} {}
128
129 template<class... Args>
130 requires std::invocable<Func2, Args...> && //
131 std::invocable<Func1, std::invoke_result_t<Func2, Args...>>
132 constexpr decltype(auto) operator()(Args&&... args) const {
133 return func1_(func2_(std::forward<Args>(args)...));
134 }
135
136}; // class Compose

171/// @brief Eye function.
172template<class Value>
173 requires std::is_object_v<Value>
174class Eye final {
175private:
176
177 STORM_NO_UNIQUE_ADDRESS Value _diagonal;
178 STORM_NO_UNIQUE_ADDRESS Value _off_diagonal;
179
180public:
181
182 constexpr Eye(Value diagonal, Value off_diagonal)
183 : _diagonal{std::move(diagonal)}, _off_diagonal{std::move(off_diagonal)} {
184 }
185
186 template<class Arg, class... RestArgs>
187 constexpr const Value& operator()(Arg&& arg, RestArgs&&... rest_args) const {
188 const bool on_diagonal = [&]() {
189 if constexpr (sizeof...(RestArgs) == 0) {
190 using Index = std::remove_cvref_t<Arg>;
191 static_assert(std::constructible_from<Index, size_t>);
192 return std::forward<Arg>(arg) == Index{0};
193 } else {
194 return ((arg == std::forward<RestArgs>(rest_args)) && ...);
195 }
196 }();
197 return on_diagonal ? _diagonal : _off_diagonal;
198 }
199
200}; // class Eye

295/// @brief Approximately equals function.
296class ApproxEqual final {
297private:
298
299 long double _tolerance;
300
301public:
302
303 constexpr explicit ApproxEqual(
304 long double tolerance = sqrt(std::numeric_limits<long double>::epsilon()))
305 : _tolerance{tolerance} {
306 STORM_ASSERT(tolerance > 0.0l, "Negative tolerance!");
307 }
308
309 template<class Arg1, class Arg2>
310 constexpr bool operator()(Arg1&& arg1, Arg2&& arg2) const {
311 const auto delta = static_cast<long double>(
312 abs(std::forward<Arg1>(arg1) - std::forward<Arg2>(arg2)));
313 return delta <= _tolerance;
314 }

316}; // class ApproxEqual
