39/// @brief Wrapper for @c std::array with strict indexing.
40/// @tparam Index Array index type.
41/// @tparam Value Array value type.
42/// @tparam Size Array size.
43template<index Index, class Value, size_t Size>
44 requires std::is_object_v<Value>
45class IndexedArray final : public std::array<Value, Size>
46{
47public:
48
49 /// @brief Element at @p index.
50 /// @{
51 constexpr Value& at(size_t) = delete;
52 constexpr const Value& at(size_t) const = delete;
53 constexpr Value& at(Index index)
54 {
55 return std::array<Value, Size>::at(static_cast<size_t>(index));
56 }
57 constexpr const Value& at(Index index) const
58 {
59 return std::array<Value, Size>::at(static_cast<size_t>(index));
60 }
61 /// @}
62
63 /// @brief Element at @p index.
64 /// @{
65 constexpr Value& operator[](size_t) = delete;
66 constexpr const Value& operator[](size_t) const = delete;
67 constexpr Value& operator[](Index index) noexcept
68 {
69 STORM_ASSERT_(index < Size, "Index is out of range!");
70 return std::array<Value, Size>::operator[](static_cast<size_t>(index));
71 }
72 constexpr const Value& operator[](Index index) const noexcept
73 {
74 STORM_ASSERT_(index < Size, "Index is out of range!");
75 return std::array<Value, Size>::operator[](static_cast<size_t>(index));
76 }
77 /// @}
78
79}; // class IndexedArray

83/// @brief Wrapper for @c std::vector with strict indexing.
84/// @tparam Index Vector index type.
85/// @tparam Value Vector value type.
86/// @tparam Allocator Vector allocator type.
87template<index Index, class Value, class Allocator = std::allocator<Value>>
88 requires std::is_object_v<Value> && std::is_object_v<Allocator>
89class IndexedVector final : public std::vector<Value, Allocator>
90{
91public:
92
93 /// @brief Construct an empty indexed vector.
94 constexpr IndexedVector() = default;
95 /// @brief Construct an empty indexed vector with a given @p allocator.
96 constexpr explicit IndexedVector(const Allocator& allocator) noexcept
97 : std::vector<Value, Allocator>(allocator)
98 {
99 }
100
101 /// @brief Construct an indexed vector with @p count copies of @p value.
102 constexpr IndexedVector(size_t count, const Value& value,
103 const Allocator& allocator = Allocator{})
104 : std::vector<Value, Allocator>(count, value, allocator)
105 {
106 }
107 /// @brief Construct an indexed array with @p count default-inserted values.
108 constexpr explicit IndexedVector(size_t count,
109 const Allocator& allocator = Allocator{})
110 : std::vector<Value, Allocator>(count, allocator)
111 {
112 }
113
114 /// @brief Construst an indexex vector with the iterator range.
115 /// @param allocator Allocator instance.
116 template<class Iterator>
117 constexpr IndexedVector(Iterator first, Iterator last,
118 const Allocator& allocator = Allocator{})
119 : std::vector<Value, Allocator>(first, last, allocator)
120 {
121 }
122
123 /// @brief Copy-construct an indexed vector.
124 constexpr IndexedVector(const IndexedVector& other)
125 : std::vector<Value, Allocator>(other)
126 {
127 }
128 /// @brief Copy-construct an indexed vector.
129 /// @param allocator Allocator instance.
130 constexpr IndexedVector(const IndexedVector& other,
131 const Allocator& allocator)
132 : std::vector<Value, Allocator>(other, allocator)
133 {
134 }
135
136 /// @brief Move-construct an indexed vector.
137 constexpr IndexedVector(IndexedVector&& other) noexcept
138 : std::vector<Value, Allocator>(
139 static_cast<std::vector<Value, Allocator>&&>(other))
140 {
141 }
142 /// @brief Move-construct an indexed vector.
143 /// @param allocator Allocator instance.
144 constexpr IndexedVector(IndexedVector&& other, const Allocator& allocator)
145 : std::vector<Value, Allocator>(
146 static_cast<std::vector<Value, Allocator>&&>(other), allocator)
147 {
148 }
149
150 /// @brief Construct an indexed vector with an initializer list @p list.
151 /// @param allocator Allocator instance.
152 constexpr IndexedVector(std::initializer_list<Value> list,
153 const Allocator& allocator = Allocator{})
154 : std::vector<Value, Allocator>(list, allocator)
155 {
156 }
157
158 /// @brief Copy-assignment operator.
159 constexpr IndexedVector& operator=(const IndexedVector& other)
160 {
161 std::vector<Value, Allocator>::operator=(other);
162 return *this;
163 }
164
165 /// @brief Move-assignment operator.
166 constexpr IndexedVector& operator=(IndexedVector&& other) noexcept(
167 noexcept(std::allocator_traits<
168 Allocator>::propagate_on_container_move_assignment::value ||
169 std::allocator_traits<Allocator>::is_always_equal::value))
170 {
171 std::vector<Value, Allocator>::operator=(
172 static_cast<std::vector<Value, Allocator>&&>(other));
173 return *this;
174 }
175
176 /// @brief Assign an initializer list @p list.
177 constexpr IndexedVector& operator=(std::initializer_list<Value> list)
178 {
179 return std::vector<Value, Allocator>::operator=(list), *this;
180 }
181
182 /// @brief Swap the indexed array contents.
183 constexpr void swap(IndexedVector& other) noexcept(noexcept(
184 std::allocator_traits<Allocator>::propagate_on_container_swap::value ||
185 std::allocator_traits<Allocator>::is_always_equal::value))
186 {
187 std::vector<Value, Allocator>::swap(other);
188 }
189
190 /// @brief Element at @p index.
191 /// @{
192 constexpr Value& at(size_t) = delete;
193 constexpr const Value& at(size_t) const = delete;
194 constexpr Value& at(Index index)
195 {
196 return std::vector<Value, Allocator>::at(static_cast<size_t>(index));
197 }
198 constexpr const Value& at(Index index) const
199 {
200 return std::vector<Value, Allocator>::at(static_cast<size_t>(index));
201 }
202 /// @}
203
204 /// @brief Element at @p index.
205 /// @{
206 constexpr Value& operator[](size_t) = delete;
207 constexpr const Value& operator[](size_t) const = delete;
208 constexpr Value& operator[](Index index) noexcept
209 {
210 STORM_ASSERT_(index < this->size(), "Index is out of range!");
211 return std::vector<Value, Allocator>::operator[](
212 static_cast<size_t>(index));
213 }
214 constexpr const Value& operator[](Index index) const noexcept
215 {
216 STORM_ASSERT_(index < this->size(), "Index is out of range!");
217 return std::vector<Value, Allocator>::operator[](
218 static_cast<size_t>(index));
219 }
220 /// @}
221
222}; // class IndexedVector
