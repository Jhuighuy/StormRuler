35/// @brief Matrix product view.
36template<matrix_view Matrix1, matrix_view Matrix2>
37class MatrixProductView final :
38 public MatrixViewInterface<MatrixProductView<Matrix1, Matrix2>> {
39private:
40
41 STORM_NO_UNIQUE_ADDRESS Matrix1 mat1_;
42 STORM_NO_UNIQUE_ADDRESS Matrix2 mat2_;
43
44public:
45
46 /// @brief Construct a matrix product view.
47 constexpr explicit MatrixProductView(Matrix1 mat1, Matrix2 mat2)
48 : mat1_{std::move(mat1)}, mat2_{std::move(mat2)} {
49 STORM_ASSERT(num_cols(mat1_) == num_rows(mat2),
50 "Shapes of the matrix product arguments are invalid!");
51 }
52
53 /// @brief Get the matrix shape.
54 constexpr auto shape() const noexcept {
55 return std::array{num_rows(mat1_), num_cols(mat2_)};
56 }
57
58 /// @brief Get the matrix element at @p indices.
59 constexpr auto operator()(size_t row_index, size_t col_index) const noexcept {
60 STORM_ASSERT(in_range(shape(), row_index, col_index),
61 "Indices are out of range!");
62 // This is a default very slow generic implementation.
63 auto val = mat1_(row_index, 0) * mat2_(0, col_index);
64 const size_t cross_size = num_cols(mat1_);
65 for (size_t cross_index = 1; cross_index < cross_size; ++cross_index) {
66 val += mat1_(row_index, cross_index) * mat2_(cross_index, col_index);
67 }
68 return val;
69 }
70
71}; // class MatrixProductView
72
73template<class Matrix1, class Matrix2>
74MatrixProductView(Matrix1&&, Matrix2&&)
75 -> MatrixProductView<matrix_view_t<Matrix1>, matrix_view_t<Matrix2>>;

77/// @brief Multiply the matrices @p mat1 and @p mat2.
78template<viewable_matrix Matrix1, viewable_matrix Matrix2>
79constexpr auto matmul(Matrix1&& mat1, Matrix2&& mat2) {
80 return MatrixProductView(std::forward<Matrix1>(mat1),
81 std::forward<Matrix2>(mat2));
82}

86/// @brief Cross product of the 3x1 matrices view.
87template<matrix_view Matrix1, matrix_view Matrix2>
88class CrossProductView :
89 MatrixViewInterface<CrossProductView<Matrix1, Matrix2>> {
90private:
91
92 STORM_NO_UNIQUE_ADDRESS Matrix1 mat1_;
93 STORM_NO_UNIQUE_ADDRESS Matrix2 mat2_;
94
95public:
96
97 /// @brief Construct a matrix product view.
98 constexpr explicit CrossProductView(Matrix1 mat1, Matrix2 mat2)
99 : mat1_{std::move(mat1)}, mat2_{std::move(mat2)} {
100 STORM_ASSERT((mat1_.shape() == shape() && mat2_.shape() == shape()),
101 "Matrices of shape 3x1 are expected!");
102 }
103
104 /// @brief Get the matrix shape.
105 constexpr static auto shape() noexcept {
106 return std::array{3_sz, 1_sz};
107 }
108
109 /// @brief Get the matrix element at @p indices.
110 constexpr auto operator()(size_t row_index, size_t col_index) const noexcept {
111 STORM_ASSERT(in_range(shape(), row_index, col_index),
112 "Indices are out of range!");
113 switch (row_index) {
114 case 0: return mat1_(1, 0) * mat2_(2, 0) - mat1_(2, 0) * mat2_(1, 0);
115 case 1: return mat1_(2, 0) * mat2_(0, 0) - mat1_(0, 0) * mat2_(2, 0);
116 case 2: return mat1_(0, 0) * mat2_(1, 0) - mat1_(1, 0) * mat2_(0, 0);
117 default: STORM_UNREACHABLE();
118 }
119 }
120
121}; // class CrossProductView

127/// @brief Cross product of the 3x1 matrices @p mat1 and @p mat2.
128template<viewable_matrix Matrix1, viewable_matrix Matrix2>
129constexpr auto cross_product(Matrix1&& mat1, Matrix2&& mat2) {
130 return CrossProductView(std::forward<Matrix1>(mat1),
131 std::forward<Matrix2>(mat2));
132}
