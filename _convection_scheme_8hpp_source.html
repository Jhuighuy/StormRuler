57/// @brief Piecewise-constant upwind convection scheme.
58template<mesh Mesh, class FluxScheme>
59 requires std::is_object_v<FluxScheme>
60class UpwindConvectionScheme final : public iConvectionScheme {
61private:
62
63 const Mesh* p_mesh_;
64 STORM_NO_UNIQUE_ADDRESS_ FluxScheme flux_scheme_;
65
66public:
67
68 /// @brief Construct the first order upwind convection scheme.
69 constexpr explicit UpwindConvectionScheme(const Mesh& mesh,
70 FluxScheme flux_scheme) noexcept
71 : p_mesh_{&mesh}, flux_scheme_{std::move(flux_scheme)} {}
72
73 void get_cell_convection(CellField<Mesh, real_t, 5>& div_f,
74 const CellField<Mesh, real_t, 5>& u) const final {
75 (*this)(div_f, u);
76 }
77
78 /// @brief Compute the first order upwind nonlinear convection.
79 template<class Real, size_t NumVars>
80 void operator()(CellField<Mesh, Real, NumVars>& div_f,
81 const CellField<Mesh, Real, NumVars>& u) const noexcept {
82 // Compute the fluxes for the interior faces.
83 std::ranges::for_each(p_mesh_->interior_faces(), [&](FaceView<Mesh> face) {
84 const CellView<Mesh> cell_inner = face.inner_cell();
85 const CellView<Mesh> cell_outer = face.outer_cell();
86
87 // Compute the flux.
88 const auto flux =
89 flux_scheme_(face.normal(), u[cell_outer], u[cell_inner]);
90 div_f[cell_inner] += (face.area() / cell_inner.volume()) * flux;
91 div_f[cell_outer] -= (face.area() / cell_outer.volume()) * flux;
92 });
93
94 // Compute the fluxes for the boundary faces.
95 for (const auto& [label, bc] : *bcs_) {
96 std::ranges::for_each(p_mesh_->faces(label), [&](FaceView<Mesh> face) {
97 const CellView<Mesh> cell_inner = face.inner_cell();
98 Subfield<Real, NumVars> u_outer{};
99 bc->get_ghost_state(face.normal(), face.center(),
100 u[face.inner_cell()].data(), u_outer.data());
101
102 // Compute the flux.
103 const auto flux = flux_scheme_(face.normal(), u_outer, u[cell_inner]);
104 div_f[cell_inner] += (face.area() / cell_inner.volume()) * flux;
105 });
106 }
107 }
108
109}; // class UpwindConvectionScheme

113/// @brief Piecewise-linear upwind convection scheme.
114template<mesh Mesh, class FluxScheme, //
115 class GradientScheme, class GradientLimiterScheme>
116 requires std::is_object_v<FluxScheme> && std::is_object_v<GradientScheme> &&
117 std::is_object_v<GradientLimiterScheme>
118class LinearUpwindConvectionScheme final : public iConvectionScheme {
119public:
120
121 const Mesh* p_mesh_;
122 STORM_NO_UNIQUE_ADDRESS_ FluxScheme flux_scheme_;
123 STORM_NO_UNIQUE_ADDRESS_ GradientScheme gradient_scheme_;
124 STORM_NO_UNIQUE_ADDRESS_ GradientLimiterScheme gradient_limiter_scheme_;
125
126public:
127
128 /// @brief Construct the second order upwind convection scheme.
129 constexpr explicit LinearUpwindConvectionScheme(
130 const Mesh& mesh, FluxScheme flux_scheme, GradientScheme gradient_scheme,
131 GradientLimiterScheme gradient_limiter_scheme) noexcept
132 : p_mesh_{&mesh}, flux_scheme_{flux_scheme}, //
133 gradient_scheme_{std::move(gradient_scheme)},
134 gradient_limiter_scheme_{std::move(gradient_limiter_scheme)} {}
135
136 void get_cell_convection(CellField<Mesh, real_t, 5>& div_f,
137 const CellField<Mesh, real_t, 5>& u) const final {
138 (*this)(div_f, u);
139 }
140
141 /// @brief Compute the second-order upwind nonlinear convection.
142 template<class Real, size_t NumVars>
143 void operator()(CellField<Mesh, Real, NumVars>& div_f,
144 const CellField<Mesh, Real, NumVars>& u) const noexcept {
145 // Compute the gradients.
146 CellVectorField<Mesh, Real, NumVars> grad_u(p_mesh_->num_cells());
147 CellField<Mesh, Real, NumVars> lim_u(p_mesh_->num_cells());
148 gradient_scheme_(grad_u, u);
149 gradient_limiter_scheme_(lim_u, u, grad_u);
150 std::ranges::for_each(p_mesh_->interior_cells(), [&](CellView<Mesh> cell) {
151 grad_u[cell] *= lim_u[cell];
152 });
153
154 // Compute the fluxes for the interior faces.
155 std::ranges::for_each(p_mesh_->interior_faces(), [&](FaceView<Mesh> face) {
156 const CellView<Mesh> cell_inner = face.inner_cell();
157 const CellView<Mesh> cell_outer = face.outer_cell();
158
159 // Reconstruct the face values.
160 const auto dr_outer = face.center() - cell_outer.center();
161 const auto dr_inner = face.center() - cell_inner.center();
162 Subfield<Real, NumVars> u_outer{}, u_inner{};
163 for (size_t i = 0; i < NumVars; ++i) {
164 u_inner[i] =
165 u[cell_inner][i] + glm::dot(grad_u[cell_inner][i], dr_inner);
166 u_outer[i] =
167 u[cell_outer][i] + glm::dot(grad_u[cell_outer][i], dr_outer);
168 }
169
170 // Compute the flux.
171 const auto flux = flux_scheme_(face.normal(), u_outer, u_inner);
172 div_f[cell_inner] += (face.area() / cell_inner.volume()) * flux;
173 div_f[cell_outer] -= (face.area() / cell_outer.volume()) * flux;
174 });
175
176 // Compute the fluxes for the boundary faces.
177 for (const auto& [label, bc] : *bcs_) {
178 std::ranges::for_each(p_mesh_->faces(label), [&](FaceView<Mesh> face) {
179 const CellView<Mesh> cell_inner = face.inner_cell();
180
181 // Reconstruct the face values.
182 const auto dr_inner = face.center() - cell_inner.center();
183 Subfield<Real, NumVars> u_outer{}, u_inner{};
184 for (size_t i = 0; i < NumVars; ++i) {
185 u_inner[i] =
186 u[cell_inner][i] + glm::dot(grad_u[cell_inner][i], dr_inner);
187 }
188 bc->get_ghost_state(face.normal(), face.center(), //
189 u_inner.data(), u_outer.data());
190
191 // Compute the flux.
192 const auto flux = flux_scheme_(face.normal(), u_outer, u_inner);
193 div_f[cell_inner] += (face.area() / cell_inner.volume()) * flux;
194 });
195 }
196 }
197
198}; // class UpwindConvectionScheme
