58/// @brief Assign the matrices.
59/// @{
60template<output_matrix OutMatrix, matrix Matrix>
61constexpr OutMatrix& assign(OutMatrix&& out_mat, Matrix&& mat) noexcept
62{
63 STORM_ASSERT_(out_mat.shape() == mat.shape(), "Matrix shapes do not match!");
64 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
65 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
66 out_mat(row_index, col_index) = mat(row_index, col_index);
67 }
68 }
69 return out_mat;
70}

96/// @brief Fill the matrix @p out_mat with a scalar @p scal.
97template<matrix OutMatrix, std::copyable Scalar>
98 requires (!matrix<Scalar>)
99constexpr OutMatrix& fill(OutMatrix& out_mat, Scalar scal)
100{
101 return assign(
102 [scal = std::move(scal)](auto& out_elem) noexcept { out_elem = scal; },
103 out_mat);

106/// @brief Fill the matrix @p out_mat elements with
107/// the uniformly-distributed random numbers.
108/// @note This is a sequential operation!
109template<output_matrix Matrix>
110 requires real_matrix<Matrix>
111STORM_CPP23_CONSTEXPR_ Matrix&
112fill_randomly(Matrix&& out_mat, //
113 matrix_element_t<Matrix> min = 0,
114 matrix_element_t<Matrix> max = 1) noexcept
115{
116 static std::mt19937_64 random_engine{}; // NOSONAR
117 std::uniform_real_distribution distribution{min, max};
118 for (size_t row_index = 0; row_index < num_rows(out_mat); ++row_index) {
119 for (size_t col_index = 0; col_index < num_cols(out_mat); ++col_index) {
120 out_mat(row_index, col_index) = distribution(random_engine);
121 }
122 }
123 return out_mat;
124}

128/// @brief Reduce the matrix @p mat coefficients to a single value.
129/// @param init Initial reduction value.
130/// @param reduce_func Reduction function.
131/// @todo Restrictions!
132/// @{
133template<class Value, class ReduceFunc, matrix Matrix>
134constexpr auto reduce(Value init, ReduceFunc reduce_func, Matrix&& mat)
135{
136 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
137 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
138 init = reduce_func(init, mat(row_index, col_index));
139 }
140 }
141 return init;
142}
143template<class Value, class ReduceFunc, class Func, //
144 matrix Matrix, matrix... RestMatrices>
145 requires compatible_matrices_v<Matrix, RestMatrices...>
146constexpr auto reduce(Value init, ReduceFunc reduce_func, Func func,
147 Matrix&& mat, RestMatrices&&... mats) noexcept
148{
149 STORM_ASSERT_((mat.shape() == mats.shape()) && ...,
150 "Matrix shapes doesn't match!");
151 for (size_t row_index = 0; row_index < num_rows(mat); ++row_index) {
152 for (size_t col_index = 0; col_index < num_cols(mat); ++col_index) {
153 init = reduce_func(
154 init, func(mat(row_index, col_index), mats(row_index, col_index)...));
155 }
156 }
157 return init;
158}

163/// @brief Sum the matrix @p mat elements.
164template<real_or_complex_matrix Matrix>
165constexpr auto sum(Matrix&& mat)
166{
167 return reduce(matrix_element_t<Matrix>{0.0}, std::plus{},
168 std::forward<Matrix>(mat));

211/// @brief Element-wise matrix @p mat @f$ L_{1} @f$-norm.
212template<numeric_matrix Matrix>
213constexpr auto norm_1(Matrix&& mat)
214{
215 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
216 static_assert(real_type<Result>,
217 "Absolute value of the matrix element should be of real type!");
218 return reduce(Result{}, Add{}, Abs{}, std::forward<Matrix>(mat));
219}

221/// @brief Element-wise matrix @p mat @f$ L_{2} @f$-norm.
222/// @{
223template<numeric_matrix Matrix>
224constexpr auto norm_2_2(Matrix&& mat)
225{
226 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
227 static_assert(real_type<Result>,
228 "Absolute value of the matrix element should be of real type!");
229 return reduce(Result{}, Add{}, AbsSquared{}, std::forward<Matrix>(mat));

238/// @brief Element-wise matrix @p mat @f$ L_{p} @f$-norm.
239/// @{
240template<numeric_matrix Matrix, numeric_type Number>
241constexpr auto norm_p_p(Matrix&& mat, Number p)
242{
243 STORM_ASSERT_(p > 0, "Invalid p-norm parameter!");
244 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
245 static_assert(real_type<Result>,
246 "Absolute value of the matrix element should be of real type!");
247 return reduce(Result{}, Add{}, Compose{Abs{}, BindLast{Pow{}, std::move(p)}},
248 std::forward<Matrix>(mat));
249}
250template<numeric_matrix Matrix, numeric_type Number>
251constexpr auto norm_p(Matrix&& mat, Number p)
252{
253 return pow(norm_p_p(std::forward<Matrix>(mat), p), 1.0 / p);
254}
255/// @}

257/// @brief Element-wise matrix @p mat @f$ L_{\infty} @f$-norm.
258template<numeric_matrix Matrix>
259constexpr auto norm_inf(Matrix&& mat)
260{
261 using Result = std::invoke_result_t<Abs, matrix_element_t<Matrix>>;
262 static_assert(real_type<Result>,
263 "Absolute value of the matrix element should be of real type!");
264 return reduce(Result{}, Max{}, Abs{}, std::forward<Matrix>(mat));
265}
