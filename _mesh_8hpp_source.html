74/// @brief Mesh entity view.
75template<mesh Mesh, index Index>
76class EntityView
77{
78private:
79
80 template<mesh, index>
81 friend class EntityView;
82
83 const Mesh* p_mesh_;
84 Index index_;
85
86protected:
87
88 /// @brief Construct an entity view with a @p mesh and @p index.
89 constexpr EntityView(const Mesh& mesh, Index index) noexcept
90 : p_mesh_{&mesh}, index_{index}
91 {
92 }
93
94 /// @brief Destroy the entity view.
95 constexpr ~EntityView() = default;
96
97public:
98
99 /// @brief Entity mesh.
100 /// @{
101 constexpr Mesh& mesh() noexcept
102 {
103 return *p_mesh_;
104 }
105 constexpr const Mesh& mesh() const noexcept
106 {
107 return *p_mesh_;
108 }
109 /// @}
110
111 /// @brief Entity index.
112 /// @{
113 constexpr Index index() const noexcept
114 {
115 return index_;
116 }
117 constexpr size_t index_sz() const noexcept
118 {
119 return static_cast<size_t>(index_);
120 }
121 /// @}
122
123 /// @brief Cast to entity index operator.
124 constexpr operator Index() const noexcept
125 {
126 return index_;
127 }
128
129 /// @brief Comparison operator.
130 constexpr auto operator<=>(const EntityView& other) const noexcept
131 {
132 STORM_ASSERT_(p_mesh_ == other.p_mesh_,
133 "Entities correspond to the different meshes!");
134 return index_ <=> other.index_;
135 }
136
137 /// @brief Entity label.
138 constexpr auto label() const noexcept
139 {
140 return mesh().label(index_);
141 }
142
143 /// @brief Range of the adjacent nodes.
144 constexpr auto nodes() const noexcept
145 {
146 return mesh().adjacent(index_, meta::type_v<NodeIndex>) |
147 detail_::to_node_view_(*p_mesh_);
148 }
149
150 /// @brief Range of the adjacent edges.
151 constexpr auto edges() const noexcept
152 {
153 return mesh().adjacent(index_, meta::type_v<EdgeIndex>) |
154 detail_::to_edge_view_(*p_mesh_);
155 }
156
157 /// @brief Range of the adjacent faces.
158 constexpr auto faces() const noexcept
159 {
160 return mesh().adjacent(index_, meta::type_v<FaceIndex<Mesh>>) |
161 detail_::to_face_view_(*p_mesh_);
162 }
163
164 /// @brief Range of the adjacent cells.
165 constexpr auto cells() const noexcept
166 {
167 return mesh().adjacent(index_, meta::type_v<CellIndex<Mesh>>) |
168 detail_::to_cell_view_(*p_mesh_);
169 }
170
171 /// @brief Iterate all the adjacent nodes.
172 template<std::invocable<NodeView<Mesh>> Func>
173 constexpr void for_each_node(Func func) const noexcept
174 {
175 std::ranges::for_each(nodes(), func);
176 }
177
178 /// @brief Iterate all the adjacent edges.
179 template<std::invocable<EdgeView<Mesh>> Func>
180 constexpr void for_each_edge(Func func) const noexcept
181 {
182 std::ranges::for_each(edges(), func);
183 }
184
185 /// @brief Iterate all the adjacent faces.
186 /// @{
187 template<std::invocable<FaceView<Mesh>> Func>
188 constexpr void for_each_face(Func func) const noexcept
189 {
190 std::ranges::for_each(faces(), func);
191 }
192 template<std::invocable<CellView<Mesh>, CellView<Mesh>> Func>
193 constexpr void for_each_face_cells(Func func) const noexcept
194 {
195 std::ranges::for_each(faces(), [&](FaceView<Mesh> face) {
196 func(face.inner_cell(), face.outer_cell());
197 });
198 }
199 /// @}
200
201 /// @brief Iterate all the adjacent cells.
202 template<std::invocable<CellView<Mesh>> Func>
203 constexpr void for_each_cell(Func func) const noexcept
204 {
205 std::ranges::for_each(cells(), func);
206 }
207
208}; // class EntityView

282 : EntityView<Mesh, FaceIndex<Mesh>>(mesh, index)
283 {
284 }
285
286 /// @brief Face shape type.
287 constexpr shapes::Type shape_type() const noexcept
288 {
289 return this->mesh().shape_type(this->index());
290 }
291
292 /// @brief Face area (or length in 2D).
293 constexpr real_t area() const noexcept
294 {
295 return this->mesh().volume(this->index());
296 }
297
298 /// @brief Face barycenter position.
299 constexpr auto center() const noexcept
300 {
301 return this->mesh().position(this->index());
302 }
303
304 /// @brief Face normal.
305 constexpr auto normal() const noexcept
306 {
307 return this->mesh().normal(this->index());
308 }
309
310 /// @brief Get the adjacent inner cell.
311 constexpr CellView<Mesh> inner_cell() const noexcept
312 {
313 STORM_ASSERT_(!this->cells().empty(),
314 "The face does not have an adjacent inner cell!");
315 return this->cells().front();
316 }
317
318 /// @brief Get the adjacent outer cell.
319 constexpr CellView<Mesh> outer_cell() const noexcept
320 {
321 STORM_ASSERT_(this->cells().size() == 2,
322 "The face does not have an adjacent outer cell!");
323 return this->cells().back();
324 }

381/// @brief CRTP interface to a mesh.
382template<crtp_derived Derived>
383class MeshInterface
384{
385private:
386
387 constexpr Derived& self_() noexcept
388 {
389 static_assert(std::derived_from<Derived, MeshInterface>);
390 return static_cast<Derived&>(*this);
391 }
392 constexpr const Derived& self_() const noexcept
393 {
394 return const_cast<MeshInterface&>(*this).self_();
395 }
396
397public:
398
399 /// @brief Number of the node labels.
400 constexpr size_t num_node_labels() const noexcept
401 {
402 return self_().num_labels(meta::type_v<NodeIndex>);
403 }
404 /// @brief Number of the edge labels.
405 constexpr size_t num_edge_labels() const noexcept
406 {
407 return self_().num_labels(meta::type_v<EdgeIndex>);
408 }
409 /// @brief Number of the face labels.
410 constexpr size_t num_face_labels() const noexcept
411 {
412 return self_().num_labels(meta::type_v<FaceIndex<Derived>>);
413 }
414 /// @brief Number of the cell labels.
415 constexpr size_t num_cell_labels() const noexcept
416 {
417 return self_().num_labels(meta::type_v<CellIndex<Derived>>);
418 }
419
420 /// @brief Number of the nodes.
421 constexpr size_t num_nodes() const noexcept
422 {
423 return self_().num_entities(meta::type_v<NodeIndex>);
424 }
425 /// @brief Number of the nodes with a @p label.
426 constexpr size_t num_nodes(Label label) const noexcept
427 {
428 return self_().num_entities(label, meta::type_v<NodeIndex>);
429 }
430
431 /// @brief Number of the edges.
432 constexpr size_t num_edges() const noexcept
433 {
434 return self_().num_entities(meta::type_v<EdgeIndex>);
435 }
436 /// @brief Number of the edges with a @p label.
437 constexpr size_t num_edges(Label label) const noexcept
438 {
439 return self_().num_entities(label, meta::type_v<EdgeIndex>);
440 }
441
442 /// @brief Number of the faces.
443 constexpr size_t num_faces() const noexcept
444 {
445 return self_().num_entities(meta::type_v<FaceIndex<Derived>>);
446 }
447 /// @brief Number of the faces with a @p label.
448 constexpr size_t num_faces(Label label) const noexcept
449 {
450 return self_().num_entities(label, meta::type_v<FaceIndex<Derived>>);
451 }
452
453 /// @brief Number of the cells.
454 constexpr size_t num_cells() const noexcept
455 {
456 return self_().num_entities(meta::type_v<CellIndex<Derived>>);
457 }
458 /// @brief Number of the cells with a @p label.
459 constexpr size_t num_cells(Label label) const noexcept
460 {
461 return self_().num_entities(label, meta::type_v<CellIndex<Derived>>);
462 }
463
464 /// @brief Range of the nodes.
465 constexpr auto nodes() const noexcept
466 {
467 return self_().entities(meta::type_v<NodeIndex>) |
468 detail_::to_node_view_(self_());
469 }
470 /// @brief Range of the nodes with a @p label.
471 constexpr auto nodes(Label label) const noexcept
472 {
473 return self_().entities(label, meta::type_v<NodeIndex>) |
474 detail_::to_node_view_(self_());
475 }
476
477 /// @brief Range of the edges.
478 constexpr auto edges() const noexcept
479 {
480 return self_().entities(meta::type_v<EdgeIndex>) |
481 detail_::to_edge_view_(self_());
482 }
483 /// @brief Range of the edges with @p label.
484 constexpr auto edges(Label label) const noexcept
485 {
486 return self_().entities(label, meta::type_v<EdgeIndex>) |
487 detail_::to_edge_view_(self_());
488 }
489
490 /// @brief Range of the faces.
491 constexpr auto faces() const noexcept
492 {
493 return self_().entities(meta::type_v<FaceIndex<Derived>>) |
494 detail_::to_face_view_(self_());
495 }
496 /// @brief Range of the faces with @p label.
497 constexpr auto faces(Label label) const noexcept
498 {
499 return self_().entities(label, meta::type_v<FaceIndex<Derived>>) |
500 detail_::to_face_view_(self_());
501 }
502
503 /// @brief Range of the cells.
504 constexpr auto cells() const noexcept
505 {
506 return self_().entities(meta::type_v<CellIndex<Derived>>) |
507 detail_::to_cell_view_(self_());
508 }
509 /// @brief Range of the cells with @p label.
510 constexpr auto cells(Label label) const noexcept
511 {
512 return self_().entities(label, meta::type_v<CellIndex<Derived>>) |
513 detail_::to_cell_view_(self_());
514 }
515
516 /// @brief Range of the interior nodes.
517 constexpr auto interior_nodes() const noexcept
518 {
519 return nodes(Label{0});
520 }
521
522 /// @brief Range of the interior edges.
523 constexpr auto interior_edges() const noexcept
524 {
525 return edges(Label{0});
526 }
527
528 /// @brief Range of the interior faces.
529 constexpr auto interior_faces() const noexcept
530 {
531 return faces(Label{0});
532 }
533
534 /// @brief Range of the interior cells.
535 constexpr auto interior_cells() const noexcept
536 {
537 return cells(Label{0});
538 }
539
540 /// @brief Range of the boundary nodes.
541 constexpr auto boundary_nodes() const noexcept
542 {
543 return nodes() | std::views::drop(num_nodes(Label{0}));
544 }
545
546 /// @brief Range of the boundary edges.
547 constexpr auto boundary_edges() const noexcept
548 {
549 return edges() | std::views::drop(num_edges(Label{0}));
550 }
551
552 /// @brief Range of the boundary faces.
553 constexpr auto boundary_faces() const noexcept
554 {
555 return faces() | std::views::drop(num_faces(Label{0}));
556 }
557
558 /// @brief Range of the boundary cells.
559 constexpr auto boundary_cells() const noexcept
560 {
561 return cells() | std::views::drop(num_cells(Label{0}));
562 }
563
564}; // class MeshInterface
