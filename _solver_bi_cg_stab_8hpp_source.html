38/// @brief The BiCGStab (Biconjugate Gradients Stabilized) linear operator
39/// equation solver.
40///
41/// BiCGStab, like the other BiCG type solvers, requires two operator
42/// multiplications per iteration. BiCGStab typically converges much smoother,
43/// than CGS.
44///
45/// References:
46/// @verbatim
47/// [1] Henk A. van der Vorst.
48/// “Bi-CGSTAB: A Fast and Smoothly Converging Variant of Bi-CG for the
49/// Solution of Nonsymmetric Linear Systems.”
50/// SIAM J. Sci. Comput. 13 (1992): 631-644.
51/// @endverbatim
52template<legacy_vector_like Vector>
53class BiCgStabSolver final : public IterativeSolver<Vector> {
54private:
55
56 real_t _alpha, _rho, _omega;
57 Vector _p_vec, _r_vec, _r_tilde_vec, _t_vec, _v_vec, _z_vec;
58
59 real_t init(const Vector& x_vec, const Vector& b_vec,
60 const Operator<Vector>& lin_op,
61 const Preconditioner<Vector>* pre_op) override {
62 const bool left_pre =
63 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Left);
64
65 _p_vec.assign(x_vec, false);
66 _r_vec.assign(x_vec, false);
67 _r_tilde_vec.assign(x_vec, false);
68 _t_vec.assign(x_vec, false);
69 _v_vec.assign(x_vec, false);
70 if (pre_op != nullptr) _z_vec.assign(x_vec, false);
71
72 // Initialize:
73 // ----------------------
74 // 𝒓 ← 𝒃 - 𝓐𝒙,
75 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
76 // 𝒛 ← 𝒓,
77 // 𝒓 ← 𝓟𝒛,
78 // 𝗲𝗻𝗱 𝗶𝗳
79 // 𝒓̃ ← 𝒓,
80 // 𝜌 ← <𝒓̃⋅𝒓>.
81 // ----------------------
82 lin_op.Residual(_r_vec, b_vec, x_vec);
83 if (left_pre) {
84 std::swap(_z_vec, _r_vec);
85 pre_op->mul(_r_vec, _z_vec);
86 }
87 _r_tilde_vec <<= _r_vec;
88 _rho = dot_product(_r_tilde_vec, _r_vec);
89
90 return sqrt(_rho);
91 }
92
93 real_t iterate(Vector& x_vec, const Vector& b_vec,
94 const Operator<Vector>& lin_op,
95 const Preconditioner<Vector>* pre_op) override {
96 const bool left_pre =
97 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Left);
98 const bool right_pre =
99 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Right);
100
101 // Continue the iterations:
102 // ----------------------
103 // 𝗶𝗳 𝘍𝘪𝘳𝘴𝘵𝘐𝘵𝘦𝘳𝘢𝘵𝘪𝘰𝘯:
104 // 𝒑 ← 𝒓.
105 // 𝗲𝗹𝘀𝗲:
106 // 𝜌̅ ← 𝜌,
107 // 𝜌 ← <𝒓̃⋅𝒓>,
108 // 𝛽 ← (𝜌/𝜌̅)⋅(𝛼/𝜔),
109 // 𝒑 ← 𝒓 + 𝛽⋅(𝒑 - 𝜔⋅𝒗).
110 // 𝗲𝗻𝗱 𝗶𝗳
111 // ----------------------
112 const bool first_iteration = this->iteration == 0;
113 if (first_iteration) {
114 _p_vec <<= _r_vec;
115 } else {
116 const real_t rho_bar =
117 std::exchange(_rho, dot_product(_r_tilde_vec, _r_vec));
118 const real_t beta{safe_divide(_alpha * _rho, _omega * rho_bar)};
119 _p_vec <<= _r_vec + beta * (_p_vec - _omega * _v_vec);
120 }
121
122 // Update the solution and the residual:
123 // ----------------------
124 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
125 // 𝒗 ← 𝓟(𝒛 ← 𝓐𝒑),
126 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
127 // 𝒗 ← 𝓐(𝒛 ← 𝓟𝒑),
128 // 𝗲𝗹𝘀𝗲:
129 // 𝒗 ← 𝓐𝒑,
130 // 𝗲𝗻𝗱 𝗶𝗳
131 // 𝛼 ← 𝜌/<𝒓̃⋅𝒗>,
132 // 𝒙 ← 𝒙 + 𝛼⋅(𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦 ? 𝒛 : 𝒑),
133 // 𝒓 ← 𝒓 - 𝛼⋅𝒗.
134 // ----------------------
135 if (left_pre) pre_op->mul(_v_vec, _z_vec, lin_op, _p_vec);
136 else if (right_pre) lin_op.mul(_v_vec, _z_vec, *pre_op, _p_vec);
137 else lin_op.mul(_v_vec, _p_vec);
138
139 _alpha = safe_divide(_rho, dot_product(_r_tilde_vec, _v_vec));
140 x_vec += _alpha * (right_pre ? _z_vec : _p_vec);
141 _r_vec -= _alpha * _v_vec;
142
143 // Update the solution and the residual again:
144 // ----------------------
145 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
146 // 𝒕 ← 𝓟(𝒛 ← 𝓐𝒓),
147 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
148 // 𝒕 ← 𝓐(𝒛 ← 𝓟𝒓),
149 // 𝗲𝗹𝘀𝗲:
150 // 𝒕 ← 𝓐𝒓,
151 // 𝗲𝗻𝗱 𝗶𝗳
152 // 𝜔 ← <𝒕⋅𝒓>/<𝒕⋅𝒕>,
153 // 𝒙 ← 𝒙 + 𝜔⋅(𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦 ? 𝒛 : 𝒓),
154 // 𝒓 ← 𝒓 - 𝜔⋅𝒕.
155 // ----------------------
156 if (left_pre) pre_op->mul(_t_vec, _z_vec, lin_op, _r_vec);
157 else if (right_pre) lin_op.mul(_t_vec, _z_vec, *pre_op, _r_vec);
158 else lin_op.mul(_t_vec, _r_vec);
159 _omega =
160 safe_divide(dot_product(_t_vec, _r_vec), dot_product(_t_vec, _t_vec));
161 x_vec += _omega * (right_pre ? _z_vec : _r_vec);
162 _r_vec -= _omega * _t_vec;
163
164 return norm_2(_r_vec);
165 }
166
167}; // class BiCgStabSolver

171/// @brief The BiCGStab(l) (Biconjugate Gradients Stabilized)
172/// linear operator equation solver.
173///
174/// BiCGStab(l), like the other BiCG type solvers, requires two operator
175/// multiplications per iteration.
176///
177/// References:
178/// @verbatim
179/// [1] Gerard L. G. Sleijpen and Diederik R. Fokkema.
180/// “BiCGStab(l) for Linear Equations involving Unsymmetric Matrices with
181/// Complex Spectrum.”
182/// Electronic Transactions on Numerical Analysis 1 (1993): 11-32.
183/// @endverbatim
184template<legacy_vector_like Vector>
185class BiCgStabLSolver final : public InnerOuterIterativeSolver<Vector> {
186private:
187
188 real_t _alpha, _rho, _omega;
189 DenseVector<real_t> _gamma, _gamma_bar, _gamma_bbar, _sigma;
190 DenseMatrix<real_t> _tau;
191 Vector _r_tilde_vec, _z_vec;
192 std::vector<Vector> _r_vecs, _u_vecs;
193
194 real_t outer_init(const Vector& x_vec, const Vector& b_vec,
195 const Operator<Vector>& lin_op,
196 const Preconditioner<Vector>* pre_op) override {
197 const size_t l = this->num_inner_iterations;
198
199 _gamma.assign(l + 1);
200 _gamma_bar.assign(l + 1);
201 _gamma_bbar.assign(l + 1);
202 _sigma.assign(l + 1);
203 _tau.assign(l + 1, l + 1);
204
205 _r_tilde_vec.assign(x_vec, false);
206 if (pre_op != nullptr) _z_vec.assign(x_vec, false);
207
208 _r_vecs.resize(l + 1);
209 _u_vecs.resize(l + 1);
210 for (Vector& r_vec : _r_vecs) r_vec.assign(x_vec, false);
211 for (Vector& u_vec : _u_vecs) u_vec.assign(x_vec, false);
212
213 // Initialize:
214 // ----------------------
215 // 𝒖₀ ← {𝟢}ᵀ,
216 // 𝒓₀ ← 𝒃 - 𝓐𝒙,
217 // 𝗶𝗳 𝓟 ≠ 𝗻𝗼𝗻𝗲:
218 // 𝒛 ← 𝒓₀,
219 // 𝒓₀ ← 𝓟𝒛,
220 // 𝗲𝗻𝗱 𝗶𝗳
221 // 𝒓̃ ← 𝒓₀,
222 // 𝜌 ← <𝒓̃⋅𝒓₀>.
223 // ----------------------
224 fill_with(_u_vecs[0], 0.0);
225 lin_op.Residual(_r_vecs[0], b_vec, x_vec);
226 if (pre_op != nullptr) {
227 std::swap(_z_vec, _r_vecs[0]);
228 pre_op->mul(_r_vecs[0], _z_vec);
229 }
230 _r_tilde_vec <<= _r_vecs[0];
231 _rho = dot_product(_r_tilde_vec, _r_vecs[0]);
232
233 return sqrt(_rho);
234 }
235
236 real_t inner_iterate(Vector& x_vec, const Vector& b_vec,
237 const Operator<Vector>& lin_op,
238 const Preconditioner<Vector>* pre_op) override {
239 const size_t l = this->num_inner_iterations;
240 const size_t j = this->inner_iteration;
241
242 // BiCG part:
243 // ----------------------
244 // 𝗶𝗳 𝘍𝘪𝘳𝘴𝘵𝘐𝘵𝘦𝘳𝘢𝘵𝘪𝘰𝘯:
245 // 𝒖₀ ← 𝒓₀,
246 // 𝗲𝗹𝘀𝗲:
247 // 𝜌̅ ← 𝜌,
248 // 𝜌 ← <𝒓̃⋅𝒓ⱼ>,
249 // 𝛽 ← 𝛼⋅𝜌/𝜌̅,
250 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑗 𝗱𝗼:
251 // 𝒖ᵢ ← 𝒓ᵢ - 𝛽⋅𝒖ᵢ,
252 // 𝗲𝗻𝗱 𝗳𝗼𝗿
253 // 𝗲𝗻𝗱 𝗶𝗳
254 // 𝗶𝗳 𝓟 ≠ 𝗻𝗼𝗻𝗲:
255 // 𝒖ⱼ₊₁ ← 𝓟(𝒛 ← 𝓐𝒖ⱼ),
256 // 𝗲𝗹𝘀𝗲:
257 // 𝒖ⱼ₊₁ ← 𝓐𝒖ⱼ,
258 // 𝗲𝗻𝗱 𝗶𝗳
259 // 𝛼 ← 𝜌/<𝒓̃⋅𝒖ⱼ₊₁>,
260 // 𝗳𝗼𝗿 𝑖 = 𝟢, 𝑗 𝗱𝗼:
261 // 𝒓ᵢ ← 𝒓ᵢ - 𝛼⋅𝒖ᵢ₊₁.
262 // 𝗲𝗻𝗱 𝗳𝗼𝗿
263 // ----------------------
264 const bool first_iteration = this->iteration == 0;
265 if (first_iteration) {
266 _u_vecs[0] <<= _r_vecs[0];
267 } else {
268 const real_t rho_bar =
269 std::exchange(_rho, dot_product(_r_tilde_vec, _r_vecs[j]));
270 const real_t beta = safe_divide(_alpha * _rho, rho_bar);
271 for (size_t i = 0; i <= j; ++i) {
272 _u_vecs[i] <<= _r_vecs[i] - beta * _u_vecs[i];
273 }
274 }
275 if (pre_op != nullptr) {
276 pre_op->mul(_u_vecs[j + 1], _z_vec, lin_op, _u_vecs[j]);
277 } else {
278 lin_op.mul(_u_vecs[j + 1], _u_vecs[j]);
279 }
280 _alpha = safe_divide(_rho, dot_product(_r_tilde_vec, _u_vecs[j + 1]));
281 for (size_t i = 0; i <= j; ++i) {
282 _r_vecs[i] -= _alpha * _u_vecs[i + 1];
283 }
284
285 // Update the solution and the residual:
286 // ----------------------
287 // 𝒙 ← 𝒙 + 𝛼⋅𝒖₀,
288 // 𝗶𝗳 𝓟 ≠ 𝗻𝗼𝗻𝗲:
289 // 𝒓ⱼ₊₁ ← 𝓟(𝒛 ← 𝓐𝒓ⱼ).
290 // 𝗲𝗹𝘀𝗲:
291 // 𝒓ⱼ₊₁ ← 𝓐𝒓ⱼ.
292 // 𝗲𝗻𝗱 𝗶𝗳
293 // ----------------------
294 x_vec += _alpha * _u_vecs[0];
295 if (pre_op != nullptr) {
296 pre_op->mul(_r_vecs[j + 1], _z_vec, lin_op, _r_vecs[j]);
297 } else {
298 lin_op.mul(_r_vecs[j + 1], _r_vecs[j]);
299 }
300
301 if (j == l - 1) {
302 // Minimal residual part:
303 // ----------------------
304 // 𝗳𝗼𝗿 𝑗 = 𝟣, 𝑙 𝗱𝗼:
305 // 𝗳𝗼𝗿 𝑖 = 𝟣, 𝑗 - 𝟣 𝗱𝗼:
306 // 𝜏ᵢⱼ ← <𝒓ᵢ⋅𝒓ⱼ>/𝜎ᵢ,
307 // 𝒓ⱼ ← 𝒓ⱼ - 𝜏ᵢⱼ⋅𝒓ᵢ,
308 // 𝗲𝗻𝗱 𝗳𝗼𝗿
309 // 𝜎ⱼ ← <𝒓ⱼ⋅𝒓ⱼ>,
310 // 𝛾̅ⱼ ← <𝒓₀⋅𝒓ⱼ>/𝜎ⱼ,
311 // 𝗲𝗻𝗱 𝗳𝗼𝗿
312 // ----------------------
313 for (size_t j = 1; j <= l; ++j) {
314 for (size_t i = 1; i < j; ++i) {
315 _tau(i, j) =
316 safe_divide(dot_product(_r_vecs[i], _r_vecs[j]), _sigma(i));
317 _r_vecs[j] -= _tau(i, j) * _r_vecs[i];
318 }
319 _sigma(j) = dot_product(_r_vecs[j], _r_vecs[j]);
320 _gamma_bar(j) =
321 safe_divide(dot_product(_r_vecs[0], _r_vecs[j]), _sigma(j));
322 }
323
324 // ----------------------
325 // 𝜔 ← 𝛾ₗ ← 𝛾̅ₗ, 𝜌 ← -𝜔⋅𝜌,
326 // 𝗳𝗼𝗿 𝑗 = 𝑙 - 𝟣, 𝟣, -𝟣 𝗱𝗼:
327 // 𝛾ⱼ ← 𝛾̅ⱼ,
328 // 𝗳𝗼𝗿 𝑖 = 𝑗 + 𝟣, 𝑙 𝗱𝗼:
329 // 𝛾ⱼ ← 𝛾ⱼ - 𝜏ⱼᵢ⋅𝛾ᵢ,
330 // 𝗲𝗻𝗱 𝗳𝗼𝗿
331 // 𝗲𝗻𝗱 𝗳𝗼𝗿
332 // 𝗳𝗼𝗿 𝑗 = 𝟣, 𝑙 - 𝟣 𝗱𝗼:
333 // 𝛾̿ⱼ ← 𝛾ⱼ₊₁,
334 // 𝗳𝗼𝗿 𝑖 = 𝑗 + 𝟣, 𝑙 - 𝟣 𝗱𝗼:
335 // 𝛾̿ⱼ ← 𝛾̿ⱼ + 𝜏ⱼᵢ⋅𝛾ᵢ₊₁.
336 // 𝗲𝗻𝗱 𝗳𝗼𝗿
337 // 𝗲𝗻𝗱 𝗳𝗼𝗿
338 // ----------------------
339 _omega = _gamma(l) = _gamma_bar(l), _rho *= -_omega;
340 for (size_t j = l - 1; j != 0; --j) {
341 _gamma(j) = _gamma_bar(j);
342 for (size_t i = j + 1; i <= l; ++i) {
343 _gamma(j) -= _tau(j, i) * _gamma(i);
344 }
345 }
346 for (size_t j = 1; j < l; ++j) {
347 _gamma_bbar(j) = _gamma(j + 1);
348 for (size_t i = j + 1; i < l; ++i) {
349 _gamma_bbar(j) += _tau(j, i) * _gamma(i + 1);
350 }
351 }
352
353 // Update the solution and the residual again:
354 // ----------------------
355 // 𝒙 ← 𝒙 + 𝛾₁⋅𝒓₀,
356 // 𝒓₀ ← 𝒓₀ - 𝛾̅ₗ⋅𝒓ₗ,
357 // 𝒖₀ ← 𝒖₀ - 𝛾ₗ⋅𝒖ₗ,
358 // 𝗳𝗼𝗿 𝑗 = 𝟣, 𝑙 - 𝟣 𝗱𝗼:
359 // 𝒙 ← 𝒙 + 𝛾̿ⱼ⋅𝒓ⱼ,
360 // 𝒓₀ ← 𝒓₀ - 𝛾̅ⱼ⋅𝒓ⱼ,
361 // 𝒖₀ ← 𝒖₀ - 𝛾ⱼ⋅𝒖ⱼ.
362 // 𝗲𝗻𝗱 𝗳𝗼𝗿
363 // ----------------------
364 x_vec += _gamma(1) * _r_vecs[0];
365 _r_vecs[0] -= _gamma_bar(l) * _r_vecs[l];
366 _u_vecs[0] -= _gamma(l) * _u_vecs[l];
367 for (size_t j = 1; j < l; ++j) {
368 x_vec += _gamma_bbar(j) * _r_vecs[j];
369 _r_vecs[0] -= _gamma_bar(j) * _r_vecs[j];
370 _u_vecs[0] -= _gamma(j) * _u_vecs[j];
371 }
372 }
373
374 return norm_2(_r_vecs[0]);
375 }
376
377public:
378
379 BiCgStabLSolver() {
380 this->num_inner_iterations = 2;
381 }
382
383}; // class BiCgStabLSolver
