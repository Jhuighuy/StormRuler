59/// @brief Generic mesh field.
60template<mesh Mesh, index Index, class Value = real_t, size_t NumVars = 1>
61class Field final :
62 public TargetMatrixInterface<Field<Mesh, Index, Value, NumVars>> {
63private:
64
65 std::string _name;
66 const Mesh* _p_mesh = nullptr;
67 IndexedVector<Index, Subfield<Value, NumVars>> _data;
68
69public:
70
71 constexpr Field() = default;
72
73 constexpr explicit Field(const Mesh& mesh)
74 : _p_mesh{&mesh}, _data(_p_mesh->num_entities(meta::type_v<Index>)) {}
75
76 /// @brief Field shape.
77 [[nodiscard]] constexpr auto shape() const noexcept {
78 return std::array{_p_mesh->num_entities(meta::type_v<Index>), NumVars};
79 }
80
81 /// @todo Document me!
82 constexpr void assign(const Field& other, bool copy = true) {
83 *this = Field{*other._p_mesh};
84 }
85
86 /// @brief Element at @p index.
87 /// @{
88 [[nodiscard]] constexpr auto& operator[](Index index) noexcept {
89 STORM_ASSERT(index < _p_mesh->num_entities(meta::type_v<Index>),
90 "Index is out of range!");
91 return _data[index];
92 }
93 [[nodiscard]] constexpr const auto& operator[](Index index) const noexcept {
94 STORM_ASSERT(index < _p_mesh->num_entities(meta::type_v<Index>),
95 "Index is out of range!");
96 return _data[index];
97 }
98 /// @}
99
100 using TargetMatrixInterface<Field<Mesh, Index, Value, NumVars>>::operator=;
101
102 /// @brief Element at @p row_index and @p col_index.
103 /// @{
104 [[nodiscard]] constexpr Value& //
105 operator()(size_t row_index, size_t col_index = 0) noexcept {
106 return _data[Index{row_index}]; //[col_index];
107 }
108 [[nodiscard]] constexpr const Value&
109 operator()(size_t row_index, size_t col_index = 0) const noexcept {
110 return _data[Index{row_index}]; //[col_index];
111 }
112 /// @}
113
114}; // class Field
