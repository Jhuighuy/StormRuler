27/// @brief Permutation utilities.
28///
29/// Permutation can be used as a mapping from the permuted to the unpermuted
30/// indices of a collection. Common usage pattern:
31/// @code
32/// std::vector<T> values = {...};
33/// std::vector<T> permuted_values(values.size());
34/// for (size_t i = 0; i < permuted_values.size(); ++i) {
35/// permuted_values[i] = values[permutation[i]];
36/// }
37/// @endcode
38///
39/// While the common usage pattern for the inverse permutations:
40/// @code
41/// std::vector<T> values = {...};
42/// std::vector<T> permuted_values(values.size());
43/// for (size_t j = 0; j < values.size(); ++j) {
44/// permuted_values[inverse_permutation[i]] = values[i];
45/// }
46/// @endcode

52/// @brief Invert the permutation. Complexity is linear.
53/// @todo Add contrains!
54/// @{
55template<std::input_iterator PermutationIterator,
56 std::sentinel_for<PermutationIterator> PermutationSentinel,
57 std::random_access_iterator InversePermutationIterator>
58 requires std::indirectly_copyable<PermutationIterator,
59 InversePermutationIterator>
60constexpr void invert_permutation(PermutationIterator perm_iterator,
61 PermutationSentinel perm_sentinel,
62 InversePermutationIterator iperm_iterator)
63{
64 using Index = std::iter_value_t<PermutationIterator>;
65 Index index{};
66 for (; perm_iterator != perm_sentinel; ++perm_iterator, ++index) {
67 STORM_ASSERT_(*perm_iterator != Index{SIZE_MAX},
68 "Invalid permutation iterator value!");
69 iperm_iterator[static_cast<size_t>(*perm_iterator)] = index;
70 }
71}
72template<std::ranges::input_range PermutationRange,
73 std::random_access_iterator InversePermutationIterator>
74 requires std::indirectly_copyable<std::ranges::iterator_t<PermutationRange>,
75 InversePermutationIterator>
76constexpr void invert_permutation(PermutationRange&& perm_range,
77 InversePermutationIterator iperm_iterator)
78{
79 invert_permutation(std::ranges::begin(perm_range),
80 std::ranges::end(perm_range), std::move(iperm_iterator));
81}

84/// @brief Apply the permutation inplace. Complexity is linear.
85/// @{
86template<std::random_access_iterator PermutationIterator,
87 std::sentinel_for<PermutationIterator> PermutationSentinel,
88 std::invocable<std::iter_value_t<PermutationIterator>,
89 std::iter_value_t<PermutationIterator>>
90 SwapFunc>
91 requires std::permutable<PermutationIterator>
92constexpr void permute_inplace(PermutationIterator perm_iterator,
93 PermutationSentinel perm_sentinel,
94 SwapFunc swap_func)
95{
96 // For implementation details see:
97 // https://devblogs.microsoft.com/oldnewthing/20170102-00/?p=95095
98 using Index = std::iter_value_t<PermutationIterator>;
99 Index index{};
100 for (; perm_iterator != perm_sentinel; ++perm_iterator, ++index) {
101 Index current_index = index;
102 PermutationIterator current_perm_iterator = perm_iterator;
103 while (*current_perm_iterator != index) {
104 STORM_ASSERT_(*current_perm_iterator != Index{SIZE_MAX},
105 "Invalid permutation iterator value!");
106 const Index new_index = *current_perm_iterator;
107 swap_func(current_index, new_index);
108 *current_perm_iterator = current_index, current_index = new_index;
109 current_perm_iterator = perm_iterator + (current_index - index);
110 }
111 *current_perm_iterator = current_index;
112 }
113#ifndef NDEBUG
114 std::ranges::fill(perm_iterator, perm_sentinel, Index{SIZE_MAX});
115#endif
116}
117template<std::ranges::random_access_range PermutationRange,
118 std::invocable<std::ranges::range_value_t<PermutationRange>,
119 std::ranges::range_value_t<PermutationRange>>
120 SwapFunc>
121 requires std::permutable<std::ranges::iterator_t<PermutationRange>>
122constexpr void permute_inplace(PermutationRange&& perm_range,
123 SwapFunc swap_func)
124{
125 permute_inplace(std::ranges::begin(perm_range), //
126 std::ranges::end(perm_range), std::move(swap_func));
127}
