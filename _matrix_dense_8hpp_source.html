98 }

102 STORM_ASSERT_(num_rows_ == coeffs.begin()->size(),

105 if constexpr (NumRows == std::dynamic_extent ||
106 NumCols == std::dynamic_extent) {
107 coeffs_ = std::make_unique<Value[]>(num_rows_ * num_cols_);
108 }
109 for (size_t i{0}; i < num_rows_; ++i) {
110 for (size_t j{0}; j < num_cols_; ++j) {
111 (*this)(i, j) = (coeffs.begin()[i]).begin()[j];
112 }
113 }
114 }

172/// @brief Perform a LU decomposition of a square matrix @p mat.
173constexpr void decompose_lu(matrix auto&& mat, //
174 matrix auto&& l_mat, matrix auto&& u_mat) noexcept {
175 const auto size{num_rows(mat)};
176 fill_diag_with(l_mat, 1.0);
177 fill_with(u_mat, 0.0);
178 for (size_t ix{0}; ix < size; ++ix) {
179 for (size_t iy{0}; iy < ix; ++iy) {
180 l_mat(ix, iy) = mat(ix, iy);
181 for (size_t iz{0}; iz < iy; ++iz) {
182 l_mat(ix, iy) -= l_mat(ix, iz) * u_mat(iz, iy);
183 }
184 l_mat(ix, iy) /= u_mat(iy, iy);
185 }
186 for (size_t iy{ix}; iy < size; ++iy) {
187 u_mat(ix, iy) = mat(ix, iy);
188 for (size_t iz{0}; iz < ix; ++iz) {
189 u_mat(ix, iy) -= l_mat(ix, iz) * u_mat(iz, iy);
190 }
191 }
192 }
193}

195constexpr void inplace_solve_lu(matrix auto&& l_mat, matrix auto&& u_mat,
196 auto& vec) {
197 const auto size{num_rows(l_mat)};
198 for (size_t ix{0}; ix < size; ++ix) {
199 for (size_t iy{0}; iy < ix; ++iy) {
200 vec(ix) -= l_mat(ix, iy) * vec(iy);
201 }
202 vec(ix) /= l_mat(ix, ix);
203 }
204 for (size_t rix{0}; rix < size; ++rix) {
205 size_t ix{size - 1 - rix};
206 for (size_t iy{ix + 1}; iy < size; ++iy) {
207 vec(ix) -= u_mat(ix, iy) * vec(iy);
208 }
209 vec(ix) /= u_mat(ix, ix);
210 }
211}

213/// @brief Inverse a square matrix @p mat using the LU decomposition.
214constexpr void inplace_inverse_lu(matrix auto&& mat,
215 matrix auto&& inv_mat) noexcept {
216 using Value = std::decay_t<decltype(mat(0, 0))>;
217 DenseMatrix<Value> L(num_rows(mat), num_cols(mat));
218 DenseMatrix<Value> U(num_rows(mat), num_cols(mat));
219 decompose_lu(mat, L, U);
220 fill_diag_with(inv_mat, 1.0);
221 for (size_t iy{0}; iy < num_rows(mat); ++iy) {
222 auto inv_mat_col = [&](size_t ix) -> Value& { return inv_mat(ix, iy); };
223 inplace_solve_lu(L, U, inv_mat_col);
224 }
225}
