41/// @brief Matrix builder view.
42template<shape Shape, boxable Func>
43 requires regular_applicable<Func, Shape> &&
44 can_reference<apply_result_t<Func, Shape>>
45class MatrixBuilderView final :
46 public MatrixViewInterface<MatrixBuilderView<Shape, Func>> {
47private:
48
49 // static_assert(std::copyable<Func>, "Boxing is not implemented yet!");
50 static_assert(!matrix<apply_result_t<Func, Shape>>,
51 "Block matrix builders are not supported yet!");
52
53 STORM_NO_UNIQUE_ADDRESS Shape _shape;
54 STORM_NO_UNIQUE_ADDRESS Func _func;
55
56public:
57
58 /// @brief Construct a builder view.
59 constexpr MatrixBuilderView(Shape shape, Func func)
60 : _shape{shape}, _func{std::move(func)} {}
61
62 /// @brief Get the matrix shape.
63 constexpr auto shape() const {
64 return _shape;
65 }
66
67 /// @brief Get the matrix element at @p indices.
68 template<class... Indices>
69 requires compatible_matrix_indices_v<MatrixBuilderView, Indices...>
70 constexpr auto operator()(Indices... indices) const {
71 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
72 return _func(indices...);
73 }
74
75}; // class MatrixBuilderView

77namespace detail {
78 template<class Shape, class Func>
79 concept can_matrix_builder_view = requires {
80 MatrixBuilderView{std::declval<Shape>(), std::declval<Func>()};
81 };
82} // namespace detail

84/// @brief Build a matrix of shape @p shape with function @p func.
85template<shape Shape, class Func>
86 requires detail::can_matrix_builder_view<Shape, Func>
87constexpr auto build(Shape shape, Func func) {
88 return MatrixBuilderView{shape, std::move(func)};
89}

93/// @brief Build a constant matrix with shape @p shape
94/// filled with elements equal to @p elem.
95template<scalar Elem, shape Shape>
96constexpr auto values(Shape shape, Elem elem) noexcept {
97 return MatrixBuilderView{shape, Constant{std::move(elem)}};

100/// @brief Build a matrix with shape @p shape filled with zeroes.
101template<scalar Elem = real_t, shape Shape>

103constexpr auto zeroes(Shape shape) noexcept {

106/// @brief Build a matrix with extents @p extents filled with zeroes.

128/// @brief Build a diagonal matrix with @p shape with a diagonal element
129/// value @p diagonal and off-diagonal element value @p off_diagonal.
130template<scalar Elem = real_t, shape Shape>
131constexpr auto eye(Shape shape, //
132 Elem diagonal = Elem{1},

145/// @brief Build a matrix with @p extent evenly spaced elements
146/// over the specified interval @p start, @p stop.
147template<scalar Elem = real_t, extent Extent>
148constexpr auto linspace(Elem start, Elem stop, Extent extent) noexcept {
149 const auto step = (stop - start) / (extent + 1);
150 const auto func = [start, step](size_t i) { return start + i * step; };
151 return MatrixBuilderView{std::tuple{extent}, std::move(func)};
152}
153
154/// @brief Build a matrix with @p extent evenly spaced elements

156template<scalar Elem = real_t, extent Extent>
157constexpr auto logspace(Elem start, Elem stop, Extent extent) noexcept {
158 static_assert(detail::always_false<Elem>,
159 "`linspace` is not implemented yet!");
160}
