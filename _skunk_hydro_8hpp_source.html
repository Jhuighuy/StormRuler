69 /** make primitive variables, 𝑸 = (𝜌,𝑝,𝒗,𝑞ᵢ,…)ᵀ. */
70 real_t* make_prim(size_t num_vars, real_t* prim) const
71 {
72 *reinterpret_cast<std::array<real_t, 5>*>(prim) = {rho, p, vel.x, vel.y,
73 vel.z};
74 return prim;
75 }
76
77 /** make conserved variables, 𝑼 = (𝜌,𝜌𝐸,𝜌𝒗,𝜌𝑞ᵢ,…)ᵀ. */
78 real_t* make_cons(size_t num_vars, real_t* cons) const
79 {
80 *reinterpret_cast<std::array<real_t, 5>*>(cons) = {
81 rho, rho * nrg, rho * vel.x, rho * vel.y, rho * vel.z};
82 for (size_t i = 5; i < num_vars; ++i) {
83 if (rest_cons != nullptr) {
84 cons[i] = rest_cons[i - 5];
85 } else if (rest_prim != nullptr) {
86 cons[i] = rho * rest_prim[i - 5];
87 }
88 }
89 return cons;
90 }
91
92 /** make flux variables, 𝑭ₙ = (𝜌𝒗ₙ,𝜌𝐻𝒗ₙ,𝜌𝒗𝒗ₙ,𝜌𝑞ᵢ𝒗ₙ,…)ᵀ. */
93 real_t* make_flux(size_t num_vars, const vec3_t& n, real_t* flux) const
94 {
95 *reinterpret_cast<std::array<real_t, 5>*>(flux) = {
96 rho * vel_n, rho * vel_n * ent, rho * vel_n * vel.x + p * n.x,
97 rho * vel_n * vel.y + p * n.y, rho * vel_n * vel.z + p * n.z};
98 for (size_t i = 5; i < num_vars; ++i) {
99 if (rest_cons != nullptr) {
100 flux[i] = vel_n * rest_cons[i - 5];
101 } else if (rest_prim != nullptr) {
102 flux[i] = rho * vel_n * rest_prim[i - 5];
103 }
104 }
105 return flux;
106 }
107};

109inline MhdHydroVars::MhdHydroVars(const vec3_t& n, const real_t* q_cons,
110 const real_t* q_prim)
111 : MhdHydroVars()
112{
113 if (q_cons != nullptr) {
114 rho = q_cons[0];
115 nrg = q_cons[1] / rho;
116 vel.x = q_cons[2] / rho;
117 vel.y = q_cons[3] / rho;
118 vel.z = q_cons[4] / rho;
119 vel_n = glm::dot(vel, n);
120 eps = nrg - 0.5 * glm::dot(vel, vel);
121 p = Gamma1 * rho * eps;
122 ent = nrg + p / rho;
123 } else if (q_prim != nullptr) {
124 rho = q_prim[0];
125 p = q_prim[1];
126 vel.x = q_prim[2];
127 vel.y = q_prim[3];
128 vel.z = q_prim[4];
129 vel_n = glm::dot(vel, n);
130 eps = p / rho / Gamma1;
131 nrg = eps + 0.5 * glm::dot(vel, vel);
132 ent = nrg + p / rho;
133 }
134 c_snd = std::sqrt(Gamma * p / rho);
135}
