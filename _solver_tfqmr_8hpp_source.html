36/// @brief Parametrized TFQMR solver.
37template<legacy_vector_like Vector, bool L1>
38class BaseTfqmrSolver : public IterativeSolver<Vector> {
39private:
40
41 real_t rho_, tau_;
42 Vector d_vec_, r_tilde_vec_, u_vec_, v_vec_, y_vec_, s_vec_, z_vec_;
43
44 real_t init(const Vector& x_vec, const Vector& b_vec,
45 const Operator<Vector>& lin_op,
46 const Preconditioner<Vector>* pre_op) override {
47 const bool left_pre =
48 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Left);
49
50 d_vec_.assign(x_vec, false);
51 r_tilde_vec_.assign(x_vec, false);
52 u_vec_.assign(x_vec, false);
53 v_vec_.assign(x_vec, false);
54 y_vec_.assign(x_vec, false);
55 s_vec_.assign(x_vec, false);
56 if (pre_op != nullptr) z_vec_.assign(x_vec, false);
57
58 // Initialize:
59 // ----------------------
60 // 𝗶𝗳 𝘓₁:
61 // 𝒅 ← 𝒙,
62 // 𝗲𝗹𝘀𝗲:
63 // 𝒅 ← {𝟢}ᵀ,
64 // 𝗲𝗻𝗱 𝗶𝗳
65 // 𝒚 ← 𝒃 - 𝓐𝒙,
66 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
67 // 𝒛 ← 𝒚,
68 // 𝒚 ← 𝓟𝒛,
69 // 𝗲𝗻𝗱 𝗶𝗳
70 // 𝒖 ← 𝒚,
71 // 𝒓̃ ← 𝒖,
72 // 𝜌 ← <𝒓̃⋅𝒓>, 𝜏 ← 𝜌¹ᐟ².
73 // ----------------------
74 if constexpr (L1) {
75 d_vec_ <<= x_vec;
76 } else {
77 fill_with(d_vec_, 0.0);
78 }
79 lin_op.Residual(y_vec_, b_vec, x_vec);
80 if (left_pre) {
81 std::swap(z_vec_, y_vec_);
82 pre_op->mul(y_vec_, z_vec_);
83 }
84 u_vec_ <<= y_vec_;
85 r_tilde_vec_ <<= u_vec_;
86 rho_ = dot_product(r_tilde_vec_, u_vec_), tau_ = sqrt(rho_);
87
88 return tau_;
89 }
90
91 real_t iterate(Vector& x_vec, const Vector& b_vec,
92 const Operator<Vector>& lin_op,
93 const Preconditioner<Vector>* pre_op) override {
94 const bool left_pre =
95 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Left);
96 const bool right_pre =
97 (pre_op != nullptr) && (this->pre_side == PreconditionerSide::Right);
98
99 // Continue the iterations:
100 // ----------------------
101 // 𝗶𝗳 𝘍𝘪𝘳𝘴𝘵𝘐𝘵𝘦𝘳𝘢𝘵𝘪𝘰𝘯:
102 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
103 // 𝒔 ← 𝓟(𝒛 ← 𝓐𝒚),
104 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
105 // 𝒔 ← 𝓐(𝒛 ← 𝓟𝒚),
106 // 𝗲𝗹𝘀𝗲:
107 // 𝒔 ← 𝓐𝒚.
108 // 𝗲𝗻𝗱 𝗶𝗳
109 // 𝒗 ← 𝒔,
110 // 𝗲𝗹𝘀𝗲:
111 // 𝜌̅ ← 𝜌,
112 // 𝜌 ← <𝒓̃⋅𝒖>,
113 // 𝛽 ← 𝜌/𝜌̅,
114 // 𝒗 ← 𝒔 + 𝛽⋅𝒗,
115 // 𝒚 ← 𝒖 + 𝛽⋅𝒚,
116 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
117 // 𝒔 ← 𝓟(𝒛 ← 𝓐𝒚),
118 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
119 // 𝒔 ← 𝓐(𝒛 ← 𝓟𝒚),
120 // 𝗲𝗹𝘀𝗲:
121 // 𝒔 ← 𝓐𝒚,
122 // 𝗲𝗻𝗱 𝗶𝗳
123 // 𝒗 ← 𝒔 + 𝛽⋅𝒗.
124 // 𝗲𝗻𝗱 𝗶𝗳
125 // ----------------------
126 const bool first_iteration = this->iteration == 0;
127 if (first_iteration) {
128 if (left_pre) pre_op->mul(s_vec_, z_vec_, lin_op, y_vec_);
129 else if (right_pre) lin_op.mul(s_vec_, z_vec_, *pre_op, y_vec_);
130 else lin_op.mul(s_vec_, y_vec_);
131 v_vec_ <<= s_vec_;
132 } else {
133 const real_t rho_bar =
134 std::exchange(rho_, dot_product(r_tilde_vec_, u_vec_));
135 const real_t beta = safe_divide(rho_, rho_bar);
136 v_vec_ <<= s_vec_ + beta * v_vec_;
137 y_vec_ <<= u_vec_ + beta * y_vec_;
138 if (left_pre) pre_op->mul(s_vec_, z_vec_, lin_op, y_vec_);
139 else if (right_pre) lin_op.mul(s_vec_, z_vec_, *pre_op, y_vec_);
140 else lin_op.mul(s_vec_, y_vec_);
141 v_vec_ <<= s_vec_ + beta * v_vec_;
142 }
143
144 // Update the solution:
145 // ----------------------
146 // 𝛼 ← 𝜌/<𝒓̃⋅𝒗>,
147 // 𝗳𝗼𝗿 𝑚 = 𝟢, 𝟣 𝗱𝗼:
148 // 𝒖 ← 𝒖 - 𝛼⋅𝒔,
149 // 𝒅 ← 𝒅 + 𝛼⋅(𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦 ? 𝒛 : 𝒚),
150 // 𝜔 ← ‖𝒖‖,
151 // 𝗶𝗳 𝘓₁:
152 // 𝗶𝗳 𝜔 < 𝜏:
153 // 𝜏 ← 𝜔, 𝒙 ← 𝒅,
154 // 𝗲𝗻𝗱 𝗶𝗳
155 // 𝗲𝗹𝘀𝗲:
156 // 𝑐𝑠, 𝑠𝑛 ← 𝘚𝘺𝘮𝘖𝘳𝘵𝘩𝘰(𝜏, 𝜔),
157 // 𝜏 ← 𝑐𝑠⋅𝜔,
158 // 𝒙 ← 𝒙 + 𝑐𝑠²⋅𝒅,
159 // 𝒅 ← 𝑠𝑛²⋅𝒅,
160 // 𝗲𝗻𝗱 𝗶𝗳
161 // 𝗶𝗳 𝑚 = 𝟢:
162 // 𝒚 ← 𝒚 - 𝛼⋅𝒗,
163 // 𝗶𝗳 𝘓𝘦𝘧𝘵𝘗𝘳𝘦:
164 // 𝒔 ← 𝓟(𝒛 ← 𝓐𝒚).
165 // 𝗲𝗹𝘀𝗲 𝗶𝗳 𝘙𝘪𝘨𝘩𝘵𝘗𝘳𝘦:
166 // 𝒔 ← 𝓐(𝒛 ← 𝓟𝒚).
167 // 𝗲𝗹𝘀𝗲:
168 // 𝒔 ← 𝓐𝒚.
169 // 𝗲𝗻𝗱 𝗶𝗳
170 // 𝗲𝗻𝗱 𝗶𝗳
171 // 𝗲𝗻𝗱 𝗳𝗼𝗿
172 // ----------------------
173 const real_t alpha = safe_divide(rho_, dot_product(r_tilde_vec_, v_vec_));
174 for (size_t m = 0; m <= 1; ++m) {
175 u_vec_ -= alpha * s_vec_;
176 d_vec_ += alpha * (right_pre ? z_vec_ : y_vec_);
177 const real_t omega = norm_2(u_vec_);
178 if constexpr (L1) {
179 if (omega < tau_) tau_ = omega, x_vec <<= d_vec_;
180 } else {
181 const auto [cs, sn, rr] = sym_ortho(tau_, omega);
182 tau_ = omega * cs;
183 x_vec += std::pow(cs, 2) * d_vec_;
184 d_vec_ *= std::pow(sn, 2);
185 }
186 if (m == 0) {
187 y_vec_ -= alpha * v_vec_;
188 if (left_pre) pre_op->mul(s_vec_, z_vec_, lin_op, y_vec_);
189 else if (right_pre) lin_op.mul(s_vec_, z_vec_, *pre_op, y_vec_);
190 else lin_op.mul(s_vec_, y_vec_);
191 }
192 }
193
194 // Compute the residual norm
195 // (or it's upper bound estimate in the ℒ₂ case):
196 // ----------------------
197 // 𝜏̃ ← 𝜏,
198 // 𝗶𝗳 𝗻𝗼𝘁 𝘓₁:
199 // 𝜏̃ ← 𝜏⋅(𝟤𝑘 + 𝟥)¹ᐟ².
200 // 𝗲𝗻𝗱 𝗶𝗳
201 // ----------------------
202 real_t tau_tilde = tau_;
203 if constexpr (!L1) {
204 const size_t k{this->iteration};
205 tau_tilde *= std::sqrt(2.0 * k + 3.0);
206 }
207
208 return tau_tilde;
209 }
210
211protected:
212
213 BaseTfqmrSolver() = default;
214
215}; // class BaseTfqmrSolver

219/// @brief The TFQMR (Transpose-Free Quasi-Minimal Residual) linear operator
220/// equation solver.
221///
222/// TFQMR, like the other BiCG type methods, normally requires two
223/// operator-vector products per iteration. But, unlike the other BiCG type
224/// methods, TFQMR does not implicitly contain the residual norm estimate,
225/// only the rough upper bound is avariable, so at the latter iterations an
226/// extra operator-vector product per iteration may be required for the explicit
227/// residual estimation.
228///
229/// TFQMR typically converges much smoother, than CGS and BiCGStab.
230/// @todo Breakdowns?
231///
232/// References:
233/// @verbatim
234/// [1] Freund, Roland W.
235/// “A Transpose-Free Quasi-Minimal Residual Algorithm for Non-Hermitian
236/// Linear Systems.”
237/// SIAM J. Sci. Comput. 14 (1993): 470-482.
238/// [2] Freund, Roland W.
239/// “Transpose-Free Quasi-Minimal Residual Methods for Non-Hermitian Linear
240/// Systems.” (1994).
241/// @endverbatim
242template<legacy_vector_like Vector>
243class TfqmrSolver final :
244 public BaseTfqmrSolver<Vector, false> {}; // class TfqmrSolver
245
246/// @brief The TFQMR1 (Transpose-Free 1-norm Quasi-Minimal Residual) linear
247/// operator equation solver.
248///
249/// TFQMR1, like the other BiCG type solvers, requires two operator-vector
250/// products per iteration. Unlike TFQMR, TFQMR1 implicitly contains the
251/// residual norm estimate, so no extra operator-vector products are required.
252///
253/// TFQMR1 typically converges much smoother, than CGS and BiCGStab and is
254/// slightly faster than TFQMR.
255/// @todo Breakdowns?
256///
257/// References:
258/// @verbatim
259/// [1] H.M Bücker,
260/// “A Transpose-Free 1-norm Quasi-Minimal Residual Algorithm
261/// for Non-Hermitian Linear Systems.“, FZJ-ZAM-IB-9706.
262/// @endverbatim
263template<legacy_vector_like Vector>
264class Tfqmr1Solver final :
265 public BaseTfqmrSolver<Vector, true> {}; // class Tfqmr1Solver
