43/// @brief Matrix roll view.
44/// @todo This should be re-thinked.
45template<array_like auto AxesArray, //
46 matrix_view Matrix, tuple_like Shifts>
47 requires boxable<Shifts>
48class MatrixRollView final :
49 public MatrixViewInterface<MatrixRollView<AxesArray, Matrix, Shifts>> {
50private:
51
52 static_assert(std::copyable<Shifts>, "Boxing is not implemented yet!");
53
54 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
55 STORM_NO_UNIQUE_ADDRESS Shifts _shifts;
56
57public:
58
59 /// @brief Construct a matrix roll view.
60 constexpr explicit MatrixRollView(Matrix mat, Shifts shifts)
61 : _mat{std::move(mat)}, _shifts{std::move(shifts)} {}
62
63 /// @brief Get the matrix shape.
64 constexpr auto shape() const {
65 return _mat.shape();
66 }
67
68 /// @brief Get the matrix element at @p indices.
69 template<class... Indices>
70 requires compatible_matrix_indices_v<MatrixRollView, Indices...>
71 constexpr auto operator()(Indices... indices) const {
72 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
73 const auto extents_tuple = _mat.shape();
74 std::array indices_array{static_cast<size_t>(indices)...};
75 ([&]<size_t... Axes>(std::index_sequence<Axes...>) {
76 const auto roll_axes = [&](auto... axis_shift_pairs) {
77 const auto roll_axis = [&](auto axis, ptrdiff_t shift) {
78 auto& index = indices_array[axis];
79 const auto extent = std::get<axis>(extents_tuple);
80 index = (index + extent - shift) % extent;
81 };
82 (std::apply(roll_axis, std::move(axis_shift_pairs)), ...);
83 };
84 const std::tuple axes{fixed_size_t<Axes>{}...};
85 zip_apply(roll_axes, {}, axes, _shifts);
86 }(detail::_to_index_sequence<AxesArray>()));
87 return std::apply(_mat, indices_array);
88 }
89
90}; // class MatrixRollView

92/// @brief Create a matrix roll view.
93template<array_like auto AxesArray, //
94 viewable_matrix Matrix, tuple_like Shifts>
95constexpr auto make_roll_view(Matrix&& mat, Shifts shifts) {
96 using View = MatrixRollView<AxesArray, matrix_view_t<Matrix>, Shifts>;
97 return View{std::forward<Matrix>(mat), std::move(shifts)};
98}

100/// @brief Roll the matrix @p mat elements along given axes.
101template<size_t... Axes, viewable_matrix Matrix,
102 std::convertible_to<ptrdiff_t>... Shifts>
103 requires compatible_matrix_axes_v<Matrix, Axes...>
104constexpr auto roll(Matrix&& mat, Shifts... shifts) {
105 constexpr auto AxesArray = [] {
106 constexpr size_t NumAxes = sizeof...(Axes);
107 constexpr size_t NumShifts = sizeof...(Shifts);
108#if STORM_COMPILER_MSVC
109 // Bug in MSVC.
110 static_cast<void>(NumAxes);
111 static_cast<void>(NumShifts);
112#endif
113 if constexpr (NumAxes == 0) {
114 return detail::_to_array(std::make_index_sequence<NumShifts>{});
115 } else if constexpr (NumAxes == NumShifts) {
116 // We have to use the `static_cast` due to the bug in MSVC.
117 return std::array{static_cast<size_t>(Axes)...};
118 } else {
119 static_assert(detail::always_false<Matrix>,
120 "Incompatible amount of the specified axes and shifts!");
121 }
122 }();
123 return make_roll_view<AxesArray>( //
124 std::forward<Matrix>(mat), std::tuple{shifts...});
125}

170/// @brief Matrix transpose view.
171/// @tparam AxesArray Array-like permutation of the extents.
172/// @todo This should be re-thinked.
173template<array_like auto AxesArray, matrix_view Matrix>
174class MatrixTransposeView final :
175 public MatrixViewInterface<MatrixTransposeView<AxesArray, Matrix>> {
176private:
177
178 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
179
180 static constexpr auto _AxesInverseArray = [] {
181 static_assert(permutations::is_permutation(AxesArray),
182 "Axes form an invalid permutation!");
183 return permutations::invert_permutation(AxesArray);
184 }();
185
186public:
187
188 /// @brief Construct a matrix transpose view.
189 constexpr explicit MatrixTransposeView(Matrix mat) : _mat{std::move(mat)} {}
190
191 /// @brief Get the matrix shape.
192 constexpr auto shape() const {
193 return ([&]<size_t... Axes>(std::index_sequence<Axes...>) {
194 const auto extents = _mat.shape();
195 return std::tuple{std::get<Axes>(extents)...};
196 }(detail::_to_index_sequence<AxesArray>()));
197 }
198
199 /// @brief Get the matrix element at @p indices.
200 template<class... Indices>
201 requires compatible_matrix_indices_v<MatrixTransposeView, Indices...>
202 constexpr auto operator()(Indices... indices) const {
203 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
204 return ([&]<size_t... Axes>(std::index_sequence<Axes...>) {
205 const std::tuple indices_tuple{std::move(indices)...};
206 return _mat(std::get<Axes>(indices_tuple)...);
207 }(detail::_to_index_sequence<_AxesInverseArray>()));
208 }
209
210}; // MatrixTransposeView

212/// @brief Create a matrix transpose view.
213template<array_like auto AxesArray, viewable_matrix Matrix>

215 using View = MatrixTransposeView<AxesArray, matrix_view_t<Matrix>>;

219/// @brief Permute the matrix's @p mat extents.
220/// @tparam Axes Permutation of the extents, if present.
221template<size_t... Axes, viewable_matrix Matrix>
222 requires compatible_matrix_axes_v<Matrix, Axes...> &&
223 (sizeof...(Axes) != 0 || matrix_rank_v<Matrix> != 1)
224constexpr auto transpose(Matrix&& mat) {
225 constexpr auto AxesArray = ([] {
226 constexpr size_t NumAxes = sizeof...(Axes);
227 constexpr size_t Rank = matrix_rank_v<Matrix>;
228#if STORM_COMPILER_MSVC
229 // Bug in MSVC.
230 static_cast<void>(NumAxes);
231 static_cast<void>(Rank);
232#endif
233 if constexpr (NumAxes == 0) {
234 return ([]<size_t... Extents>(std::index_sequence<Extents...>) {
235 return std::array{(Rank - Extents - 1_sz)...};
236 }(std::make_index_sequence<Rank>{}));
237 } else if constexpr (NumAxes == Rank) {
238 // We have to use the `static_cast` due to the bug in MSVC.
239 return std::array{static_cast<size_t>(Axes)...}; // NOSONAR
240 } else {
241 static_assert(detail::always_false<Matrix>,
242 "Invalid numer of axes specified!");
243 }
244 }());
245 return make_matrix_transpose<AxesArray>(std::forward<Matrix>(mat));
246}

250/// @brief Conjugate-transpose the matrix @p mat.
251/// @tparam Axes Permutation of the extents.
252template<size_t... Axes, viewable_matrix Matrix>
253 requires compatible_matrix_axes_v<Matrix, Axes...>
254constexpr auto conj_transpose(Matrix&& mat) {
255 return conj(transpose<Axes...>(std::forward<Matrix>(mat)));
