32/// @brief A semi-valid argument for the CRTP classes.
33template<class Derived>
34concept crtp_derived = std::is_class_v<Derived> &&
35 std::same_as<Derived, std::remove_cv_t<Derived>>;
36
37// Based on /*is-derived-from-view-interface*/<T> implementation.
38namespace detail_ {
39 template<template<crtp_derived Derived> class CrtpInterface>
40 struct DerivedFromCrtpInterfaceImpl_ {
41 template<class T, class U>
42 DerivedFromCrtpInterfaceImpl_(const T&,
43 const CrtpInterface<U>&); // not defined
44 };
45} // namespace detail_

47/// @brief Checks if and only if @c T has exactly one public base class
48/// @c CrtpInterface&lt;U&gt; for some type @c U, and @c T has no base classes
49/// of type @c CrtpInterface&lt;V&gt; for any other type @c V.
50template<class T, template<class Derived> class CrtpInterface>
51concept derived_from_crtp_interface = requires(T x) { //
52 detail_::DerivedFromCrtpInterfaceImpl_<CrtpInterface>{x, x};
53};
