90/// @brief 3D Vector cross product view.
91template<matrix_view_r<1> Vector1, matrix_view_r<1> Vector2>
92class CrossProductView final :
93 public MatrixViewInterface<CrossProductView<Vector1, Vector2>> {
94private:
95
96 STORM_NO_UNIQUE_ADDRESS Vector1 vec1_;
97 STORM_NO_UNIQUE_ADDRESS Vector2 vec2_;
98
99public:
100
101 /// @brief Construct a cross product view.
102 constexpr explicit CrossProductView(Vector1 vec1, Vector2 vec2)
103 : vec1_{std::move(vec1)}, vec2_{std::move(vec2)} {
104 STORM_ASSERT(vec1_.shape() == shape() && vec2_.shape() == shape(),
105 "3D vectors are expected!");
106 }
107
108 /// @brief Get the vector shape.
109 constexpr static auto shape() {
110 return fixed_shape_t<3>{};
111 }
112
113 /// @brief Get the vector element at @p index.
114 template<class Index>
115 requires compatible_matrix_indices_v<CrossProductView, Index> &&
116 std::convertible_to<Index, size_t>
117 constexpr auto operator()(Index index) const {
118 STORM_ASSERT(in_range(shape(), index), "Index is out of range!");
119 switch (static_cast<size_t>(index)) {
120 case 0: return vec1_(1_sz) * vec2_(2_sz) - vec1_(2_sz) * vec2_(1_sz);
121 case 1: return vec1_(2_sz) * vec2_(0_sz) - vec1_(0_sz) * vec2_(2_sz);
122 case 2: return vec1_(0_sz) * vec2_(1_sz) - vec1_(1_sz) * vec2_(0_sz);
123 default: STORM_UNREACHABLE();
124 }
125 }
126
127}; // class CrossProductView
