41/// @brief Target matrix: an output matrix that is an assignment target.
42/// Additional requirement: target matrix overrides (or uses) the operator=
43/// from the TargetMatrixInterface.
44template<class TargetMatrix>
45concept target_matrix =
46 output_matrix<TargetMatrix> &&
47 derived_from_crtp_interface<TargetMatrix, TargetMatrixInterface>;
48
49/// @brief CRTP interface to a target matrix.
50template<crtp_derived TargetMatrix>
51class TargetMatrixInterface : public NonAssignable
52{
53private:
54
55 constexpr TargetMatrix& self_() noexcept
56 {
57 static_assert(std::derived_from<TargetMatrix, TargetMatrixInterface>);
58 static_assert(target_matrix<TargetMatrix>);
59 return static_cast<TargetMatrix&>(*this);
60 }
61 constexpr const TargetMatrix& self_() const noexcept
62 {
63 return const_cast<TargetMatrixInterface&>(*this).self_();
64 }
65
66public:
67
68 /// @brief Assign the current matrix coefficients from matrix @p mat.
69 template<matrix Matrix>
70 requires std::assignable_from< //
71 matrix_element_ref_t<TargetMatrix>, matrix_element_t<Matrix>>
72 constexpr decltype(auto) operator=(Matrix&& mat) noexcept
73 {
74 return assign(self_(), std::forward<Matrix>(mat));
75 }
76
77 /// @brief Multiply-assign the current matrix by a scalar @p scal.
78 template<std::copyable Scalar>
79 requires numeric_matrix<TargetMatrix> && numeric_type<Scalar>
80 constexpr decltype(auto) operator*=(Scalar scal)
81 {
82 return assign(BindLast{MultiplyAssign{}, std::move(scal)}, *this);
83 }
84
85 /// @brief Divide-assign the current matrix by a scalar @p scal.
86 template<std::copyable Scalar>
87 requires numeric_matrix<TargetMatrix> && numeric_type<Scalar>
88 constexpr decltype(auto) operator/=(Scalar scal)
89 {
90 return assign(BindLast{DivideAssign{}, std::move(scal)}, *this);
91 }
92
93 /// @brief Add-assign the matrix @p mat to the current matrix.
94 template<matrix Matrix>
95 requires compatible_matrices_v<TargetMatrix, Matrix> && //
96 numeric_matrix<TargetMatrix> && numeric_matrix<Matrix>
97 constexpr decltype(auto) operator+=(Matrix&& mat)
98 {
99 return assign(AddAssign{}, self_(), std::forward<Matrix>(mat));
100 }
101
102 /// @brief Subtract-assign the matrix @p mat from the current matrix.
103 template<matrix Matrix>
104 requires compatible_matrices_v<TargetMatrix, Matrix> && //
105 numeric_matrix<TargetMatrix> && numeric_matrix<Matrix>
106 constexpr decltype(auto) operator-=(Matrix&& mat)
107 {
108 return assign(SubtractAssign{}, self_(), std::forward<Matrix>(mat));
109 }
110
111 /// @brief Element-wise multiply-assign the current matrix by
112 /// the matrix @p mat.
113 template<matrix Matrix>
114 requires compatible_matrices_v<TargetMatrix, Matrix> && //
115 numeric_matrix<TargetMatrix> && numeric_matrix<Matrix>
116 constexpr decltype(auto) operator*=(Matrix&& mat)
117 {
118 return assign(MultiplyAssign{}, self_(), std::forward<Matrix>(mat));
119 }
120
121 /// @brief Element-wise divide-assign the current matrix by the matrix @p mat.
122 template<matrix Matrix>
123 requires compatible_matrices_v<TargetMatrix, Matrix> && //
124 numeric_matrix<TargetMatrix> && numeric_matrix<Matrix>
125 constexpr decltype(auto) operator/=(Matrix&& mat)
126 {
127 return assign(DivideAssign{}, self_(), std::forward<Matrix>(mat));
128 }
129
130}; // class TargetMatrixInterface

134/// @brief Output matrix as target wrapper.
135template<matrix_view Matrix>
136 requires output_matrix<Matrix>
137class TargetMatrixView final :
138 public MatrixViewInterface<TargetMatrixView<Matrix>>,
139 public TargetMatrixInterface<TargetMatrixView<Matrix>>,
140 public NonMovable
141{
142private:
143
144 STORM_NO_UNIQUE_ADDRESS_ Matrix mat_;
145
146public:
147
148 /// @brief Construct a matrix target view.
149 constexpr explicit TargetMatrixView(Matrix mat) : mat_{std::move(mat)} {}
150
151 /// @brief Get the matrix shape.
152 constexpr auto shape() const noexcept
153 {
154 return mat_.shape();
155 }
156
157 /// @brief Get the matrix element at @p indices.
158 template<class... Indices>
159 requires compatible_matrix_indices_v<TargetMatrixView, Indices...>
160 constexpr auto operator()(Indices... indices) const noexcept
161 {
162 STORM_ASSERT_(in_range(shape(), indices...), "Indices are out of range!");
163 return mat_(indices...);
164 }
165
166 using TargetMatrixInterface<TargetMatrixView<Matrix>>::operator=;
167
168}; // class TargetMatrixView

171TargetMatrixView(Matrix) -> TargetMatrixView<matrix_view_t<Matrix>>;

173/// @brief Wrap the output matrix @p mat as a target.
174template<viewable_matrix Matrix>
175 requires output_matrix<Matrix>
176constexpr auto as_target(Matrix&& mat)
177{
178 return TargetMatrixView{std::forward<Matrix>(mat)};
179}
