35/// @brief The Richardson iteration linear operator equation solver.
36///
37/// References:
38/// @verbatim
39/// [1] ???
40/// @endverbatim
41template<legacy_vector_like Vector>
42class RichardsonSolver final : public IterativeSolver<Vector> {
43public:
44
45 real_t relaxation_factor = 1.0e-4;
46
47private:
48
49 Vector _r_vec, _z_vec;
50
51 real_t init(const Vector& x_vec, const Vector& b_vec,
52 const Operator<Vector>& lin_op,
53 const Preconditioner<Vector>* pre_op) override {
54 _r_vec.assign(x_vec, false);
55 if (pre_op != nullptr) _z_vec.assign(x_vec, false);
56
57 // Initialize:
58 // ----------------------
59 // 𝒓 ← 𝒃 - 𝓐𝒙,
60 // 𝗶𝗳 𝓟 ≠ 𝗻𝗼𝗻𝗲:
61 // 𝒛 ← 𝒓,
62 // 𝒓 ← 𝓟𝒛.
63 // 𝗲𝗻𝗱 𝗶𝗳
64 // ----------------------
65 lin_op.Residual(_r_vec, b_vec, x_vec);
66 if (pre_op != nullptr) {
67 std::swap(_z_vec, _r_vec);
68 pre_op->mul(_r_vec, _z_vec);
69 }
70
71 return norm_2(_r_vec);
72 }
73
74 real_t iterate(Vector& x_vec, const Vector& b_vec,
75 const Operator<Vector>& lin_op,
76 const Preconditioner<Vector>* pre_op) override {
77 const real_t& omega = relaxation_factor;
78
79 // Update the solution and the residual:
80 // ----------------------
81 // 𝒙 ← 𝒙 + 𝜔⋅𝒓,
82 // 𝒓 ← 𝒃 - 𝓐𝒙,
83 // 𝗶𝗳 𝓟 ≠ 𝗻𝗼𝗻𝗲:
84 // 𝒛 ← 𝒓,
85 // 𝒓 ← 𝓟𝒛.
86 // 𝗲𝗻𝗱 𝗶𝗳
87 // ----------------------
88 x_vec += omega * _r_vec;
89 lin_op.Residual(_r_vec, b_vec, x_vec);
90 if (pre_op != nullptr) {
91 std::swap(_z_vec, _r_vec);
92 pre_op->mul(_r_vec, _z_vec);
93 }
94
95 return norm_2(_r_vec);
96 }
97
98}; // class RichardsonSolver
