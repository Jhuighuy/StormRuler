58/// @brief OpenGL buffer.
59template<class Type>
60class Buffer : detail_::noncopyable_ {
61private:
62
63 GLuint buffer_id_;
64 GLsizei buffer_size_ = 0;
65
66public:
67
68 /// @brief Construct a buffer.
69 Buffer() {
70 glGenBuffers(1, &buffer_id_);
71 }
72
73 /// @brief Construct a buffer with a size.
74 /// @param usage Intended buffer usage.
75 explicit Buffer(GLsizei size, BufferUsage usage = BufferUsage::static_draw)
76 : Buffer{} {
77 assign(size, usage);
78 }
79
80 /// @brief Construct a buffer with a @p size copies of @p value.
81 /// @param usage Intended buffer usage.
82 explicit Buffer(GLsizei size, const Type& value,
83 BufferUsage usage = BufferUsage::static_draw)
84 : Buffer{} {
85 assign(size, value, usage);
86 }
87
88 /// @brief Construct a buffer with a range.
89 /// @param usage Intended buffer usage.
90 template<std::ranges::input_range Range>
91 requires std::same_as<std::ranges::range_value_t<Range>, Type>
92 explicit Buffer(Range&& values, BufferUsage usage = BufferUsage::static_draw)
93 : Buffer{} {
94 assign(std::forward<Range>(values), usage);
95 }
96
97 /// @brief Move-construct a buffer.
98 Buffer(Buffer&&) = default;
99 /// @brief Move-assign the buffer.
100 Buffer& operator=(Buffer&&) = default;
101
102 /// @brief Destruct the buffer.
103 ~Buffer() {
104 glDeleteBuffers(1, &buffer_id_);
105 }
106
107 /// @brief Cast to buffer ID.
108 constexpr operator GLuint() const noexcept {
109 return buffer_id_;
110 }
111
112 /// @brief Buffer size.
113 constexpr GLsizei size() const noexcept {
114 return buffer_size_;
115 }
116
117 /// @brief Bind the buffer to @p target.
118 void bind(BufferTarget target) const {
119 glBindBuffer(static_cast<GLenum>(target), buffer_id_);
120 }
121
122 /// @brief Assign the buffer @p size.
123 /// @param usage Intended buffer usage.
124 void assign(GLsizei size, BufferUsage usage = BufferUsage::static_draw) {
125 glBindBuffer(GL_ARRAY_BUFFER, buffer_id_);
126 buffer_size_ = size;
127 const auto num_bytes = static_cast<GLsizeiptr>(size * sizeof(Type));
128 glBufferData(GL_ARRAY_BUFFER, num_bytes, /*data*/ nullptr,
129 static_cast<GLenum>(usage));
130 }
131
132 /// @brief Assign the buffer @p size copies of @p value.
133 void assign(GLsizei size, const Type& value,
134 BufferUsage usage = BufferUsage::static_draw) {
135 /// @todo C++23:
136 // assign(std::views::repeat(value, size));
137 assign(std::views::iota(size) |
138 std::views::transform([&](auto) { return value; }));
139 }
140
141 /// @brief Assign the buffer @p values.
142 /// @param usage Intended buffer usage.
143 template<std::ranges::input_range Range>
144 requires std::same_as<std::ranges::range_value_t<Range>, Type>
145 void assign(Range&& values, BufferUsage usage = BufferUsage::static_draw) {
146 glBindBuffer(GL_ARRAY_BUFFER, buffer_id_);
147 if constexpr (std::ranges::sized_range<Range>) {
148 assign(static_cast<GLsizei>(values.size()));
149 const auto pointer = glMapBuffer(GL_ARRAY_BUFFER, GL_WRITE_ONLY);
150 if (pointer == nullptr) STORM_THROW_GL_("Failed to map the buffer!");
151 std::ranges::copy(values, static_cast<Type*>(pointer));
152 if (glUnmapBuffer(GL_ARRAY_BUFFER) != GL_TRUE) {
153 STORM_THROW_GL_("Failed to unmap the buffer!");
154 }
155 } else {
156 std::vector<Type> temp{};
157 std::ranges::copy(values, std::back_inserter(temp));
158 buffer_size_ = static_cast<GLsizei>(temp.size());
159 const auto num_bytes =
160 static_cast<GLsizeiptr>(buffer_size_ * sizeof(Type));
161 glBufferData(GL_ARRAY_BUFFER, num_bytes, temp.data(),
162 static_cast<GLenum>(usage));
163 }
164 }
165
166 /// @todo Refactor as `operator[]`.
167 /// @brief Get the buffer value at @p index.
168 Type get(size_t index) const {
169 STORM_ASSERT_(index < static_cast<size_t>(buffer_size_),
170 "Index is out of range!");
171 glBindBuffer(GL_ARRAY_BUFFER, buffer_id_);
172 const auto pointer = glMapBufferRange(
173 GL_ARRAY_BUFFER, static_cast<GLintptr>(index * sizeof(Type)),
174 static_cast<GLsizeiptr>(sizeof(Type)), GL_MAP_READ_BIT);
175 if (pointer == nullptr) STORM_THROW_GL_("Failed to map the buffer!");
176 Type value = *static_cast<const Type*>(pointer);
177 if (glUnmapBuffer(GL_ARRAY_BUFFER) != GL_TRUE) {
178 STORM_THROW_GL_("Failed to unmap the buffer!");
179 }
180 return value;
181 }
182
183 /// @brief Set the buffer @p value at @p index.
184 void set(size_t index, const Type& value) {
185 STORM_ASSERT_(index < static_cast<size_t>(buffer_size_),
186 "Index is out of range!");
187 glBindBuffer(GL_ARRAY_BUFFER, buffer_id_);
188 const auto offset = static_cast<GLintptr>(index * sizeof(Type));
189 glBufferSubData(GL_ARRAY_BUFFER, offset, sizeof(Type), &value);
190 }
191
192}; // class Buffer
