40/// @brief Shape type.
41enum class Type : std::uint8_t {
42 segment, ///< Segment shape type.
43 triangle, ///< Triangle shape type.
44 triangle_strip, ///< Triangle strip shape type.
45 quadrangle, ///< Quadrangle shape type.
46 polygon, ///< Polygon shape type.
47 tetrahedron, ///< Tetrahedron shape type.
48 pyramid, ///< Pyramid shape type.
49 pentahedron, ///< Pentahedron shape type.
50 hexahedron, ///< Hexahedron shape type.
51 count_,
52}; // enum class shape_type

54/// @brief Shape concept.
55template<class Shape>
56concept shape =
57 requires(Shape shape) {
58 // clang-format off
59 { shape.type() } -> std::same_as<Type>;
60 { shape.nodes() } -> std::ranges::forward_range;
61 // clang-format off
62 } &&
63 std::same_as<
64 std::ranges::range_value_t<decltype(std::declval<Shape>().nodes())>,
65 NodeIndex>;
66
67/// @brief 1D Shape concept.
68template<class Shape>
69concept shape1D = shape<Shape> && //
70 !requires(Shape shape) { shape.edges(); } &&
71 !requires(Shape shape) { shape.faces(); };
72
73/// @brief 2D Shape concept.
74template<class Shape>
75concept shape2D = shape<Shape> && //
76 requires(Shape shape) {
77 std::apply([](shape1D auto&&...) {}, shape.edges());
78 } && !requires(Shape shape) { shape.faces(); };
79
80/// @brief 3D Shape concept.
81template<class Shape>
82concept shape3D = shape<Shape> && //
83 requires(Shape shape) {
84 std::apply([](shape1D auto&&...) {}, shape.edges());
85 } && //
86 requires(Shape shape) {
87 std::apply([](shape2D auto&&...) {}, shape.faces());
88 };
89
90/// @brief Shape topological dimensionality.
91template<shape Shape>
92 requires (shape1D<Shape> || shape2D<Shape> || shape3D<Shape>)
93inline constexpr size_t shape_dim_v = []() {
94 if constexpr (shape1D<Shape>) return 1;
95 if constexpr (shape2D<Shape>) return 2;
96 if constexpr (shape3D<Shape>) return 3;
97}();
98
99/// @brief Get the shape parts.
100template<size_t Index, shape Shape>
101 requires ((Index == 0) ||
102 (Index == 1 && (shape2D<Shape> || shape3D<Shape>) ) ||
103 (Index == 2 && shape3D<Shape>) )
104constexpr auto parts(const Shape& shape) noexcept
105{
106 if constexpr (Index == 0) {
107 return shape.nodes();
108 } else if constexpr (Index == 1) {
109 return shape.edges();
110 } else if constexpr (Index == 2) {
111 return shape.faces();
112 }
113}

117/// @brief Complex shape concept.
118template<class Shape, class Mesh>
119concept complex_shape =
120 shape<Shape> && mesh<Mesh> &&
121 requires(Shape shape, const Mesh& mesh) {
122 // clang-format off
123 { shape.pieces(mesh) } -> std::ranges::forward_range;
124 // clang-format on
125 } &&
126 shape<std::ranges::range_value_t<decltype( //
127 std::declval<Shape>().pieces(std::declval<Mesh>()))>>;
128
129/// @brief Piece type of a complex.
130template<class Shape, class Mesh>
131 requires complex_shape<Shape, Mesh>
132using piece_t = std::ranges::range_value_t<decltype( //
133 std::declval<Shape>().pieces(std::declval<Mesh>()))>;
134
135namespace detail_
136{
137 template<class Shape, class Mesh>
138 concept can_volume_ =
139 requires(Shape shape, const Mesh& mesh) { volume(shape, mesh); };
140 template<class Shape, class Mesh>
141 concept can_barycenter_ =
142 requires(Shape shape, const Mesh& mesh) { barycenter(shape, mesh); };
143 template<class Shape, class Mesh>
144 concept can_normal_ =
145 requires(Shape shape, const Mesh& mesh) { normal(shape, mesh); };
146} // namespace detail_

148/// @brief Compute the complex @p shape "volume"
149/// (length in 1D, area in 2D, volume in 3D).
150template<shape Shape, mesh Mesh>
151 requires (complex_shape<Shape, Mesh> &&
152 detail_::can_volume_<piece_t<Shape, Mesh>, Mesh>)
153constexpr real_t volume(const Shape& shape, const Mesh& mesh) noexcept
154{
155 const auto pieces = shape.pieces(mesh);
156 auto vol = volume(pieces.front(), mesh);
157 for (const auto& piece : pieces | std::views::drop(1)) {
158 vol += volume(piece, mesh);
159 }
160 return vol;
161}
162
163/// @brief Compute the simplex @p shape barycenter.
164template<shape Shape, mesh Mesh>
165 requires (!complex_shape<Shape, Mesh> &&
166 (mesh_dim_v<Mesh> >= shape_dim_v<Shape>) )
167constexpr mesh_vec_t<Mesh> barycenter(const Shape& shape,
168 const Mesh& mesh) noexcept
169{
170 const auto nodes = shape.nodes();
171 mesh_vec_t<Mesh> sum_position{mesh.position(nodes.front())};
172 for (const auto& node : nodes | std::views::drop(1)) {
173 sum_position += mesh.position(node);
174 }
175 return sum_position / nodes.size();
176}
177
178/// @brief Compute the complex @p shape barycenter.
179template<shape Shape, mesh Mesh>
180 requires (complex_shape<Shape, Mesh> &&
181 detail_::can_volume_<piece_t<Shape, Mesh>, Mesh> &&
182 detail_::can_barycenter_<piece_t<Shape, Mesh>, Mesh>)
183constexpr mesh_vec_t<Mesh> barycenter(const Shape& shape,
184 const Mesh& mesh) noexcept
185{
186 const auto pieces = shape.pieces(mesh);
187 auto vol = volume(pieces.front(), mesh);
188 auto vol_center = vol * barycenter(pieces.front(), mesh);
189 for (const auto& piece : pieces | std::views::drop(1)) {
190 const auto dv = volume(piece, mesh);
191 vol += dv, vol_center += dv * barycenter(piece, mesh);
192 }
193 return vol_center / vol;
194}
195
196/// @brief Compute the complex @p shape normal.
197template<shape Shape, mesh Mesh>
198 requires (complex_shape<Shape, Mesh> &&
199 detail_::can_volume_<piece_t<Shape, Mesh>, Mesh> &&
200 detail_::can_normal_<piece_t<Shape, Mesh>, Mesh>)
201constexpr mesh_vec_t<Mesh> normal(const Shape& shape, const Mesh& mesh) noexcept
202{
203 const auto pieces = shape.pieces(mesh);
204 auto vol_normal = volume(pieces.front(), mesh) * normal(pieces.front(), mesh);
205 for (const auto& piece : pieces | std::views::drop(1)) {
206 vol_normal += volume(piece, mesh) * normal(piece, mesh);
207 }
208 return normalize(vol_normal);
209}

213/// @brief Segment shape.
214/// @verbatim
215/// +==========================================================================+
216/// | | Edges: |
217/// | n1 @ f1 |------------------------------|
218/// | \ | e1 = (n1,n2) |
219/// | \ | |
220/// | v e1 |------------------------------|
221/// | \ | Faces: |
222/// | \ |------------------------------|
223/// | n2 @ f2 | f1 = (n1) |
224/// | | f2 = (n2) |
225/// +==========================================================================+
226/// @endverbatim
227class Seg final
228{
229public:
230
231 /// @brief Segment node.

234 /// @}

240

244 return Type::segment;
245 }
246
247 /// @brief Segment nodes.
248 constexpr auto nodes() const noexcept
249 {
250 return std::array{n1, n2};
251 }

253}; // class Seg

255/// @brief Segment @p seg length ("volume").
256template<mesh Mesh>
257constexpr real_t volume(const Seg& seg, const Mesh& mesh) noexcept
258{
259 const auto v1{mesh.position(seg.n1)}, v2{mesh.position(seg.n2)};
260 return length(v2 - v1);
261}

277/// @brief Triangular shape.
278/// @verbatim
279/// +==========================================================================+
280/// | | Edges / |
281/// | n3 | / Faces: |
282/// | @ |--------------------------------|

313 return Type::triangle;
314 }

330/// @brief Triangle @p tri "volume" (area).
331template<mesh Mesh>
332 requires (mesh_dim_v<Mesh> >= 2)
333constexpr real_t volume(const Triangle& tri, const Mesh& mesh) noexcept
334{
335 const auto v1{mesh.position(tri.n1)}, v2{mesh.position(tri.n2)};
336 const auto v3{mesh.position(tri.n3)};
337 if constexpr (mesh_dim_v<Mesh> == 2) {
338 // const glm::dmat2 d{v2 - v1, v3 - v1};
339 // return glm::abs(glm::determinant(d)) / 2.0;
340 /// @todo This looks incomplete:
341 auto d0 = v2 - v1, d1 = v3 - v1;
342 return 0.5 * abs(d0(0, 0) * d1(1, 0) - d0(1, 0) * d1(0, 0));
343 } else {
344 return length(cross_product(v2 - v1, v3 - v1)) / 2.0;
345 }
346}
347
348/// @brief Triangle @p tri normal.
349template<mesh Mesh>
350 requires (mesh_dim_v<Mesh> == 3)
351constexpr mesh_vec_t<Mesh> normal(const Triangle& tri,
352 const Mesh& mesh) noexcept
353{
354 const auto v1{mesh.position(tri.n1)}, v2{mesh.position(tri.n2)};
355 const auto v3{mesh.position(tri.n3)};
356 return normalize(cross(v2 - v1, v3 - v1));
357}

426/// @brief Triangle strip shape.
427/// @verbatim
428/// +==========================================================================+
429/// | | |
430/// | n2 n4 n6 n8 | |
431/// | @---<---@---<---@---<---@ | |
432/// | / \ / \ / \ / | |
433/// | v \ / \ / \ ^ | INCOMPLETE.. |
434/// | / \ / \ / \ / | |
435/// | @--->---@--->---@--->---@ | |
436/// | n1 n3 n5 n9 | |
437/// | | |
438/// +==========================================================================+
439/// @endverbatim
440class TriangleStrip final

446
447 /// @brief Construct a triangle strip.

453 }
454
455 /// @brief Triangle strip type.
456 /// May fallback to triangle or quadrangle types.

459 STORM_ASSERT_(n.size() >= 3, "Triangle strip must have least 3 nodes!");
460 switch (n.size()) {

483 template<mesh Mesh>

572/// @brief Tetrahedral shape.
573/// @verbatim
574/// +==========================================================================+
575/// | | Edges: |
576/// | f4 |------------------------|
577/// | n4 ^ | e1 = (n1,n2) |
578/// | @ | | e2 = (n2,n3) |
579/// | f2 /|\. | f3 | e3 = (n3,n1) |
580/// | ^ / | `\. ^ | e4 = (n1,n4) |
581/// | \ / | `\. / | e5 = (n2,n4) |
582/// | \` | | `\/ | e6 = (n3,n4) |
583/// | ,\ | o /`\ | |
584/// | e4 ^ * | * `^.e6 | |
585/// | / e5 ^ `\ |------------------------|
586/// | n1 @-------|--<----------@ n3 | Faces: |
587/// | \ | e3 ,/ |------------------------|
588/// | \ | o ,/` | f1 = (n1,n3,n2) |
589/// | \ ^ e5 | ,/` | f2 = (n1,n2,n4) |
590/// | e1 v | ,^ e2 | f3 = (n2,n3,n4) |
591/// | \ | ,/` | f4 = (n3,n1,n4) |
592/// | \ | ,/` | | |
593/// | \|/` | | |
594/// | @ n2 v | |
595/// | f1 | |
596/// | | |
597/// +==========================================================================+
598/// @endverbatim
599class Tetrahedron final
600{
601public:
602
603 /// @brief Tetrahedron node.
604 /// @{
605 NodeIndex n1{}, n2{}, n3{}, n4{};
606 /// @}
607
608 /// @brief Construct a tetrahedron.
609 constexpr Tetrahedron() = default;
610 /// @brief Construct a tetrahedron with nodes.
611 constexpr Tetrahedron(NodeIndex i1, NodeIndex i2, //
612 NodeIndex i3, NodeIndex i4) noexcept
613 : n1{i1}, n2{i2}, n3{i3}, n4{i4}
614 {
615 }
616
617 /// @brief Tetrahedron type.
618 constexpr static auto type() noexcept
619 {
620 return Type::tetrahedron;
621 }
622
623 /// @brief Tetrahedron nodes.
624 constexpr auto nodes() const noexcept
625 {
626 return std::array{n1, n2, n3, n4};
627 }
628
629 /// @brief Tetrahedron edges.
630 constexpr auto edges() const noexcept
631 {
632 return std::array{Seg{n1, n2}, Seg{n2, n3}, Seg{n3, n1},
633 Seg{n1, n4}, Seg{n2, n4}, Seg{n3, n4}};
634 }
635
636 /// @brief Tetrahedron faces.
637 constexpr auto faces() const noexcept
638 {
639 return std::array{Triangle{n1, n3, n2}, Triangle{n1, n2, n4},
640 Triangle{n2, n3, n4}, Triangle{n3, n1, n4}};
641 }
642
643}; // class Tetrahedron

645/// @brief Tetrahedron @p tet barycenter.
646template<mesh Mesh>
647 requires (mesh_dim_v<Mesh> >= 3)
648constexpr mesh_vec_t<Mesh> barycenter(const Tetrahedron& tet,
649 const Mesh& mesh) noexcept
650{
651 const auto v1{mesh.position(tet.n1)}, v2{mesh.position(tet.n2)};
652 const auto v3{mesh.position(tet.n3)}, v4{mesh.position(tet.n4)};
653 return (v1 + v2 + v3 + v4) / 4.0;
654}
