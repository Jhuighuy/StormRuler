43/// @brief Target matrix: an output matrix that is an assignment target.
44/// Additional requirement: target matrix overrides (or uses) the @c operator=
45/// from the TargetMatrixInterface.
46template<class TargetMatrix>
47concept target_matrix =
48 output_matrix<TargetMatrix> &&
49 derived_from_crtp_interface<TargetMatrix, TargetMatrixInterface>;
50
51/// @brief CRTP interface to a target matrix.
52/// @todo Should I be non-movable? non-assignable?
53template<crtp_derived TargetMatrix>
54class TargetMatrixInterface {
55private:
56
57 constexpr TargetMatrix& _self() noexcept {
58 static_assert(std::derived_from<TargetMatrix, TargetMatrixInterface>);
59 static_assert(target_matrix<TargetMatrix>);
60 return static_cast<TargetMatrix&>(*this);
61 }
62 constexpr const TargetMatrix& _self() const noexcept {
63 return const_cast<TargetMatrixInterface&>(*this)._self();
64 }
65
66public:
67
68 template<std::copyable Scalar>
69 requires std::assignable_from<matrix_element_ref_t<TargetMatrix>, Scalar>
70 constexpr TargetMatrix& fill(Scalar scalar) {
71 return fill(_self(), std::move(scalar));
72 }
73
74 /// @brief Move-assign the matrix @p mat elements from the current matrix.
75 template<matrix Matrix>
76 requires assignable_matrix<TargetMatrix, Matrix>
77 constexpr TargetMatrix& move_assign(Matrix&& mat) {
78 return move_elements(_self(), std::forward<Matrix>(mat));
79 }
80
81 /// @brief Copy-assign the matrix @p mat elements from the current matrix.
82 template<matrix Matrix>
83 requires assignable_matrix<TargetMatrix, Matrix>
84 constexpr TargetMatrix& assign(Matrix&& mat) {
85 return copy_elements(_self(), std::forward<Matrix>(mat));
86 }
87
88 /// @todo TO BE REMOVED!
89 template<matrix Matrix>
90 requires assignable_matrix<TargetMatrix, Matrix>
91 constexpr TargetMatrix& operator=(Matrix&& mat) {
92 return assign(std::forward<Matrix>(mat));
93 }
94
95 /// @brief Multiply-assign the current matrix by a scalar @p scal.
96 template<scalar Scalar>
97 constexpr TargetMatrix& operator*=(Scalar scal) {
98 return eval_elements(BindLast{MultiplyAssign{}, std::move(scal)}, *this);
99 }
100
101 /// @brief Divide-assign the current matrix by a scalar @p scal.
102 template<scalar Scalar>
103 constexpr TargetMatrix& operator/=(Scalar scal) {
104 return eval_elements(BindLast{DivideAssign{}, std::move(scal)}, *this);
105 }
106
107 /// @brief Add-assign the matrix @p mat to the current matrix.
108 template<matrix Matrix>
109 requires compatible_matrices_v<TargetMatrix, Matrix>
110 constexpr TargetMatrix& operator+=(Matrix&& mat) {
111 return eval_elements(AddAssign{}, _self(), std::forward<Matrix>(mat));
112 }
113
114 /// @brief Subtract-assign the matrix @p mat from the current matrix.
115 template<matrix Matrix>
116 requires compatible_matrices_v<TargetMatrix, Matrix>
117 constexpr TargetMatrix& operator-=(Matrix&& mat) {
118 return eval_elements(SubtractAssign{}, _self(), std::forward<Matrix>(mat));
119 }
120
121 /// @brief Element-wise multiply-assign the current matrix by
122 /// the matrix @p mat.
123 template<matrix Matrix>
124 requires compatible_matrices_v<TargetMatrix, Matrix>
125 constexpr TargetMatrix& operator*=(Matrix&& mat) {
126 return eval_elements(MultiplyAssign{}, _self(), std::forward<Matrix>(mat));
127 }
128
129 /// @brief Element-wise divide-assign the current matrix by the matrix @p mat.
130 template<matrix Matrix>
131 requires compatible_matrices_v<TargetMatrix, Matrix>
132 constexpr TargetMatrix& operator/=(Matrix&& mat) {
133 return eval_elements(DivideAssign{}, _self(), std::forward<Matrix>(mat));
134 }
135
136}; // class TargetMatrixInterface

140/// @brief Output matrix as target wrapper.
141template<matrix_view Matrix>
142 requires output_matrix<Matrix>
143class TargetMatrixView final :
144 public MatrixViewInterface<TargetMatrixView<Matrix>>,
145 public TargetMatrixInterface<TargetMatrixView<Matrix>>,
146 public NonMovableInterface {
147private:
148
149 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
150
151public:
152
153 /// @brief Construct a matrix target view.
154 constexpr explicit TargetMatrixView(Matrix mat) : _mat{std::move(mat)} {}
155
156 /// @brief Assign the current matrix elements from matrix @p mat.
157 template<matrix OtherMatrix>
158 requires assignable_matrix<TargetMatrixView, OtherMatrix> &&
159 (!std::same_as<TargetMatrixView, std::remove_cvref_t<OtherMatrix>>)
160 constexpr TargetMatrixView& operator=(OtherMatrix&& mat) noexcept {
161 return this->assign(std::forward<OtherMatrix>(mat));
162 }
163
164 /// @brief Get the matrix shape.
165 constexpr auto shape() const noexcept {
166 return _mat.shape();
167 }
168
169 /// @brief Get the matrix element at @p indices.
170 /// @{
171 template<class... Indices>
172 requires compatible_matrix_indices_v<TargetMatrixView, Indices...>
173 constexpr auto& operator()(Indices... indices) noexcept {
174 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
175 return _mat(indices...);
176 }
177 template<class... Indices>
178 requires compatible_matrix_indices_v<TargetMatrixView, Indices...>
179 constexpr const auto& operator()(Indices... indices) const noexcept {
180 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
181 return _mat(indices...);
182 }
183 /// @}
184
185}; // class TargetMatrixView
186
187template<class Matrix>
188TargetMatrixView(Matrix) -> TargetMatrixView<matrix_view_t<Matrix>>;

190/// @brief Wrap the output matrix @p mat as a target.
191template<viewable_matrix Matrix>
192 requires output_matrix<Matrix>
193constexpr auto to_target(Matrix&& mat) noexcept {
194 return TargetMatrixView{std::forward<Matrix>(mat)};
195}
