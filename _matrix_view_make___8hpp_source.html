34/// @brief Matrix making view.
35template<std::copy_constructible Func>
36 requires std::is_object_v<Func> &&
37 std::regular_invocable<Func, size_t, size_t>
38class MakeMatrixView final : public MatrixViewInterface<MakeMatrixView<Func>> {
39private:
40
41 MatrixShape shape_;
42 STORM_NO_UNIQUE_ADDRESS_ Func func_;
43
44public:
45
46 /// @brief Construct a making view.
47 constexpr MakeMatrixView(MatrixShape shape, Func func)
48 : shape_{shape}, func_{std::move(func)} {}
49
50 /// @brief Get the matrix shape.
51 constexpr auto shape() const noexcept {
52 return shape_;
53 }
54
55 /// @brief Get the matrix element at @p indices.
56 constexpr auto operator()(size_t row_index, size_t col_index) const noexcept {
57 STORM_ASSERT_(shape_.in_range(row_index, col_index),
58 "Indices are out of range.");
59 return func_(row_index, col_index);
60 }
61
62}; // class MakeMatrixView

66/// @brief Make a constant matrix of @p shape.
67/// @param value Matrix element value.
68template<std::copyable Element>
69constexpr auto make_constant_matrix(MatrixShape shape, Element value) {
70 return MakeMatrixView( //
71 shape,
72 [value = std::move(value)]( //
73 [[maybe_unused]] size_t row_index,
74 [[maybe_unused]] size_t col_index) noexcept -> const Element& {
75 return value;
76 });
77}
78
79/// @brief Make a diagonal matrix of @p shape.
80/// @param diagonal Matrix diagonal element value.
81/// @param off_diagonal Matrix off-diagonal element value.
82template<std::copyable Element>
83constexpr auto make_diagonal_matrix(MatrixShape shape, Element diagonal,
84 Element off_diagonal = Element{}) {
85 return MakeMatrixView(
86 shape,
87 [diagonal = std::move(diagonal), //
88 off_diagonal = std::move(off_diagonal)](
89 size_t row_index, size_t col_index) noexcept -> const Element& {
90 return (row_index == col_index) ? diagonal : off_diagonal;
91 });
92}
