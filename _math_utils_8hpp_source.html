36/// @brief Floored modulo function.
37template<std::integral Dividend, std::integral Divisor>
38constexpr auto floored_mod(Dividend dividend, Divisor divisor) noexcept {
39 if constexpr (std::unsigned_integral<Dividend> &&
40 std::unsigned_integral<Divisor>) {
41 return dividend % divisor;
42 } else {
43 return ((dividend % divisor) + divisor) % divisor;
44 }
45}

47/// @brief If @p y is zero, return zero,
48/// otherwise return value of @p x divided by @p y.
49template<std::floating_point Arg>
50constexpr auto safe_divide(Arg x, Arg y) noexcept {
51 return (y == Arg{0}) ? Arg{0} : (x / y);
52}

54/// @brief If @p y is zero, return zero, otherwise return onverse @p y.
55template<std::floating_point Arg>
56constexpr auto safe_inverse(Arg y) noexcept {
57 return safe_divide(Arg{1}, y);
58}

163/// @brief Generate the Givens rotation.
164template<class Value>
165constexpr auto sym_ortho(Value a, Value b) noexcept {
166 // Compute:
167 // ----------------------
168 // 𝑟𝑟 ← (𝑎² + 𝑏²)¹ᐟ²,
169 // 𝑐𝑠 ← 𝑎/𝑟𝑟, 𝑠𝑛 ← 𝑏/𝑟𝑟.
170 // ----------------------
171 Value cs, sn, rr;
172 rr = hypot(a, b);
173 if (rr > Value{0}) {
174 cs = a / rr, sn = b / rr;
175 } else {
176 cs = Value{1}, sn = Value{0};
177 }
178 return std::tuple{cs, sn, rr};
179}
