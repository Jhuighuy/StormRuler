27namespace Storm::Vulture::gl {

29/// @brief RAII OpenGL debug output.
30class DebugOutput {
31public:
32
33 /// @brief Enable OpenGL debug output.
34 /// Sticks to GL_ARB_debug_output extension since debug output
35 /// is inside OpenGL since 4.3, and we are using 3.3.
36 DebugOutput() noexcept {
37 if (GLEW_ARB_debug_output != GL_TRUE) {
38 return;
39 }
40 glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS_ARB);
41 glDebugMessageCallbackARB(&on_message_, nullptr);
42 STORM_INFO_("OpenGL debug output enabled.");
43 }
44
45 /// @brief Disable OpenGL debug output.
46 ~DebugOutput() noexcept {
47 if (GLEW_ARB_debug_output != GL_TRUE) {
48 return;
49 }
50 glDisable(GL_DEBUG_OUTPUT_SYNCHRONOUS_ARB);
51 STORM_INFO_("OpenGL debug output disabled.");
52 }
53
54private:
55
56 static void on_message_(GLenum source, GLenum type, GLuint id,
57 GLenum severity, [[maybe_unused]] GLsizei length,
58 const GLchar* message,
59 [[maybe_unused]] const void* user_param) {
60 const char* error_source;
61 switch (source) {
62 case GL_DEBUG_SOURCE_API_ARB: //
63 error_source = "API call";
64 break;
65 case GL_DEBUG_SOURCE_WINDOW_SYSTEM_ARB:
66 error_source = "window system API all";
67 break;
68 case GL_DEBUG_SOURCE_SHADER_COMPILER_ARB:
69 error_source = "shader compiler";
70 break;
71 case GL_DEBUG_SOURCE_THIRD_PARTY_ARB:
72 error_source = "third party API";
73 break;
74 case GL_DEBUG_SOURCE_APPLICATION_ARB: //
75 error_source = "application";
76 break;
77 case GL_DEBUG_SOURCE_OTHER_ARB: //
78 error_source = "other";
79 break;
80 default: //
81 error_source = "unknown source";
82 break;
83 }
84
85 const char* debug_type;
86 switch (type) {
87 default:
88 case GL_DEBUG_TYPE_OTHER_ARB: //
89 debug_type = "other issue";
90 break;
91 case GL_DEBUG_TYPE_ERROR_ARB: //
92 debug_type = "error";
93 break;
94 case GL_DEBUG_TYPE_DEPRECATED_BEHAVIOR_ARB:
95 debug_type = "deprecated behavior";
96 break;
97 case GL_DEBUG_TYPE_UNDEFINED_BEHAVIOR_ARB:
98 debug_type = "undefined behavior";
99 break;
100 case GL_DEBUG_TYPE_PORTABILITY_ARB: //
101 debug_type = "portability issue";
102 break;
103 case GL_DEBUG_TYPE_PERFORMANCE_ARB: //
104 debug_type = "performance issue";
105 break;
106 }
107
108 switch (severity) {
109 default:
110 STORM_DEBUG_("OpenGL: {} {} {:#x}: {}", //
111 error_source, debug_type, id, message);
112 break;
113 case GL_DEBUG_SEVERITY_LOW_ARB:
114 STORM_INFO_("OpenGL: {} {} {:#x}: {}", //
115 error_source, debug_type, id, message);
116 break;
117 case GL_DEBUG_SEVERITY_MEDIUM_ARB:
118 STORM_WARNING_("OpenGL: {} {} {:#x}: {}", //
119 error_source, debug_type, id, message);
120 break;
121 case GL_DEBUG_SEVERITY_HIGH_ARB:
122 STORM_ERROR_("OpenGL: {} {} {:#x}: {}", //
123 error_source, debug_type, id, message);
124 break;
125 }
126 }
127
128}; // class DebugOutput
