59/// @brief Mask matrix view.
60template<matrix_view Matrix>
61 requires boolean_testable<matrix_element_t<Matrix>>
62class MaskMatrixView final :
63 public MatrixViewInterface<MaskMatrixView<Matrix>> {
64private:
65
66 using _Indices = std::array<size_t, matrix_rank_v<Matrix>>;
67
68 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
69 std::vector<_Indices> _true_indices;
70
71public:
72
73 /// @brief Construct a mask matrix view.
74 template<matrix OtherMatrix>
75 requires different_from<MaskMatrixView, OtherMatrix> &&
76 std::constructible_from<Matrix, OtherMatrix>
77 constexpr MaskMatrixView(OtherMatrix&& mat) // NOSONAR
78 : _mat{std::forward<OtherMatrix>(mat)} {
79 const auto cache_extents = [&](auto... extents) {
80 const auto cache_indices = [&](auto... indices) {
81 if (_mat(indices...)) _true_indices.push_back(_Indices{indices...});
82 };
83 detail::mdfor(cache_indices, extents...);
84 };
85 std::apply(cache_extents, shape());
86 }
87
88 /// @brief Get the matrix shape.
89 constexpr auto shape() const {
90 return _mat.shape();
91 }
92
93 /// @brief Get the matrix element at @p indices.
94 template<class... Indices>
95 requires compatible_matrix_indices_v<MaskMatrixView, Indices...>
96 constexpr auto operator()(Indices... indices) const {
97 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
98 return _mat(indices...);
99 }
100
101 /// @brief Get the range of indices,
102 /// where the matrix contains a trueish elements.
103 constexpr const auto& true_indices() const noexcept {
104 return _true_indices;
105 }
106
107}; // class MaskMatrixView

114/// @brief Index-like matrix coordate.
115template<class Coord>
116concept index_coord = detail::index_like<std::remove_cvref_t<Coord>>;
117
118/// @brief Placeholder matrix coordinate.
119template<class Coord>
120concept placeholder_coord =
121 std::same_as<std::remove_cvref_t<Coord>, Underscore>;
122
123/// @brief New axis matrix coordinate.
124template<class Coord>
125concept new_axis_coord = std::same_as<std::remove_cvref_t<Coord>, NewAxis>;
126
127/// @brief Range matrix coordinate:
128/// sized random-access range with index-like values.
129template<class Coord>
130concept range_coord = std::ranges::sized_range<Coord> &&
131 std::ranges::random_access_range<Coord> &&
132 detail::index_like<std::ranges::range_value_t<Coord>>;
133
134/// @brief Mask matrix coordinate: matrix with boolean-testable elements.
135template<class Coord>
136concept mask_coord = matrix<Coord> && //
137 boolean_testable<matrix_element_t<Coord>>;
138
139/// @brief Simple matrix coordinate: index like or a placeholder.
140/// This is a coordinate type that would not be wrapped into a view.
141template<class Coord>
142concept simple_coord =
143 index_coord<Coord> || placeholder_coord<Coord> || new_axis_coord<Coord>;
144
145/// @brief Matrix coordinate: either is a simple coordinate,
146/// either a random-access range with index-like values,
147/// either a matrix with boolean-testable elements.
148template<class Coord>
149concept coord = simple_coord<Coord> || range_coord<Coord> || mask_coord<Coord>;
150
151/// @brief Matrix coordinate rank (how much indices it eats).
152/// @{
153template<coord Coord>
154inline constexpr size_t coord_rank_v = [] {
155 if constexpr (new_axis_coord<Coord>) return 0_sz;
156 else if constexpr (mask_coord<Coord>) return matrix_rank_v<Coord>;
157 else return 1_sz;
158}();
159template<coord... Coords>
160inline constexpr size_t coords_rank_v = (0 + ... + coord_rank_v<Coords>);
161/// @}
162
163/// @brief Matrix coordinate that can be safely casted
164/// into a matrix coordinate view.
165template<class Coord>
166concept viewable_coord =
167 simple_coord<Coord> ||
168 (range_coord<Coord> && std::ranges::viewable_range<Coord>) ||
169 (mask_coord<Coord> && viewable_matrix<Coord>);
170
171/// @brief Matrix coordinate view.
172template<class Coord>
173concept coord_view = coord<Coord> && std::movable<Coord> &&
174 ((simple_coord<Coord> && std::is_object_v<Coord>) ||
175 (range_coord<Coord> && std::ranges::view<Coord>) ||
176 (mask_coord<Coord> && matrix_view<Coord>) );
177
178/// @brief Wrap the viewable coordinate @p coord into a coordinate view.
179template<viewable_coord Coord>
180constexpr auto to_coord_view(Coord&& coord) {
181 if constexpr (coord_view<std::remove_cvref_t<Coord>>) {
182 return std::forward<Coord>(coord);
183 } else if constexpr (std::ranges::viewable_range<Coord>) {
184 return std::views::all(std::forward<Coord>(coord));
185 } else if constexpr (viewable_matrix<Coord>) {
186 return MaskMatrixView{std::forward<Coord>(coord)};
187 } else {
188 static_assert(detail::always_false<Coord>, "Unexpected coordinate type!");
189 }
190}
191
192/// @brief Suitable coordinate view type for a viewable coordinate.
193template<viewable_coord Coord>
194using coord_view_t = decltype(to_coord_view(std::declval<Coord>()));

198/// @brief Matrix part view.
199template<matrix_view Matrix, coord_view... Coords>
200 requires (matrix_rank_v<Matrix> == (coord_rank_v<Coords> + ...))
201class MatrixPartView final :
202 public MatrixViewInterface<MatrixPartView<Matrix, Coords...>> {
203private:
204
205 STORM_NO_UNIQUE_ADDRESS Matrix _mat;
206 STORM_NO_UNIQUE_ADDRESS std::tuple<Coords...> _coords;
207
208public:
209
210 /// @brief Construct a matrix part view.
211 constexpr MatrixPartView(Matrix mat, Coords... coords)
212 : _mat{std::move(mat)}, _coords{std::move(coords)...} {
213 const auto check_coords = [&](auto... extents) {
214 _check_coords(extents...);
215 };
216 std::apply(check_coords, _mat.shape());
217 }
218
219 /// @brief Get the matrix shape.
220 constexpr auto shape() const {
221 const auto map_extents = [&](auto... extents) {
222 return _map_extents(extents...);
223 };
224 return std::apply(map_extents, _mat.shape());
225 }
226
227 /// @brief Get the matrix element at @p indices.
228 /// @{
229 template<class... Indices>
230 requires compatible_matrix_indices_v<MatrixPartView, Indices...>
231 constexpr decltype(auto) operator()(Indices... indices) {
232 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
233 return std::apply(_mat, _map_indices(indices...));
234 }
235 template<class... Indices>
236 requires compatible_matrix_indices_v<MatrixPartView, Indices...>
237 constexpr decltype(auto) operator()(Indices... indices) const {
238 STORM_ASSERT(in_range(shape(), indices...), "Indices are out of range!");
239 return std::apply(_mat, _map_indices(indices...));
240 }
241 /// @}
242
243private:
244
245 static constexpr size_t _LastCoordIndex = sizeof...(Coords) - 1;
246
247 // Check that coords are in range.
248 template<size_t CoordIndex = 0>
249 constexpr void _check_coords(auto... extents) const {
250 const auto recure = [&](auto... remaining) {
251 if constexpr (CoordIndex < _LastCoordIndex) {
252 _check_coords<CoordIndex + 1>(remaining...);
253 }
254 };
255
256 const auto& coord = std::get<CoordIndex>(_coords);
257 using Coord = std::remove_cvref_t<decltype(coord)>;
258 if constexpr (index_coord<Coord>) {
259 [&](auto current_extent, auto... rest_extents) {
260 const auto extent_sz = static_cast<size_t>(current_extent);
261 STORM_ASSERT(static_cast<size_t>(coord) < extent_sz,
262 "Index-like coordinate is out of range!");
263 recure(rest_extents...);
264 }(extents...);
265 } else if constexpr (new_axis_coord<Coord>) {
266 recure(extents...);
267 } else if constexpr (range_coord<Coord>) {
268 [&](auto current_extent, auto... rest_extents) {
269 const auto max_coord_element_sz = *std::ranges::max_element(
270 coord, {}, [](auto value) { return static_cast<size_t>(value); });
271 const auto extent_sz = static_cast<size_t>(current_extent);
272 STORM_ASSERT(max_coord_element_sz < extent_sz,
273 "Range-like coordinate maximum element is out of range!");
274 recure(rest_extents...);
275 }(extents...);
276 } else if constexpr (mask_coord<Coord>) {
277 const auto check_mask = [&](auto current_extents, auto... rest_extents) {
278 STORM_ASSERT(coord.shape() == current_extents,
279 "Mask-like coordinate shape is mismatched!");
280 recure(rest_extents...);
281 };
282 constexpr size_t CoordRank = matrix_rank_v<Coord>;
283 pack_n<CoordRank>(check_mask, extents...);
284 }
285 }
286
287 // Map the original matrix extents to the view extents.
288 template<size_t CoordIndex = 0>
289 constexpr auto _map_extents(auto... extents) const {
290 const auto recure = [&](auto done, auto... remaining) {
291 const auto _done = to_tuple(done);
292 if constexpr (CoordIndex == _LastCoordIndex) {
293 return _done;
294 } else {
295 const auto _remaining = _map_extents<CoordIndex + 1>(remaining...);
296 return std::tuple_cat(_done, _remaining);
297 }
298 };
299
300 const auto& coord = std::get<CoordIndex>(_coords);
301 using Coord = std::remove_cvref_t<decltype(coord)>;
302 if constexpr (index_coord<Coord>) {
303 return [&](auto /*current_extent*/, auto... rest_extents) {
304 return recure(std::tuple{}, rest_extents...);
305 }(extents...);
306 } else if constexpr (placeholder_coord<Coord>) {
307 return recure(extents...);
308 } else if constexpr (new_axis_coord<Coord>) {
309 return recure(fixed_extent_t<1>{}, extents...);
310 } else if constexpr (range_coord<Coord>) {
311 return [&](auto /*current_extent*/, auto... rest_extents) {
312 return recure(std::ranges::size(coord), rest_extents...);
313 }(extents...);
314 } else if constexpr (mask_coord<Coord>) {
315 const auto map_mask = [&](auto... rest_extents) {
316 return recure(std::ranges::size(coord.true_indices()), rest_extents...);
317 };
318 constexpr size_t CoordRank = matrix_rank_v<Coord>;
319 return drop_n<CoordRank>(map_mask, extents...);
320 } else {
321 static_assert(detail::always_false<Coord>, "Unexpected coordinate type!");
322 }
323 }
324
325 // Map the view indices to the original matrix indices.
326 template<size_t CoordIndex = 0>
327 constexpr auto _map_indices(auto... indices) const {
328 const auto recure = [&](auto done, auto... remaining) {
329 const auto _done = to_tuple(done);
330 if constexpr (CoordIndex == _LastCoordIndex) {
331 return _done;
332 } else {
333 const auto _remaining = _map_indices<CoordIndex + 1>(remaining...);
334 return std::tuple_cat(_done, _remaining);
335 }
336 };
337
338 const auto& coord = std::get<CoordIndex>(_coords);
339 using Coord = std::remove_cvref_t<decltype(coord)>;
340 if constexpr (index_coord<Coord>) {
341 return recure(coord, indices...);
342 } else if constexpr (placeholder_coord<Coord>) {
343 return recure(indices...);
344 } else if constexpr (new_axis_coord<Coord>) {
345 return [&](auto /*current_index*/, auto... rest_indices) {
346 return recure(std::tuple{}, rest_indices...);
347 }(indices...);
348 } else if constexpr (range_coord<Coord>) {
349 return [&](auto current_index, auto... rest_indices) {
350 return recure(coord[current_index], rest_indices...);
351 }(indices...);
352 } else if constexpr (mask_coord<Coord>) {
353 return [&](auto current_index, auto... rest_indices) {
354 return recure(coord.true_indices()[current_index], rest_indices...);
355 }(indices...);
356 } else {
357 static_assert(detail::always_false<Coord>, "Unexpected coordinate type!");
358 }
359 }
360
361}; // class MatrixPartView

369/// @brief Take a part of the matrix @p mat at coordinates @p coords.
370template<viewable_matrix Matrix, viewable_coord... Coords>
371 requires (matrix_rank_v<Matrix> >= coords_rank_v<Coords...>)
372constexpr auto at(Matrix&& mat, Coords&&... coords) {
373 constexpr size_t Delta = matrix_rank_v<Matrix> - coords_rank_v<Coords...>;
374 if constexpr (Delta == 0) {
375 if constexpr ((index_coord<Coords> && ...)) {
376 return mat(std::forward<Coords>(coords)...);
377 } else {
378 return MatrixPartView{std::forward<Matrix>(mat),
379 std::forward<Coords>(coords)...};
380 }
381 } else {
382 const auto make_view = [&](auto... _s) {
383 return MatrixPartView{std::forward<Matrix>(mat),
384 std::forward<Coords>(coords)..., _s...};
385 };
386 return n_times<Delta>(make_view, _);
387 }
